332double ErrorContext::calculate_attack_confidence() const {
333 // This assumes mutex is already held by caller
334
335 if (error_events_.empty()) {
336 return 0.0;
337 }
338
339 double confidence = 0.0;
340
341 // Factor 1: Rate of security-relevant errors
342 auto recent_security_errors = 0;
343 auto cutoff_time = std::chrono::steady_clock::now() - std::chrono::minutes(5);
344
345 for (const auto& event : error_events_) {
346 if (event.timestamp >= cutoff_time && event.is_security_relevant) {
347 recent_security_errors++;
348 }
349 }
350
351 if (recent_security_errors > 5) {
352 confidence += 0.4; // High rate of security errors
353 } else if (recent_security_errors > 2) {
354 confidence += 0.2; // Moderate rate
355 } else if (recent_security_errors > 0) {
356 confidence += 0.1; // Any security error adds some confidence
357 }
358
359 // Factor 2: Types of security errors
360 if (security_metrics_.authentication_failures > 3) {
361 confidence += 0.3;
362 } else if (security_metrics_.authentication_failures > 0) {
363 confidence += 0.15; // Any auth failure adds some confidence
364 }
365 if (security_metrics_.replay_detections > 0) {
366 confidence += 0.2;
367 }
368 if (security_metrics_.tampering_detections > 0) {
369 confidence += 0.3;
370 }
371
372 // Factor 3: Pattern consistency
373 if (security_metrics_.authentication_failures > 0 &&
374 security_metrics_.record_integrity_failures > 0) {
375 confidence += 0.2; // Multiple attack vectors
376 }
377
378 return std::min(confidence, 1.0);
379}
