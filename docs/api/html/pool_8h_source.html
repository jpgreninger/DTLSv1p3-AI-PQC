174 pointer allocate(size_type n) {

176
177 size_t bytes = n * sizeof(T);
178 auto& pool = GlobalPoolManager::instance().get_pool(
179 next_power_of_two(bytes));
180
181 auto buffer = pool.acquire();
182 if (!buffer || buffer->capacity() < bytes) {
183 throw std::bad_alloc();

185
186 // Store buffer pointer for deallocation
187 void* ptr = buffer->mutable_data();
188 buffer_map_[ptr] = std::move(buffer);

190 return static_cast<pointer>(ptr);
191 }

193 void deallocate(pointer p, size_type n) noexcept {
194 if (!p) return;
195
196 auto it = buffer_map_.find(p);
197 if (it != buffer_map_.end()) {
198 size_t bytes = n * sizeof(T);
199 auto& pool = GlobalPoolManager::instance().get_pool(
200 next_power_of_two(bytes));
201 pool.release(std::move(it->second));
202 buffer_map_.erase(it);
203 }
204 }

218 n--;
