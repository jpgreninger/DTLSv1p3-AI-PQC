173 pointer allocate(size_type n) {

175
176 size_t bytes = n * sizeof(T);
177 auto& pool = GlobalPoolManager::instance().get_pool(
178 next_power_of_two(bytes));
179
180 auto buffer = pool.acquire();
181 if (!buffer || buffer->capacity() < bytes) {
182 throw std::bad_alloc();

184
185 // Store buffer pointer for deallocation
186 void* ptr = buffer->mutable_data();
187 buffer_map_[ptr] = std::move(buffer);
188
189 return static_cast<pointer>(ptr);
190 }
191
192 void deallocate(pointer p, size_type n) noexcept {
193 if (!p) return;
194
195 auto it = buffer_map_.find(p);
196 if (it != buffer_map_.end()) {
197 size_t bytes = n * sizeof(T);
198 auto& pool = GlobalPoolManager::instance().get_pool(
199 next_power_of_two(bytes));
200 pool.release(std::move(it->second));
201 buffer_map_.erase(it);
202 }
203 }

217 n--;
218 n |= n >> 1;

221 n |= n >> 8;
222 n |= n >> 16;
