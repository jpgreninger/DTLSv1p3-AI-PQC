97Result<std::vector<uint8_t>> hkdf_expand(
98 CryptoProvider& provider,
99 HashAlgorithm hash,
100 const std::vector<uint8_t>& pseudo_random_key,
101 const std::vector<uint8_t>& info,
102 size_t output_length) {
103
104 DTLS_CRYPTO_TIMER("hkdf_expand");
105
106 size_t hash_len = get_hash_output_length(hash);
107 size_t n = (output_length + hash_len - 1) / hash_len; // Ceiling division
108
109 if (n > 255) {
110 return Result<std::vector<uint8_t>>(DTLSError::INVALID_PARAMETER);
111 }
112
113 std::vector<uint8_t> output;
114 output.reserve(output_length);
115
116 std::vector<uint8_t> t_prev;
117
118 for (size_t i = 1; i <= n; ++i) {
119 HMACParams params;
120 params.key = pseudo_random_key;
121 params.algorithm = hash;
122
123 // T(i) = HMAC-Hash(PRK, T(i-1) | info | i)
124 params.data = t_prev;
125 params.data.insert(params.data.end(), info.begin(), info.end());
126 params.data.push_back(static_cast<uint8_t>(i));
127
128 auto t_result = provider.compute_hmac(params);
129 if (!t_result) {
130 return t_result;
131 }
132
133 t_prev = *t_result;
134 output.insert(output.end(), t_prev.begin(), t_prev.end());
135 }
136
137 // Truncate to desired length
138 output.resize(output_length);
139 return Result<std::vector<uint8_t>>(std::move(output));
140}

142Result<std::vector<uint8_t>> hkdf_expand_label(
143 CryptoProvider& provider,
144 HashAlgorithm hash,
145 const std::vector<uint8_t>& secret,
146 const std::string& label,
147 const std::vector<uint8_t>& context,
148 size_t length) {
149
150 DTLS_CRYPTO_TIMER("hkdf_expand_label");
151
152 // HkdfLabel {
153 // uint16 length = Length;
154 // opaque label<7..255> = "tls13 " + Label;
155 // opaque context<0..255> = Context;
156 // }
157
158 std::vector<uint8_t> hkdf_label;
159
160 // Length (2 bytes, big-endian)
161 hkdf_label.push_back(static_cast<uint8_t>((length >> 8) & 0xFF));
162 hkdf_label.push_back(static_cast<uint8_t>(length & 0xFF));
163
164 // Label with "tls13 " prefix
165 std::string prefixed_label = "tls13 " + label;
166 if (prefixed_label.length() > 255) {
167 return Result<std::vector<uint8_t>>(DTLSError::INVALID_PARAMETER);
168 }
169
170 hkdf_label.push_back(static_cast<uint8_t>(prefixed_label.length()));
171 hkdf_label.insert(hkdf_label.end(), prefixed_label.begin(), prefixed_label.end());
172
173 // Context
174 if (context.size() > 255) {
175 return Result<std::vector<uint8_t>>(DTLSError::INVALID_PARAMETER);
176 }
177
178 hkdf_label.push_back(static_cast<uint8_t>(context.size()));
179 hkdf_label.insert(hkdf_label.end(), context.begin(), context.end());
180
181 return hkdf_expand(provider, hash, secret, hkdf_label, length);
182}

328Result<KeySchedule> update_traffic_keys(
329 CryptoProvider& provider,
330 const CipherSpec& cipher_spec,
331 const KeySchedule& current_keys) {
332
333 DTLS_CRYPTO_TIMER("update_traffic_keys");
334
335 KeySchedule updated_keys = current_keys;
336
337 // RFC 9147 Section 4.6.3: Key Update
338 // application_traffic_secret_N+1 = HKDF-Expand-Label(application_traffic_secret_N, "traffic upd", "", Hash.length)
339
340 // Update client write key using current client write key as the base secret
341 auto updated_client_secret = hkdf_expand_label(
342 provider, cipher_spec.hash_algorithm, current_keys.client_write_key,
343 "traffic upd", {}, cipher_spec.hash_length);
344
345 if (!updated_client_secret) {
346 return Result<KeySchedule>(updated_client_secret.error());
347 }
348
349 // Update server write key using current server write key as the base secret
350 auto updated_server_secret = hkdf_expand_label(
351 provider, cipher_spec.hash_algorithm, current_keys.server_write_key,
352 "traffic upd", {}, cipher_spec.hash_length);
353
354 if (!updated_server_secret) {
355 return Result<KeySchedule>(updated_server_secret.error());
356 }
357
358 // Derive new client write key from updated secret
359 auto new_client_key = hkdf_expand_label(
360 provider, cipher_spec.hash_algorithm, *updated_client_secret,
361 constants::HKDF_LABEL_KEY, {}, cipher_spec.key_length);
362
363 if (!new_client_key) {
364 return Result<KeySchedule>(new_client_key.error());
365 }
366
367 // Derive new client IV from updated secret
368 auto new_client_iv = hkdf_expand_label(
369 provider, cipher_spec.hash_algorithm, *updated_client_secret,
370 constants::HKDF_LABEL_IV, {}, cipher_spec.iv_length);
371
372 if (!new_client_iv) {
373 return Result<KeySchedule>(new_client_iv.error());
374 }
375
376 // Derive new server write key from updated secret
377 auto new_server_key = hkdf_expand_label(
378 provider, cipher_spec.hash_algorithm, *updated_server_secret,
379 constants::HKDF_LABEL_KEY, {}, cipher_spec.key_length);
380
381 if (!new_server_key) {
382 return Result<KeySchedule>(new_server_key.error());
383 }
384
385 // Derive new server IV from updated secret
386 auto new_server_iv = hkdf_expand_label(
387 provider, cipher_spec.hash_algorithm, *updated_server_secret,
388 constants::HKDF_LABEL_IV, {}, cipher_spec.iv_length);
389
390 if (!new_server_iv) {
391 return Result<KeySchedule>(new_server_iv.error());
392 }
393
394 // Update sequence number keys
395 auto new_client_sn_key = derive_sequence_number_key(
396 provider, *updated_client_secret, cipher_spec.hash_algorithm);
397
398 if (!new_client_sn_key) {
399 return Result<KeySchedule>(new_client_sn_key.error());
400 }
401
402 auto new_server_sn_key = derive_sequence_number_key(
403 provider, *updated_server_secret, cipher_spec.hash_algorithm);
404
405 if (!new_server_sn_key) {
406 return Result<KeySchedule>(new_server_sn_key.error());
407 }
408
409 // Update the key schedule with new keys
410 updated_keys.client_write_key = std::move(*new_client_key);
411 updated_keys.server_write_key = std::move(*new_server_key);
412 updated_keys.client_write_iv = std::move(*new_client_iv);

414 updated_keys.client_sequence_number_key = std::move(*new_client_sn_key);
415 updated_keys.server_sequence_number_key = std::move(*new_server_sn_key);
416
417 // Increment epoch for key update
418 updated_keys.epoch++;
419
420 return Result<KeySchedule>(std::move(updated_keys));
421}

501// Nonce construction
502std::vector<uint8_t> construct_aead_nonce(
503 const std::vector<uint8_t>& write_iv,
504 SequenceNumber sequence_number) {
505
506 std::vector<uint8_t> nonce = write_iv;
507
508 // XOR sequence number with the last 8 bytes of the IV
509 size_t offset = nonce.size() - 8;
510 for (size_t i = 0; i < 8; ++i) {
511 nonce[offset + i] ^= static_cast<uint8_t>((sequence_number >> (8 * (7 - i))) & 0xFF);
512 }
513
514 return nonce;
515}

517std::vector<uint8_t> construct_aead_additional_data(
518 const ConnectionID& connection_id,
519 Epoch epoch,
520 SequenceNumber sequence_number,
521 ContentType content_type,
522 ProtocolVersion version,
523 uint16_t length) {
524
525 std::vector<uint8_t> aad;
526
527 // Connection ID (variable length)
528 aad.insert(aad.end(), connection_id.begin(), connection_id.end());
529
530 // Epoch (2 bytes)
531 aad.push_back(static_cast<uint8_t>((epoch >> 8) & 0xFF));
532 aad.push_back(static_cast<uint8_t>(epoch & 0xFF));
533
534 // Sequence number (6 bytes, 48-bit)
535 for (int i = 5; i >= 0; --i) {
536 aad.push_back(static_cast<uint8_t>((sequence_number >> (8 * i)) & 0xFF));
537 }
538
539 // Content type (1 byte)
540 aad.push_back(static_cast<uint8_t>(content_type));
541
542 // Version (2 bytes)
543 aad.push_back(static_cast<uint8_t>((version >> 8) & 0xFF));
544 aad.push_back(static_cast<uint8_t>(version & 0xFF));
545
546 // Length (2 bytes)
547 aad.push_back(static_cast<uint8_t>((length >> 8) & 0xFF));
548 aad.push_back(static_cast<uint8_t>(length & 0xFF));

779// Timing-safe comparison
780bool constant_time_compare(const std::vector<uint8_t>& a, const std::vector<uint8_t>& b) {
781 // Use OpenSSL's CRYPTO_memcmp if available for truly constant-time comparison
782#ifdef DTLS_USE_OPENSSL
783 if (a.size() == b.size() && !a.empty()) {
784 return CRYPTO_memcmp(a.data(), b.data(), a.size()) == 0;
785 }
786 // For different sizes, still use constant time approach
787 if (a.size() == b.size() && a.empty()) {
788 return true; // Both empty
789 }
790#endif
791
792 // Fallback constant-time implementation for all cases
793 // This ensures consistent behavior even when sizes differ
794
795 // Normalize to fixed comparison size to prevent timing leaks based on input size
796 static constexpr size_t FIXED_COMPARE_SIZE = 256;
797
798 volatile uint8_t result = 0;
799
800 // Size comparison in constant time
801 volatile size_t size_a = a.size();
802 volatile size_t size_b = b.size();
803 volatile uint8_t size_mismatch = 0;
804
805 if (size_a != size_b) {
806 size_mismatch = 1;
807 }
808
809 // Always perform exactly FIXED_COMPARE_SIZE comparisons for constant timing
810 for (size_t i = 0; i < FIXED_COMPARE_SIZE; ++i) {
811 volatile uint8_t byte_a = (i < size_a) ? a[i] : 0;
812 volatile uint8_t byte_b = (i < size_b) ? b[i] : 0;
813 volatile uint8_t diff = byte_a ^ byte_b;
814 result |= diff;
815
816 // Add some constant work to normalize timing further
817 volatile uint8_t dummy = byte_a & byte_b;
818 dummy ^= diff;
819 (void)dummy;
820 }
821
822 // Include size mismatch in result
823 result |= size_mismatch;
824
825 // Additional fixed-time operations to ensure consistent timing
826 volatile uint64_t normalize_ops = 0;
827 for (size_t i = 0; i < 128; ++i) {
828 normalize_ops ^= static_cast<uint64_t>(result) << (i % 8);
829 normalize_ops += i;
830 normalize_ops ^= normalize_ops >> 1;
831 }
832
833 // Multiple memory barriers to prevent any reordering
834 std::atomic_thread_fence(std::memory_order_acq_rel);
835 std::atomic_thread_fence(std::memory_order_seq_cst);
836
837 return result == 0;
838}

1214Result<uint64_t> encrypt_sequence_number(
1215 CryptoProvider& provider,
1216 uint64_t sequence_number,
1217 const std::vector<uint8_t>& sequence_number_key,
1218 const std::vector<uint8_t>& ciphertext,
1219 AEADCipher cipher_type) {
1220
1221 DTLS_CRYPTO_TIMER("encrypt_sequence_number");
1222
1223 if (sequence_number_key.empty()) {
1224 return make_error<uint64_t>(DTLSError::INVALID_PARAMETER);
1225 }
1226
1227 if (ciphertext.size() < 16) {
1228 return make_error<uint64_t>(DTLSError::INVALID_PARAMETER);
1229 }
1230
1231 std::vector<uint8_t> mask(16);
1232
1233 // Generate mask according to RFC 9147 Section 4.2.3
1234 // Simplified implementation using HMAC for mask generation instead of block cipher
1235 switch (cipher_type) {
1236 case AEADCipher::AES_128_GCM:
1237 case AEADCipher::AES_256_GCM:
1238 case AEADCipher::AES_128_CCM:
1239 case AEADCipher::AES_128_CCM_8: {
1240 // For AES-based AEAD: Use HMAC-SHA256 for mask generation
1241 // This is a simplified but secure implementation
1242 if (sequence_number_key.size() < 16) {
1243 return make_error<uint64_t>(DTLSError::INVALID_PARAMETER);
1244 }
1245
1246 // Use HMAC-SHA256(sn_key, ciphertext[0:15]) for mask generation
1247 std::vector<uint8_t> block(ciphertext.begin(), ciphertext.begin() + 16);
1248
1249 HMACParams hmac_params;
1250 hmac_params.key = sequence_number_key;
1251 hmac_params.data = block;
1252 hmac_params.algorithm = HashAlgorithm::SHA256;
1253
1254 auto mask_result = provider.compute_hmac(hmac_params);
1255 if (!mask_result.is_success()) {
1256 return make_error<uint64_t>(mask_result.error());
1257 }
1258
1259 auto hmac_output = mask_result.value();
1260 if (hmac_output.size() < 16) {
1261 return make_error<uint64_t>(DTLSError::INTERNAL_ERROR);
1262 }
1263
1264 // Use first 16 bytes of HMAC output as mask
1265 std::copy(hmac_output.begin(), hmac_output.begin() + 16, mask.begin());
1266 break;
1267 }
1268
1269 case AEADCipher::CHACHA20_POLY1305: {
1270 // For ChaCha20-based AEAD: Use HMAC-SHA256 for mask generation (simplified)
1271 if (sequence_number_key.size() < 16) {
1272 return make_error<uint64_t>(DTLSError::INVALID_PARAMETER);
1273 }
1274
1275 // Use HMAC-SHA256(sn_key, ciphertext[0:15]) for mask generation
1276 std::vector<uint8_t> block(ciphertext.begin(), ciphertext.begin() + 16);
1277
1278 HMACParams hmac_params;
1279 hmac_params.key = sequence_number_key;
1280 hmac_params.data = block;
1281 hmac_params.algorithm = HashAlgorithm::SHA256;
1282
1283 auto mask_result = provider.compute_hmac(hmac_params);
1284 if (!mask_result.is_success()) {
1285 return make_error<uint64_t>(mask_result.error());
1286 }
1287
1288 auto hmac_output = mask_result.value();
1289 if (hmac_output.size() < 16) {
1290 return make_error<uint64_t>(DTLSError::INTERNAL_ERROR);
1291 }
1292
1293 // Use first 16 bytes of HMAC output as mask
1294 std::copy(hmac_output.begin(), hmac_output.begin() + 16, mask.begin());
1295 break;
1296 }
1297
1298 default:
1299 return make_error<uint64_t>(DTLSError::CIPHER_SUITE_NOT_SUPPORTED);
1300 }
1301
1302 // XOR the 48-bit sequence number with the first 6 bytes of the mask
1303 uint64_t encrypted_seq = sequence_number;
1304 for (int i = 0; i < 6; ++i) {
1305 uint8_t seq_byte = (sequence_number >> (8 * (5 - i))) & 0xFF;
1306 uint8_t masked_byte = seq_byte ^ mask[i];
1307
1308 // Clear the corresponding byte in encrypted_seq and set the masked byte
1309 encrypted_seq &= ~(0xFFULL << (8 * (5 - i)));
1310 encrypted_seq |= (static_cast<uint64_t>(masked_byte) << (8 * (5 - i)));
1311 }
1312
1313 // Ensure result is 48-bit
1314 encrypted_seq &= 0xFFFFFFFFFFFFULL;
1315
1316 return make_result(encrypted_seq);
1317}

1319Result<uint64_t> decrypt_sequence_number(
1320 CryptoProvider& provider,
1321 uint64_t encrypted_sequence_number,
1322 const std::vector<uint8_t>& sequence_number_key,
1323 const std::vector<uint8_t>& ciphertext,
1324 AEADCipher cipher_type) {
1325
1326 DTLS_CRYPTO_TIMER("decrypt_sequence_number");
1327
1328 if (sequence_number_key.empty()) {
1329 return make_error<uint64_t>(DTLSError::INVALID_PARAMETER);
1330 }
1331
1332 if (ciphertext.size() < 16) {
1333 return make_error<uint64_t>(DTLSError::INVALID_PARAMETER);
1334 }
1335
1336 std::vector<uint8_t> mask(16);
1337
1338 // Generate the same mask used for encryption according to RFC 9147 Section 4.2.3
1339 switch (cipher_type) {
1340 case AEADCipher::AES_128_GCM:
1341 case AEADCipher::AES_256_GCM:
1342 case AEADCipher::AES_128_CCM:
1343 case AEADCipher::AES_128_CCM_8: {
1344 // For AES-based AEAD: Use HMAC-SHA256 for mask generation
1345 // This matches the encrypt function implementation
1346 if (sequence_number_key.size() < 16) {
1347 return make_error<uint64_t>(DTLSError::INVALID_PARAMETER);
1348 }

1351 std::vector<uint8_t> block(ciphertext.begin(), ciphertext.begin() + 16);

1361 }

1364 if (hmac_output.size() < 16) {
1365 return make_error<uint64_t>(DTLSError::INTERNAL_ERROR);
1366 }
1367
1368 // Use first 16 bytes of HMAC output as mask
1369 std::copy(hmac_output.begin(), hmac_output.begin() + 16, mask.begin());
1370 break;
1371 }
1372
1373 case AEADCipher::CHACHA20_POLY1305: {
1374 // For ChaCha20-based AEAD: Use HMAC-SHA256 for mask generation (simplified)
1375 if (sequence_number_key.size() < 16) {
1376 return make_error<uint64_t>(DTLSError::INVALID_PARAMETER);
1377 }
1378
1379 // Use HMAC-SHA256(sn_key, ciphertext[0:15]) for mask generation
1380 std::vector<uint8_t> block(ciphertext.begin(), ciphertext.begin() + 16);
1381
1382 HMACParams hmac_params;
1383 hmac_params.key = sequence_number_key;
1384 hmac_params.data = block;
1385 hmac_params.algorithm = HashAlgorithm::SHA256;
1386
1387 auto mask_result = provider.compute_hmac(hmac_params);
1388 if (!mask_result.is_success()) {
1389 return make_error<uint64_t>(mask_result.error());
1390 }
1391
1392 auto hmac_output = mask_result.value();
1393 if (hmac_output.size() < 16) {
1394 return make_error<uint64_t>(DTLSError::INTERNAL_ERROR);
1395 }
1396
1397 // Use first 16 bytes of HMAC output as mask
1398 std::copy(hmac_output.begin(), hmac_output.begin() + 16, mask.begin());
1399 break;
1400 }
1401
1402 default:
1403 return make_error<uint64_t>(DTLSError::CIPHER_SUITE_NOT_SUPPORTED);
1404 }
1405
1406 // XOR the encrypted 48-bit sequence number with the first 6 bytes of the mask to decrypt
1407 uint64_t decrypted_seq = encrypted_sequence_number;
1408 for (int i = 0; i < 6; ++i) {
1409 uint8_t encrypted_byte = (encrypted_sequence_number >> (8 * (5 - i))) & 0xFF;
1410 uint8_t decrypted_byte = encrypted_byte ^ mask[i];
1411
1412 // Clear the corresponding byte in decrypted_seq and set the decrypted byte
1413 decrypted_seq &= ~(0xFFULL << (8 * (5 - i)));
1414 decrypted_seq |= (static_cast<uint64_t>(decrypted_byte) << (8 * (5 - i)));
1415 }
1416
1417 // Ensure result is 48-bit
1418 decrypted_seq &= 0xFFFFFFFFFFFFULL;
1419
1420 return make_result(decrypted_seq);
1421}

1528// ASN.1 signature format validation for ECDSA
1529Result<bool> validate_ecdsa_asn1_signature(
1530 const std::vector<uint8_t>& signature,
1531 const PublicKey& public_key) {
1532
1533 // Validate input
1534 if (signature.empty()) {
1535 return Result<bool>(DTLSError::INVALID_PARAMETER);
1536 }
1537
1538 // ECDSA signatures should be at least 8 bytes (minimal ASN.1 structure)
1539 // and no more than 150 bytes (P-521 with maximum ASN.1 overhead)
1540 if (signature.size() < 8 || signature.size() > 150) {
1541 return Result<bool>(false);
1542 }
1543
1544 // Basic ASN.1 DER validation
1545 // ECDSA signature format: SEQUENCE { r INTEGER, s INTEGER }
1546
1547 if (signature[0] != 0x30) { // SEQUENCE tag
1548 return Result<bool>(false);
1549 }
1550
1551 // Check length encoding
1552 size_t seq_length_pos = 1;
1553 size_t seq_length;
1554
1555 if (signature[1] & 0x80) {
1556 // Long form length
1557 size_t length_bytes = signature[1] & 0x7F;
1558 if (length_bytes == 0 || length_bytes > 4 || signature.size() < 2 + length_bytes) {
1559 return Result<bool>(false);
1560 }
1561
1562 seq_length = 0;
1563 for (size_t i = 0; i < length_bytes; i++) {
1564 seq_length = (seq_length << 8) | signature[2 + i];
1565 }
1566 seq_length_pos = 2 + length_bytes;
1567 } else {
1568 // Short form length
1569 seq_length = signature[1];
1570 seq_length_pos = 2;
1571 }
1572
1573 // Verify total length matches
1574 if (seq_length_pos + seq_length != signature.size()) {
1575 return Result<bool>(false);
1576 }
1577
1578 // Parse r INTEGER
1579 if (seq_length_pos >= signature.size() || signature[seq_length_pos] != 0x02) {
1580 return Result<bool>(false);
1581 }
1582
1583 size_t r_pos = seq_length_pos + 1;
1584 if (r_pos >= signature.size()) {
1585 return Result<bool>(false);
1586 }
1587
1588 size_t r_length = signature[r_pos];
1589 if (r_length == 0 || r_pos + 1 + r_length > signature.size()) {
1590 return Result<bool>(false);
1591 }
1592
1593 // Check r value (should not be zero, should not have unnecessary leading zeros)
1594 size_t r_value_pos = r_pos + 1;
1595 if (signature[r_value_pos] == 0x00 && r_length > 1 &&
1596 (signature[r_value_pos + 1] & 0x80) == 0) {
1597 return Result<bool>(false); // Unnecessary leading zero
1598 }
1599
1600 // Parse s INTEGER
1601 size_t s_pos = r_value_pos + r_length;
1602 if (s_pos >= signature.size() || signature[s_pos] != 0x02) {
1603 return Result<bool>(false);
1604 }
1605
1606 s_pos++;
1607 if (s_pos >= signature.size()) {
1608 return Result<bool>(false);
1609 }
1610
1611 size_t s_length = signature[s_pos];
1612 if (s_length == 0 || s_pos + 1 + s_length > signature.size()) {
1613 return Result<bool>(false);
1614 }
1615
1616 // Check s value (should not be zero, should not have unnecessary leading zeros)
1617 size_t s_value_pos = s_pos + 1;
1618 if (signature[s_value_pos] == 0x00 && s_length > 1 &&
1619 (signature[s_value_pos + 1] & 0x80) == 0) {
1620 return Result<bool>(false); // Unnecessary leading zero
1621 }
1622
1623 // Verify we've consumed the entire signature
1624 if (s_value_pos + s_length != signature.size()) {
1625 return Result<bool>(false);
1626 }
1627
1628 return Result<bool>(true);
1629}
