62 std::lock_guard<std::mutex> lock(mutex_);
63
64 if (!initialized_) {
65 return Result<void>(DTLSError::INVALID_STATE);
66 }
67
68 stats_.messages_received++;
69
70 uint32_t message_seq = message.header().message_seq;
71
72 // Handle ACK messages specially
73 if (message.message_type() == HandshakeType::ACK) {
74 if (message.holds<ACK>()) {
75 const ACK& ack_message = message.get<ACK>();
76 auto ack_result = handle_ack_message(ack_message);
77 if (!ack_result.is_success()) {
78 return ack_result;
79 }
80
81 stats_.acks_received++;
82 fire_event(HandshakeEvent::ACK_RECEIVED, {});
83 }
84 return Result<void>();
85 }
86
87 // Track received sequence numbers
88 received_sequences_.push_back(message_seq);
89
90 // Check for out-of-order delivery
91 if (message_seq != expected_receive_sequence_) {
92 out_of_order_sequences_.push_back(message_seq);
93 } else {
94 // Update expected sequence for in-order message
95 expected_receive_sequence_ = message_seq + 1;
96
97 // Check if we can advance expected sequence with out-of-order messages
98 std::sort(out_of_order_sequences_.begin(), out_of_order_sequences_.end());
99 while (!out_of_order_sequences_.empty() &&
100 out_of_order_sequences_[0] == expected_receive_sequence_) {
101 expected_receive_sequence_++;
102 out_of_order_sequences_.erase(out_of_order_sequences_.begin());
103 }
104 }
105
106 fire_event(HandshakeEvent::MESSAGE_RECEIVED, {});
107
108 // Generate ACK if configured and appropriate
109 if (config_.enable_ack_processing) {
110 auto ack_result = generate_ack_for_received_messages();
111 if (!ack_result.is_success()) {
112 return ack_result;

166Result<void> HandshakeManager::process_timeouts() {
167 std::lock_guard<std::mutex> lock(mutex_);
168
169 if (!initialized_) {
170 return Result<void>(DTLSError::INVALID_STATE);
171 }
172
173 auto now = std::chrono::steady_clock::now();
174 std::vector<uint32_t> timed_out_messages;
175
176 // Check for timed out messages
177 for (auto& [seq, tracked_msg] : outbound_messages_) {

196 }

200 for (uint32_t seq : reliability_timeouts) {

202 auto retrans_result = handle_retransmission(seq);

212Result<void> HandshakeManager::handle_ack_message(const ACK& ack_message) {
213 // Process ACK with reliability manager
214 auto reliability_result = reliability_manager_->process_ack_message(ack_message);
215 if (!reliability_result.is_success()) {
216 return reliability_result;
217 }
218
219 // Update local tracking based on ACK ranges
220 for (const auto& range : ack_message.ack_ranges()) {
221 for (uint32_t seq = range.start_sequence; seq <= range.end_sequence; ++seq) {
222 auto it = outbound_messages_.find(seq);
223 if (it != outbound_messages_.end()) {
224 if (!it->second.acknowledged) {
225 // Calculate RTT for this message
226 auto now = std::chrono::steady_clock::now();
227 auto rtt = std::chrono::duration_cast<std::chrono::milliseconds>(
228 now - it->second.send_time);
229
230 // Update RTO statistics
231 update_rto_statistics(rtt);
232 reliability_manager_->update_rtt(seq, rtt);
233
234 it->second.acknowledged = true;
235 }
236 }
237 }
238 }
239
240 // Clean up acknowledged messages
241 auto it = outbound_messages_.begin();
242 while (it != outbound_messages_.end()) {
243 if (it->second.acknowledged) {
244 it = outbound_messages_.erase(it);
245 } else {
246 ++it;
247 }
248 }
249
250 return Result<void>();

302Result<void> HandshakeManager::handle_retransmission(uint32_t message_sequence) {
303 auto it = outbound_messages_.find(message_sequence);
304 if (it == outbound_messages_.end() || it->second.acknowledged) {
305 return Result<void>(); // Message already acknowledged or not found
306 }
307
308 TrackedMessage& tracked_msg = it->second;
309
310 // Check retransmission limit

342void HandshakeManager::update_rto_statistics(std::chrono::milliseconds rtt) {
343 // RFC 6298 RTO calculation
344 const double alpha = 0.125; // SRTT gain
345 const double beta = 0.25; // RTTVAR gain
346 const int k = 4; // RTTVAR multiplier
347
348 if (srtt_.count() == 0) {
349 // First measurement
350 srtt_ = rtt;
351 rttvar_ = rtt / 2;
352 } else {
353 // Subsequent measurements
354 auto rttvar_new = std::chrono::milliseconds(static_cast<long long>(
355 (1.0 - beta) * rttvar_.count() + beta * std::abs(srtt_.count() - rtt.count())
356 ));
357
358 auto srtt_new = std::chrono::milliseconds(static_cast<long long>(
359 (1.0 - alpha) * srtt_.count() + alpha * rtt.count()
360 ));
361
362 srtt_ = srtt_new;
363 rttvar_ = rttvar_new;
364 }
365
366 // Calculate new RTO
367 rto_ = std::max(
368 std::chrono::milliseconds(srtt_.count() + k * rttvar_.count()),
369 std::chrono::milliseconds(1000) // Minimum 1 second
370 );
371
372 rto_ = std::min(rto_, config_.max_timeout);
373
374 stats_.current_rto = rto_;
375}
