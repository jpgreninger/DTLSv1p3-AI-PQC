38std::unique_ptr<ZeroCopyBuffer> BufferPool::acquire() {
39 std::lock_guard<std::mutex> lock(pool_mutex_);
40
41 total_allocations_.fetch_add(1, std::memory_order_relaxed);
42
43 if (!available_buffers_.empty()) {
44 auto buffer = std::move(available_buffers_.front());
45 available_buffers_.pop();
46
47 // Update peak usage statistics
48 size_t current_usage = pool_size_.load() - available_buffers_.size();
49 size_t current_peak = peak_usage_.load();
50 while (current_usage > current_peak &&
51 !peak_usage_.compare_exchange_weak(current_peak, current_usage)) {
52 // Retry if another thread updated peak_usage_
53 }
54
55 // Clear the buffer for reuse
56 buffer->clear();
57 return buffer;
58 }
59
60 // No buffers available, try to expand pool
61 if (pool_size_.load() < max_pool_size_.load()) {
62 auto buffer = create_buffer();
63 if (buffer) {
64 pool_size_.fetch_add(1, std::memory_order_relaxed);
65 return buffer;
66 }
67 }
68
69 // Failed to allocate
70 allocation_failures_.fetch_add(1, std::memory_order_relaxed);
71 return nullptr;

74void BufferPool::release(std::unique_ptr<ZeroCopyBuffer> buffer) {
75 if (!buffer) {
76 return;
77 }
78
79 // Validate buffer belongs to this pool
80 if (!is_valid_buffer(buffer.get())) {
81 return; // Invalid buffer, don't add to pool
82 }
83
84 total_deallocations_.fetch_add(1, std::memory_order_relaxed);
85
86 std::lock_guard<std::mutex> lock(pool_mutex_);
87
88 // Check if pool is at capacity
89 if (available_buffers_.size() >= max_pool_size_.load()) {
90 // Pool is full, let buffer be destroyed
91 return;
92 }
93
94 // Secure zero the buffer before returning to pool
95 buffer->secure_zero();

99Result<void> BufferPool::expand_pool(size_t additional_buffers) {
100 if (additional_buffers == 0) {
101 return Result<void>();
102 }
103
104 size_t current_max = max_pool_size_.load();
105 size_t new_max = current_max + additional_buffers;
106
107 // Update max pool size first
108 max_pool_size_.store(new_max, std::memory_order_relaxed);
109
110 std::lock_guard<std::mutex> lock(pool_mutex_);
111
112 size_t created = 0;
113 for (size_t i = 0; i < additional_buffers; ++i) {
114 auto buffer = create_buffer();
115 if (buffer) {
116 available_buffers_.push(std::move(buffer));
117 created++;
118 } else {
119 // Failed to create buffer, break
120 break;
121 }
122 }
123
124 if (created > 0) {
125 pool_size_.fetch_add(created, std::memory_order_relaxed);
126 return Result<void>();
127 }
128
129 return Result<void>(DTLSError::OUT_OF_MEMORY);
130}

132Result<void> BufferPool::shrink_pool(size_t target_size) {
133 std::lock_guard<std::mutex> lock(pool_mutex_);
134
135 size_t current_available = available_buffers_.size();
136 if (current_available <= target_size) {
137 return Result<void>(); // Already at or below target
138 }
139
140 size_t to_remove = current_available - target_size;
141
142 for (size_t i = 0; i < to_remove && !available_buffers_.empty(); ++i) {
143 available_buffers_.pop();
