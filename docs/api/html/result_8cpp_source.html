49// Execute a function and wrap any thrown DTLSException in a Result
50template<typename F>
51auto try_execute(F&& func) -> Result<decltype(func())> {
52 try {
53 if constexpr (std::is_void_v<decltype(func())>) {
54 func();
55 return Result<void>();
56 } else {
57 return Result<decltype(func())>(func());
58 }
59 } catch (const DTLSException& e) {
60 return Result<decltype(func())>(e.dtls_error());
61 } catch (const std::exception& e) {
62 // Convert standard exceptions to internal error
63 return Result<decltype(func())>(DTLSError::INTERNAL_ERROR);
64 }
65}
