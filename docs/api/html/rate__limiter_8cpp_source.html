17bool TokenBucket::try_consume(size_t tokens) {
18 std::lock_guard<std::mutex> lock(bucket_mutex_);
19
20 refill_tokens();
21
22 // Get current token count and check if we have enough
23 size_t current = current_tokens_.load();
24 if (current >= tokens) {
25 current_tokens_.store(current - tokens);
26 return true;
27 }
28
29 return false;

44void TokenBucket::refill_tokens() {
45 auto now = std::chrono::steady_clock::now();
46 auto time_passed = std::chrono::duration_cast<std::chrono::milliseconds>(
47 now - last_refill_.load());
48
49 if (time_passed.count() > 0) {
50 size_t tokens_to_add = (time_passed.count() * tokens_per_second_) / 1000;
51 if (tokens_to_add > 0) {
52 // Read current tokens atomically, calculate new value, and store atomically
53 size_t current = current_tokens_.load();
54 size_t new_tokens = std::min(max_tokens_, current + tokens_to_add);
55 current_tokens_.store(new_tokens);
56 last_refill_.store(now);
57 }
58 }

116RateLimitResult RateLimiter::check_connection_attempt(const NetworkAddress& source_address) {
117 total_requests_++;
118
119 // Check whitelist first
120 if (config_.enable_whitelist && is_whitelisted(source_address)) {
121 allowed_requests_++;
122 return RateLimitResult::ALLOWED;
123 }
124
125 auto* source_data = get_or_create_source_data(source_address);
126 if (!source_data) {
127 denied_requests_++;
128 return RateLimitResult::RESOURCE_EXHAUSTED;
129 }
130
131 // Check if blacklisted
132 if (is_blacklisted(source_address)) {
133 denied_requests_++;
134 source_data->stats.denied_requests++;
135 return RateLimitResult::BLACKLISTED;
136 }
137
138 // Update statistics
139 source_data->stats.total_requests++;
140 source_data->stats.last_request = std::chrono::steady_clock::now();
141
142 // Check concurrent connection limit
143 if (source_data->active_connections.load() >= config_.max_concurrent_connections) {
144 denied_requests_++;
145 source_data->stats.denied_requests++;
146 record_violation(source_address, "concurrent_connection_limit");
147 return RateLimitResult::RATE_LIMITED;
148 }
149
150 // Check token bucket
151 if (!source_data->token_bucket->try_consume(1)) {
152 denied_requests_++;
153 source_data->stats.denied_requests++;
154 record_violation(source_address, "token_bucket_exhausted");
155 return RateLimitResult::RATE_LIMITED;
156 }
157
158 // Check burst detection
159 if (source_data->burst_window->add_event_and_check_burst(config_.max_burst_count)) {
160 denied_requests_++;
161 source_data->stats.denied_requests++;
162 record_violation(source_address, "burst_detected");
163 return RateLimitResult::RATE_LIMITED;
164 }
165
166 allowed_requests_++;
167 source_data->stats.allowed_requests++;
168 return RateLimitResult::ALLOWED;
169}

171RateLimitResult RateLimiter::check_handshake_attempt(const NetworkAddress& source_address) {
172 // First check connection attempt

183 // Check handshake-specific rate limit

185 denied_requests_++;

188 return RateLimitResult::RATE_LIMITED;
189 }
190
191 return RateLimitResult::ALLOWED;

209void RateLimiter::record_violation(const NetworkAddress& source_address,

335void RateLimiter::cleanup_expired_entries() {
336 auto now = std::chrono::steady_clock::now();
337
338 // Only run cleanup periodically
339 if (now - last_cleanup_.load() < cleanup_interval_) {
340 return;
341 }
342
343 std::unique_lock<std::shared_mutex> lock(sources_mutex_);
344
345 auto it = sources_.begin();
346 while (it != sources_.end()) {
347 auto& source_data = it->second;
348
349 // Remove expired blacklist entries and inactive sources
350 bool should_remove = false;
351
352 if (source_data->is_blacklisted.load() &&
353 now > source_data->blacklist_expiry.load()) {
354 source_data->is_blacklisted = false;
355 }
356
357 // Remove completely inactive sources after a long period
358 auto inactive_threshold = std::chrono::hours{24};
359 if (source_data->active_connections.load() == 0 &&
360 now - source_data->stats.last_request > inactive_threshold) {
361 should_remove = true;
362 }
363
364 if (should_remove) {
365 it = sources_.erase(it);
366 } else {
367 ++it;
368 }
369 }
370
371 last_cleanup_ = now;
372}
