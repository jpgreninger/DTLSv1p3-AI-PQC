234AttackType DoSProtectionEngine::analyze_ip_behavior(const IPResourceUsage& usage) const {
235 // Memory exhaustion attack detection
236 if (usage.oversized_requests > 5 || usage.total_memory_usage > config_.max_per_connection_memory * 0.9) {
237 return AttackType::MEMORY_EXHAUSTION;
238 }
239
240 // Connection flooding detection
241 if (usage.active_connections > config_.max_connections_per_ip * 0.8 ||
242 usage.connection_timestamps.size() > config_.max_new_connections_per_second * 2) {
243 return AttackType::CONNECTION_FLOODING;
244 }
245
246 // Packet flooding detection
247 if (usage.packets_in_window > config_.max_packets_per_second * 2) {
248 return AttackType::PACKET_FLOODING;
249 }
250
251 // Handshake flooding detection
252 if (usage.pending_handshakes > config_.max_incomplete_handshakes_per_ip * 0.8) {
253 return AttackType::HANDSHAKE_FLOODING;
254 }
255
256 // Malformed packet detection
257 if (usage.malformed_packets > 10) {
258 return AttackType::COMPUTATIONAL_EXHAUSTION;
259 }
260
261 return AttackType::NONE;
262}

377void DoSProtectionEngine::cleanup_old_tracking_data() {
378 std::lock_guard<std::mutex> lock(ip_tracking_mutex_);
379
380 auto cutoff = std::chrono::steady_clock::now() - std::chrono::hours(1);
381
382 for (auto it = ip_usage_map_.begin(); it != ip_usage_map_.end();) {
383 auto& usage = it->second;
384
385 // Remove IPs with no recent activity and no active connections
386 if (usage.last_activity < cutoff && usage.active_connections == 0 &&
387 !usage.is_blacklisted) {
388 it = ip_usage_map_.erase(it);
389 } else {
390 // Clean up old timestamps for remaining IPs
391 usage.connection_timestamps.erase(
392 std::remove_if(usage.connection_timestamps.begin(),
393 usage.connection_timestamps.end(),
394 [cutoff](const auto& timestamp) {
395 return timestamp < cutoff;
396 }),
397 usage.connection_timestamps.end());
398
399 usage.packet_timestamps.erase(
400 std::remove_if(usage.packet_timestamps.begin(),
401 usage.packet_timestamps.end(),
402 [cutoff](const auto& timestamp) {
403 return timestamp < cutoff;
404 }),
405 usage.packet_timestamps.end());
406 ++it;
407 }
408 }
409
410 // Process scheduled blacklist and rate limit removals
411 process_scheduled_removals();
412}

414void DoSProtectionEngine::process_scheduled_removals() {
415 auto now = std::chrono::steady_clock::now();
416
417 // Process blacklist removals
418 {
419 std::lock_guard<std::mutex> timer_lock(timer_mutex_);
420 std::lock_guard<std::mutex> ip_lock(ip_tracking_mutex_);
421
422 auto it = scheduled_blacklist_removals_.begin();
423 while (it != scheduled_blacklist_removals_.end() && it->first <= now) {
424 const auto& ip = it->second;
425 auto ip_it = ip_usage_map_.find(ip);
426 if (ip_it != ip_usage_map_.end()) {
427 ip_it->second.is_blacklisted = false;
428 // Restore partial trust score
429 ip_it->second.trust_score = std::min(0.5, ip_it->second.trust_score + 0.2);
430 }
431 it = scheduled_blacklist_removals_.erase(it);
432 }
433 }
434
435 // Process rate limit removals
436 {
437 std::lock_guard<std::mutex> timer_lock(timer_mutex_);
438 std::lock_guard<std::mutex> ip_lock(ip_tracking_mutex_);
439
440 auto it = scheduled_rate_limit_removals_.begin();
441 while (it != scheduled_rate_limit_removals_.end() && it->first <= now) {
442 const auto& ip = it->second.first;
443 auto ip_it = ip_usage_map_.find(ip);
444 if (ip_it != ip_usage_map_.end()) {
445 ip_it->second.is_rate_limited = false;
446 // Restore partial trust score
447 ip_it->second.trust_score = std::min(0.8, ip_it->second.trust_score + 0.1);
448 }
449 it = scheduled_rate_limit_removals_.erase(it);
450 }
451 }
452}

648Result<std::unique_ptr<ZeroCopyBuffer>>
649DoSResistantAllocator::allocate_protected_buffer(size_t size, const std::string& source_ip,
650 const std::string& purpose) {
651 if (allocation_lockdown_.load()) {
652 return Result<std::unique_ptr<ZeroCopyBuffer>>(DTLSError::SERVICE_UNAVAILABLE);
653 }
654
655 // Validate buffer size
656 if (!validate_buffer_size(size, purpose)) {
657 return Result<std::unique_ptr<ZeroCopyBuffer>>(DTLSError::BUFFER_TOO_LARGE);
658 }
659
660 // Check DoS protection
661 auto protection_result = DoSProtectionEngine::instance().check_memory_allocation(size, source_ip);
662 if (!protection_result) {
663 return Result<std::unique_ptr<ZeroCopyBuffer>>(protection_result.error());
664 }
665
666 // Check resource quotas
667 auto quota_result = ResourceQuotaManager::instance().allocate_memory_quota(source_ip, size);
668 if (!quota_result) {
669 return Result<std::unique_ptr<ZeroCopyBuffer>>(quota_result.error());
670 }
671
672 // Track allocation pattern
673 track_allocation_pattern(source_ip, size, purpose);
674
675 // Check for suspicious patterns
676 if (detect_allocation_abuse(source_ip)) {
677 ResourceQuotaManager::instance().release_memory_quota(source_ip, size);
678 return Result<std::unique_ptr<ZeroCopyBuffer>>(DTLSError::ABUSE_DETECTED);
679 }
680
681 // Allocate buffer
682 auto buffer = std::make_unique<ZeroCopyBuffer>(size);
683
684 // Track allocation in DoS protection system

710void DoSResistantAllocator::track_allocation_pattern(const std::string& source_ip,
711 size_t size, const std::string& purpose) {
712 std::lock_guard<std::mutex> lock(patterns_mutex_);
713 auto& pattern = ip_allocation_patterns_[source_ip];
714
715 pattern.total_bytes_requested += size;
716 pattern.allocation_count++;
717 pattern.recent_allocations.push_back(std::chrono::steady_clock::now());
718 pattern.purpose_breakdown[purpose] += size;
719
720 // Clean up old allocation timestamps
721 auto cutoff = std::chrono::steady_clock::now() - std::chrono::minutes(5);
722 pattern.recent_allocations.erase(
723 std::remove_if(pattern.recent_allocations.begin(),
724 pattern.recent_allocations.end(),
725 [cutoff](const auto& timestamp) {
726 return timestamp < cutoff;
727 }),
728 pattern.recent_allocations.end());
729}

731bool DoSResistantAllocator::detect_allocation_abuse(const std::string& source_ip) const {
732 std::lock_guard<std::mutex> lock(patterns_mutex_);
733 auto it = ip_allocation_patterns_.find(source_ip);
734 if (it == ip_allocation_patterns_.end()) {
735 return false;
736 }
737
738 const auto& pattern = it->second;
739
740 // Check for rapid allocation bursts
741 if (pattern.recent_allocations.size() > 50) { // 50 allocations in 5 minutes
742 return true;
743 }
744
745 // Check for excessive total allocation
746 if (pattern.total_bytes_requested > 10 * 1024 * 1024) { // 10MB total
747 return true;
748 }
749
750 return false;
751}
