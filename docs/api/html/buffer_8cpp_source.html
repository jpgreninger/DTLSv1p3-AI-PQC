510bool constant_time_compare(const BufferView& a, const BufferView& b) noexcept {
511 // Always perform comparison, even if sizes differ
512 // This prevents early exit timing attacks
513 size_t min_size = (a.size() < b.size()) ? a.size() : b.size();
514 size_t max_size = (a.size() > b.size()) ? a.size() : b.size();
515
516 const std::byte* ptr_a = a.data();
517 const std::byte* ptr_b = b.data();
518
519 // Mark as volatile to prevent compiler optimizations
520 volatile std::byte result{0};
521
522 // Compare all bytes up to the minimum size
523 for (size_t i = 0; i < min_size; ++i) {
524 std::byte diff = (ptr_a[i] ^ ptr_b[i]);
525 result = result | diff;
526 }
527
528 // If sizes differ, XOR remaining bytes with themselves
529 // This maintains constant-time behavior regardless of size difference
530 if (a.size() != b.size()) {
531 result = result | std::byte{1}; // Set result to indicate size mismatch
532
533 // Perform dummy operations to maintain constant timing
534 const std::byte* longer_ptr = (a.size() > b.size()) ? ptr_a : ptr_b;
535 for (size_t i = min_size; i < max_size; ++i) {
536 volatile std::byte dummy = longer_ptr[i];
537 (void)dummy; // Suppress unused variable warning
538 }
539 }
540
541 return result == std::byte{0};
542}
