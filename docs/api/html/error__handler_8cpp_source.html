27Result<bool> ErrorHandler::process_error(DTLSError error,
28 std::shared_ptr<ErrorContext> context) {
29 // Update statistics
30 stats_.total_errors++;
31
32 // Update DoS counters
33 update_dos_counters();
34
35 // Determine if this is a fatal error
36 bool is_fatal = is_fatal_error(error);
37 if (is_fatal) {
38 stats_.fatal_errors++;
39 } else if (is_retryable_error(error)) {
40 stats_.retryable_errors++;
41 }
42
43 // Record error in context if available
44 if (context) {
45 context->record_error(error, "protocol", error_message(error),
46 is_security_relevant_error(error));
47 }
48
49 // Log error securely
50 auto log_result = log_error_securely(error, context);
51 if (!log_result.is_success()) {
52 // Logging failure shouldn't fail the operation, just note it
53 }
54
55 // Generate alert if appropriate
56 if (alert_manager_ && should_generate_alert_for_error(error)) {
57 auto alert_result = alert_manager_->generate_alert_for_error(error, context);
58 if (alert_result.is_success() && alert_result.value().has_value()) {
59 stats_.alerts_generated++;
60 }
61 }
62
63 // Determine if connection should continue
64 bool should_continue = !should_terminate_connection(error);
65
66 if (!should_continue) {
67 stats_.connections_terminated++;
68 }
69
70 return make_result(should_continue);
71}

73Result<void> ErrorHandler::handle_invalid_record(ContentType record_type,
74 std::shared_ptr<ErrorContext> context) {
75 // RFC 9147 Section 4.2.1: "In general, invalid records SHOULD be silently discarded"
76 stats_.invalid_records_discarded++;
77
78 // Update DoS protection counters
79 {
80 std::lock_guard<std::mutex> lock(dos_state_.reset_mutex);
81 dos_state_.invalid_records_this_second++;
82
83 // Check if we're under attack
84 if (dos_state_.invalid_records_this_second > config_.max_invalid_records_per_second) {
85 stats_.dos_attacks_detected++;
86
87 if (error_reporter_) {
88 error_reporter_->report_security_incident(
89 DTLSError::REPLAY_ATTACK_DETECTED,
90 "invalid_record_flood",
91 0.8, // High confidence
92 context
93 );
94 }
95 }
96 }
97
98 // Log for diagnostic purposes if configured
99 if (config_.log_invalid_records && error_reporter_) {
100 std::string category = "record_validation";
101 std::string message = "Invalid record silently discarded (content_type=" +
102 std::to_string(static_cast<int>(record_type)) + ")";
103
104 error_reporter_->report_error(
105 ErrorReporter::LogLevel::DEBUG,
106 DTLSError::INVALID_RECORD_HEADER,
107 category,
108 message,
109 context
110 );
111 }
112
113 // Record in error context
114 if (context) {
115 context->record_error(DTLSError::INVALID_RECORD_HEADER,
116 "record_layer",
117 "Invalid record discarded per RFC 9147",
118 false); // Not security-relevant by itself
119 }
120
121 // RFC 9147: Generally should NOT generate alert for invalid records,
122 // especially for UDP transport due to DoS risks
123 if (config_.generate_alerts_on_invalid_records && is_transport_secure()) {
124 if (alert_manager_) {
125 auto alert_result = alert_manager_->handle_invalid_record(
126 record_type,
127 context ? context->get_connection_info().connection_id_hash : "",
128 context
129 );
130 // Alert manager will apply its own policy
131 }
132 }
133
134 return make_result();
135}

137Result<bool> ErrorHandler::handle_authentication_failure(Epoch epoch,
138 std::shared_ptr<ErrorContext> context) {
139 // RFC 9147: "track records that fail authentication"
140 {
141 std::lock_guard<std::mutex> lock(dos_state_.reset_mutex);
142 dos_state_.auth_failures_this_epoch++;
143
144 // Check if authentication failures exceed threshold per epoch
145 if (dos_state_.auth_failures_this_epoch > config_.max_auth_failures_per_epoch) {
146 stats_.dos_attacks_detected++;
147
148 // High confidence DoS/replay attack
149 if (error_reporter_) {
150 error_reporter_->report_security_incident(
151 DTLSError::AUTHENTICATION_FAILED,
152 "authentication_failure_flood",
153 0.9, // Very high confidence
154 context
155 );
156 }
157
158 // RFC 9147: "implementations SHOULD close the connection if
159 // authentication failure records exceed a specific limit"
160 return make_result(false); // Connection should be closed
161 }
162 }
163
164 // Record authentication failure
165 if (context) {
166 context->record_security_error(DTLSError::AUTHENTICATION_FAILED,
167 "authentication_failure", 0.6);
168 }
169
170 // Log the authentication failure
171 if (error_reporter_) {
172 error_reporter_->report_error(
173 ErrorReporter::LogLevel::WARNING,
174 DTLSError::AUTHENTICATION_FAILED,
175 "authentication",
176 "Record authentication failed for epoch " + std::to_string(epoch),
177 context
178 );
179 }
180
181 return make_result(true); // Connection can continue for now
182}

184Result<std::vector<uint8_t>> ErrorHandler::generate_alert_if_appropriate(
185 AlertDescription alert_desc,
186 std::shared_ptr<ErrorContext> context) {
187
188 // Delegate to alert manager if available
189 if (alert_manager_) {
190 auto result = alert_manager_->generate_alert(
191 AlertLevel::FATAL, // RFC 9147: must be fatal to prevent probing
192 alert_desc,
193 context ? context->get_connection_info().connection_id_hash : "",
194 context
195 );
196
197 if (result.is_success() && result.value().has_value()) {
198 return make_result(result.value().value());
199 }
200 }
201
202 // Fallback: create alert directly if no alert manager
203 if (is_transport_secure()) {
204 auto alert_data = AlertManager::serialize_alert(AlertLevel::FATAL, alert_desc);
205 stats_.alerts_generated++;
206 return make_result(alert_data);
207 }
208
209 // For insecure transports (UDP), don't generate alert per RFC 9147
210 return make_error<std::vector<uint8_t>>(DTLSError::OPERATION_NOT_SUPPORTED);
211}

281bool ErrorHandler::should_generate_alert_for_error(DTLSError error) const {
282 // RFC 9147: For UDP, generating alerts is NOT RECOMMENDED due to DoS risks
283 if (!is_transport_secure()) {
284 return false;
285 }
286
287 // For secure transports, generate alerts for protocol violations
288 switch (error) {
289 case DTLSError::PROTOCOL_VERSION_NOT_SUPPORTED:
290 case DTLSError::HANDSHAKE_FAILURE:
291 case DTLSError::ILLEGAL_PARAMETER:
292 case DTLSError::DECODE_ERROR:
293 case DTLSError::CERTIFICATE_VERIFY_FAILED:
294 case DTLSError::INSUFFICIENT_SECURITY:
295 return true;
296 default:
297 return false;
298 }
