123bool EpochManager::is_valid_epoch(uint16_t epoch) const {
124 std::lock_guard<std::mutex> lock(mutex_);
125
126 // Current epoch is always valid
127 if (epoch == current_epoch_) {
128 return true;
129 }
130
131 // Previous epoch may be valid during transition
132 if (epoch == current_epoch_ - 1 && current_epoch_ > 0) {
133 return epoch_keys_.find(epoch) != epoch_keys_.end();
134 }
135
136 return false;
137}

139Result<void> EpochManager::set_epoch_keys(uint16_t epoch,
140 const std::vector<uint8_t>& read_key,
141 const std::vector<uint8_t>& write_key,
142 const std::vector<uint8_t>& read_iv,
143 const std::vector<uint8_t>& write_iv) {
144 std::lock_guard<std::mutex> lock(mutex_);
145
146 // Epoch 0 uses null protection (empty keys are valid)
147 // For other epochs, keys must be non-empty
148 if (epoch != 0 && (read_key.empty() || write_key.empty() || read_iv.empty() || write_iv.empty())) {
149 return Result<void>(DTLSError::INVALID_KEY_MATERIAL);
150 }
151
152 EpochCryptoParams params;
153 params.read_key = read_key;
154 params.write_key = write_key;
155 params.read_iv = read_iv;
156 params.write_iv = write_iv;
157 params.cipher_suite = CipherSuite::TLS_AES_128_GCM_SHA256; // Default
158
159 epoch_keys_[epoch] = std::move(params);
160 return Result<void>();
161}

259Result<DTLSCiphertext> RecordLayer::protect_record(const DTLSPlaintext& plaintext) {
260 if (!plaintext.is_valid()) {
261 return Result<DTLSCiphertext>(DTLSError::INVALID_PLAINTEXT_RECORD);
262 }
263
264 uint16_t current_epoch = epoch_manager_->get_current_epoch();
265
266 // Get crypto parameters for current epoch
267 auto crypto_params_result = epoch_manager_->get_epoch_crypto_params(current_epoch);
268 if (!crypto_params_result.is_success()) {
269 return Result<DTLSCiphertext>(crypto_params_result.error());
270 }
271
272 const auto& crypto_params = crypto_params_result.value();
273
274 // Initially use unencrypted sequence number - will encrypt after AEAD operation
275 SequenceNumber48 sequence_number = plaintext.get_sequence_number();
276
277 // Epoch 0 uses null protection (no encryption)
278 if (current_epoch == 0 || crypto_params.write_key.empty()) {
279 // Create DTLSCiphertext with no encryption
280 memory::Buffer encrypted_payload(plaintext.get_fragment().size());
281 std::memcpy(encrypted_payload.mutable_data(),
282 plaintext.get_fragment().data(),
283 plaintext.get_fragment().size());
284
285 DTLSCiphertext ciphertext(ContentType::APPLICATION_DATA,
286 plaintext.get_version(),
287 plaintext.get_epoch(),
288 sequence_number,
289 std::move(encrypted_payload));
290
291 update_stats_protected();
292 return Result<DTLSCiphertext>(std::move(ciphertext));
293 }
294
295 // Construct AEAD nonce with original (unencrypted) sequence number
296 auto nonce_result = construct_aead_nonce(current_epoch,
297 plaintext.get_sequence_number(),
298 crypto_params.write_iv);
299 if (!nonce_result.is_success()) {
300 return Result<DTLSCiphertext>(nonce_result.error());
301 }
302
303 // Construct additional authenticated data using encrypted sequence number
304 ConnectionID cid = connection_id_manager_->is_connection_id_enabled() ?
305 connection_id_manager_->get_peer_connection_id() : ConnectionID{};
306
307 // Create temporary header with encrypted sequence number for AAD
308 struct TempHeader {
309 ContentType content_type;
310 ProtocolVersion version;
311 uint16_t epoch;
312 uint64_t sequence_number;
313 uint16_t length;
314 } temp_header = {
315 plaintext.get_type(),
316 plaintext.get_version(),
317 plaintext.get_epoch(),
318 static_cast<uint64_t>(sequence_number),
319 plaintext.get_length()
320 };
321
322 auto aad_result = construct_additional_data_dtls(temp_header, cid);
323 if (!aad_result.is_success()) {
324 return Result<DTLSCiphertext>(aad_result.error());
325 }
326
327 // Perform AEAD encryption
328 crypto::AEADEncryptionParams encrypt_params;
329 encrypt_params.key = crypto_params.write_key;
330 encrypt_params.nonce = nonce_result.value();
331 encrypt_params.additional_data = aad_result.value();
332 encrypt_params.plaintext = std::vector<uint8_t>(
333 reinterpret_cast<const uint8_t*>(plaintext.get_fragment().data()),
334 reinterpret_cast<const uint8_t*>(plaintext.get_fragment().data()) + plaintext.get_fragment().size());
335
336 auto encrypt_result = crypto_provider_->encrypt_aead(encrypt_params);
337 if (!encrypt_result.is_success()) {
338 update_stats_decryption_failed();
339 return Result<DTLSCiphertext>(encrypt_result.error());
340 }
341
342 const auto& encrypt_output = encrypt_result.value();
343
344 // Create encrypted payload buffer including authentication tag
345 memory::Buffer encrypted_record(encrypt_output.ciphertext.size() + encrypt_output.tag.size());
346 std::memcpy(encrypted_record.mutable_data(),
347 encrypt_output.ciphertext.data(),
348 encrypt_output.ciphertext.size());
349 std::memcpy(encrypted_record.mutable_data() + encrypt_output.ciphertext.size(),
350 encrypt_output.tag.data(),
351 encrypt_output.tag.size());
352
353 // Now encrypt sequence number using first 16 bytes of ciphertext (RFC 9147 Section 4.2.3)
354 SequenceNumber48 encrypted_sequence_number = sequence_number;
355 if (encrypted_record.size() >= 16) {
356 // Derive sequence number encryption key
357 auto seq_key_result = crypto::utils::derive_sequence_number_mask(
358 *crypto_provider_, crypto_params.write_key, "sn",
359 crypto::utils::get_cipher_suite_hash(current_cipher_suite_));
360 if (seq_key_result.is_success()) {
361 // Get cipher type from current cipher suite
362 AEADCipher cipher_type = AEADCipher::AES_128_GCM; // Default
363 switch (current_cipher_suite_) {
364 case CipherSuite::TLS_AES_128_GCM_SHA256:
365 cipher_type = AEADCipher::AES_128_GCM;
366 break;
367 case CipherSuite::TLS_AES_256_GCM_SHA384:
368 cipher_type = AEADCipher::AES_256_GCM;
369 break;
370 case CipherSuite::TLS_CHACHA20_POLY1305_SHA256:
371 cipher_type = AEADCipher::CHACHA20_POLY1305;
372 break;
373 case CipherSuite::TLS_AES_128_CCM_SHA256:
374 cipher_type = AEADCipher::AES_128_CCM;
375 break;
376 case CipherSuite::TLS_AES_128_CCM_8_SHA256:
377 cipher_type = AEADCipher::AES_128_CCM_8;
378 break;
379 default:
380 cipher_type = AEADCipher::AES_128_GCM;
381 break;
382 }
383
384 // Encrypt sequence number using first 16 bytes of ciphertext
385 std::vector<uint8_t> ciphertext_prefix(16);
386 std::memcpy(ciphertext_prefix.data(), encrypted_record.data(), 16);
387 auto encrypted_seq_result = crypto::utils::encrypt_sequence_number(
388 *crypto_provider_,
389 static_cast<uint64_t>(sequence_number),
390 seq_key_result.value(),
391 ciphertext_prefix,
392 cipher_type);
393 if (encrypted_seq_result.is_success()) {
394 encrypted_sequence_number = SequenceNumber48(encrypted_seq_result.value());
395 }
396 }
397 }
398
399 DTLSCiphertext ciphertext(ContentType::APPLICATION_DATA,
400 plaintext.get_version(),
401 plaintext.get_epoch(),
402 encrypted_sequence_number,
403 std::move(encrypted_record));
404
405 // Add connection ID if enabled
406 if (connection_id_manager_->is_connection_id_enabled() && !cid.empty()) {
407 ciphertext.set_connection_id(cid);
408 }
409
410 update_stats_protected();
411
412 // Update key update record counter
413 {
414 std::lock_guard<std::mutex> lock(key_update_mutex_);
415 key_update_stats_.records_since_last_update++;
416 }
417
418 return Result<DTLSCiphertext>(std::move(ciphertext));
419}

421Result<DTLSPlaintext> RecordLayer::unprotect_record(const DTLSCiphertext& ciphertext) {
422 if (!ciphertext.is_valid()) {
423 return Result<DTLSPlaintext>(DTLSError::INVALID_CIPHERTEXT_RECORD);
424 }
425
426 uint16_t record_epoch = ciphertext.get_epoch();
427
428 // Validate epoch
429 if (!epoch_manager_->is_valid_epoch(record_epoch)) {
430 return Result<DTLSPlaintext>(DTLSError::INVALID_EPOCH);
431 }
432
433 // Get crypto parameters for record epoch
434 auto crypto_params_result = epoch_manager_->get_epoch_crypto_params(record_epoch);
435 if (!crypto_params_result.is_success()) {
436 return Result<DTLSPlaintext>(crypto_params_result.error());
437 }
438
439 const auto& crypto_params = crypto_params_result.value();
440
441 // Decrypt sequence number using RFC 9147 Section 4.1.3
442 uint64_t decrypted_seq_num = ciphertext.get_encrypted_sequence_number();
443 if (record_epoch > 0 && !crypto_params.read_key.empty()) {
444 // Derive sequence number decryption key
445 auto seq_key_result = crypto::utils::derive_sequence_number_mask(
446 *crypto_provider_, crypto_params.read_key, "sn",
447 crypto::utils::get_cipher_suite_hash(current_cipher_suite_));
448 if (!seq_key_result.is_success()) {
449 return Result<DTLSPlaintext>(seq_key_result.error());
450 }
451
452 // Decrypt the sequence number
453 // Get cipher type from current cipher suite
454 AEADCipher cipher_type = AEADCipher::AES_128_GCM; // Default
455 switch (current_cipher_suite_) {
456 case CipherSuite::TLS_AES_128_GCM_SHA256:
457 cipher_type = AEADCipher::AES_128_GCM;
458 break;
459 case CipherSuite::TLS_AES_256_GCM_SHA384:
460 cipher_type = AEADCipher::AES_256_GCM;
461 break;
462 case CipherSuite::TLS_CHACHA20_POLY1305_SHA256:
463 cipher_type = AEADCipher::CHACHA20_POLY1305;
464 break;
465 case CipherSuite::TLS_AES_128_CCM_SHA256:
466 cipher_type = AEADCipher::AES_128_CCM;
467 break;
468 case CipherSuite::TLS_AES_128_CCM_8_SHA256:
469 cipher_type = AEADCipher::AES_128_CCM_8;
470 break;
471 default:
472 cipher_type = AEADCipher::AES_128_GCM;
473 break;
474 }
475
476 // Extract first 16 bytes of encrypted record for sequence number decryption
477 const auto& encrypted_record = ciphertext.get_encrypted_record();
478 std::vector<uint8_t> ciphertext_prefix(16, 0);
479 if (encrypted_record.size() >= 16) {
480 std::memcpy(ciphertext_prefix.data(), encrypted_record.data(), 16);
481 } else {
482 // Fallback: pad to 16 bytes if ciphertext is shorter
483 std::memcpy(ciphertext_prefix.data(), encrypted_record.data(), encrypted_record.size());
484 }
485
486 auto decrypted_seq_result = crypto::utils::decrypt_sequence_number(
487 *crypto_provider_, ciphertext.get_encrypted_sequence_number(), seq_key_result.value(),
488 ciphertext_prefix, cipher_type);
489 if (!decrypted_seq_result.is_success()) {
490 return Result<DTLSPlaintext>(decrypted_seq_result.error());
491 }
492
493 decrypted_seq_num = decrypted_seq_result.value();
494 }
495
496 // Epoch 0 uses null protection (no decryption)
497 if (record_epoch == 0 || crypto_params.read_key.empty()) {
498 // Create DTLSPlaintext with no decryption
499 memory::Buffer payload(ciphertext.get_encrypted_record().size());
500 std::memcpy(payload.mutable_data(),
501 ciphertext.get_encrypted_record().data(),
502 ciphertext.get_encrypted_record().size());
503
504 DTLSPlaintext plaintext(ciphertext.get_type(),
505 ciphertext.get_version(),
506 ciphertext.get_epoch(),
507 SequenceNumber48(decrypted_seq_num),
508 std::move(payload));
509
510 update_stats_unprotected();
511 return Result<DTLSPlaintext>(std::move(plaintext));
512 }
513
514 // Construct AEAD nonce with original (decrypted) sequence number
515 auto nonce_result = construct_aead_nonce(record_epoch,
516 decrypted_seq_num,
517 crypto_params.read_iv);
518 if (!nonce_result.is_success()) {
519 return Result<DTLSPlaintext>(nonce_result.error());
520 }
521
522 // Construct additional authenticated data using encrypted sequence number (as sent)
523 ConnectionID cid;
524 if (ciphertext.has_cid()) {
525 cid = ciphertext.get_connection_id_vector();
526 }
527
528 // Create temporary header with encrypted sequence number for AAD verification
529 struct TempHeader {
530 ContentType content_type;
531 ProtocolVersion version;
532 uint16_t epoch;
533 uint64_t sequence_number;
534 uint16_t length;
535 } temp_header = {
536 ciphertext.get_type(),
537 ciphertext.get_version(),
538 ciphertext.get_epoch(),
539 static_cast<uint64_t>(ciphertext.get_encrypted_sequence_number()),
540 ciphertext.get_length()
541 };
542
543 auto aad_result = construct_additional_data_dtls(temp_header, cid);
544 if (!aad_result.is_success()) {
545 return Result<DTLSPlaintext>(aad_result.error());
546 }
547
548 // Extract ciphertext and authentication tag from encrypted record
549 const auto& encrypted_record = ciphertext.get_encrypted_record();
550 if (encrypted_record.size() < 16) { // Minimum auth tag size
551 return Result<DTLSPlaintext>(DTLSError::INVALID_CIPHERTEXT_RECORD);
552 }
553
554 // Assume GCM tag size of 16 bytes for now (should be determined by cipher suite)
555 size_t tag_size = 16;
556 size_t ciphertext_size = encrypted_record.size() - tag_size;
557
558 // Perform AEAD decryption
559 crypto::AEADDecryptionParams decrypt_params;
560 decrypt_params.key = crypto_params.read_key;
561 decrypt_params.nonce = nonce_result.value();
562 decrypt_params.additional_data = aad_result.value();
563 decrypt_params.ciphertext = std::vector<uint8_t>(
564 reinterpret_cast<const uint8_t*>(encrypted_record.data()),
565 reinterpret_cast<const uint8_t*>(encrypted_record.data()) + ciphertext_size);
566 decrypt_params.tag = std::vector<uint8_t>(
567 reinterpret_cast<const uint8_t*>(encrypted_record.data()) + ciphertext_size,
568 reinterpret_cast<const uint8_t*>(encrypted_record.data()) + encrypted_record.size());
569
570 auto decrypt_result = crypto_provider_->decrypt_aead(decrypt_params);
571 if (!decrypt_result.is_success()) {
572 update_stats_decryption_failed();
573 return Result<DTLSPlaintext>(decrypt_result.error());
574 }
575
576 const auto& plaintext_data = decrypt_result.value();
577
578 // Create DTLSPlaintext with decrypted sequence number
579 memory::Buffer payload(plaintext_data.size());
580 std::memcpy(payload.mutable_data(), plaintext_data.data(), plaintext_data.size());
581
582 DTLSPlaintext plaintext(ciphertext.get_type(),
583 ciphertext.get_version(),
584 ciphertext.get_epoch(),
585 SequenceNumber48(decrypted_seq_num),
586 std::move(payload));
587
588 update_stats_unprotected();
589 return Result<DTLSPlaintext>(std::move(plaintext));
590}

592Result<DTLSCiphertext> RecordLayer::prepare_outgoing_record(const DTLSPlaintext& plaintext) {
593 // Get next sequence number
594 uint64_t sequence_number = send_sequence_manager_->get_next_sequence_number();
595
596 // Check for sequence number overflow
597 if (send_sequence_manager_->would_overflow()) {
598 return Result<DTLSCiphertext>(DTLSError::SEQUENCE_NUMBER_OVERFLOW);
599 }
600
601 // Create record with assigned sequence number
602 DTLSPlaintext outgoing_record = plaintext;
603 outgoing_record.set_sequence_number(SequenceNumber48(sequence_number));
604 outgoing_record.set_epoch(epoch_manager_->get_current_epoch());
605
606 // Protect the record
607 auto ciphertext_result = protect_record(outgoing_record);
608 if (!ciphertext_result.is_success()) {
609 return ciphertext_result;
610 }
611
612 update_stats_sent();
613 return ciphertext_result;

616Result<DTLSPlaintext> RecordLayer::process_incoming_record(const DTLSCiphertext& ciphertext) {
617 uint16_t record_epoch = ciphertext.get_epoch();
618
619 // Decrypt sequence number for anti-replay checking
620 uint64_t decrypted_seq_num = ciphertext.get_encrypted_sequence_number();
621 if (record_epoch > 0) {
622 // Get crypto parameters for sequence number decryption
623 auto crypto_params_result = epoch_manager_->get_epoch_crypto_params(record_epoch);
624 if (crypto_params_result.is_success() && !crypto_params_result.value().read_key.empty()) {
625 // Derive sequence number decryption key
626 auto seq_key_result = crypto::utils::derive_sequence_number_mask(
627 *crypto_provider_, crypto_params_result.value().read_key, "sn",
628 crypto::utils::get_cipher_suite_hash(current_cipher_suite_));
629 if (seq_key_result.is_success()) {
630 // Decrypt the sequence number for anti-replay check
631 // Get cipher type from current cipher suite
632 AEADCipher cipher_type = AEADCipher::AES_128_GCM; // Default
633 switch (current_cipher_suite_) {
634 case CipherSuite::TLS_AES_128_GCM_SHA256:
635 cipher_type = AEADCipher::AES_128_GCM;
636 break;
637 case CipherSuite::TLS_AES_256_GCM_SHA384:
638 cipher_type = AEADCipher::AES_256_GCM;
639 break;
640 case CipherSuite::TLS_CHACHA20_POLY1305_SHA256:
641 cipher_type = AEADCipher::CHACHA20_POLY1305;
642 break;
643 case CipherSuite::TLS_AES_128_CCM_SHA256:
644 cipher_type = AEADCipher::AES_128_CCM;
645 break;
646 case CipherSuite::TLS_AES_128_CCM_8_SHA256:
647 cipher_type = AEADCipher::AES_128_CCM_8;
648 break;
649 default:
650 cipher_type = AEADCipher::AES_128_GCM;
651 break;
652 }
653
654 // Extract first 16 bytes of encrypted record for sequence number decryption
655 const auto& encrypted_record = ciphertext.get_encrypted_record();
656 std::vector<uint8_t> ciphertext_prefix(16, 0);
657 if (encrypted_record.size() >= 16) {
658 std::memcpy(ciphertext_prefix.data(), encrypted_record.data(), 16);
659 } else {
660 // Fallback: pad to 16 bytes if ciphertext is shorter
661 std::memcpy(ciphertext_prefix.data(), encrypted_record.data(), encrypted_record.size());
662 }
663
664 auto decrypted_seq_result = crypto::utils::decrypt_sequence_number(
665 *crypto_provider_, ciphertext.get_encrypted_sequence_number(), seq_key_result.value(),
666 ciphertext_prefix, cipher_type);
667 if (decrypted_seq_result.is_success()) {
668 decrypted_seq_num = decrypted_seq_result.value();
669 }
670 }
671 }
672 }
673
674 // Get or create anti-replay window for this epoch
675 auto window_it = receive_windows_.find(record_epoch);
676 if (window_it == receive_windows_.end()) {
677 receive_windows_[record_epoch] = std::make_unique<AntiReplayWindow>();
678 window_it = receive_windows_.find(record_epoch);
679 }
680
681 // Check for replay attack using decrypted sequence number
682 if (!window_it->second->is_valid_sequence_number(decrypted_seq_num)) {
683 update_stats_replay_detected();
684 return Result<DTLSPlaintext>(DTLSError::REPLAY_ATTACK_DETECTED);
685 }
686
687 // Validate connection ID if present
688 if (ciphertext.has_cid()) {
689 auto cid = ciphertext.get_connection_id_vector();
690 if (!connection_id_manager_->is_valid_connection_id(cid)) {
691 return Result<DTLSPlaintext>(DTLSError::INVALID_CONNECTION_ID);
692 }
693 }
694
695 // Unprotect the record
696 auto plaintext_result = unprotect_record(ciphertext);
697 if (!plaintext_result.is_success()) {
698 return plaintext_result;
699 }
700
701 // Mark sequence number as received (prevents replays) using decrypted sequence number
702 window_it->second->mark_received(decrypted_seq_num);
703
704 update_stats_received();
705 return plaintext_result;
706}

757Result<std::vector<uint8_t>> RecordLayer::construct_aead_nonce(uint16_t epoch,
758 uint64_t sequence_number,
759 const std::vector<uint8_t>& base_iv) const {
760 if (base_iv.size() != 12) { // Standard AEAD nonce size
761 return Result<std::vector<uint8_t>>(DTLSError::INVALID_IV_SIZE);
762 }
763
764 std::vector<uint8_t> nonce = base_iv;
765
766 // XOR sequence number into the nonce (RFC 9147)
767 uint64_t seq_net = htobe64(sequence_number);
768 uint8_t* seq_bytes = reinterpret_cast<uint8_t*>(&seq_net);
769
770 // XOR the last 8 bytes of nonce with sequence number
771 for (size_t i = 0; i < 8; ++i) {
772 nonce[4 + i] ^= seq_bytes[i];
773 }
774
775 return Result<std::vector<uint8_t>>(std::move(nonce));
776}

778Result<std::vector<uint8_t>> RecordLayer::construct_additional_data(const RecordHeader& header,
779 const ConnectionID& cid) const {
780 std::vector<uint8_t> aad;
781
782 // Connection ID (if present)
783 if (!cid.empty()) {
784 aad.insert(aad.end(), cid.begin(), cid.end());
785 }
786
787 // Record header without length
788 aad.push_back(static_cast<uint8_t>(header.content_type));
789
790 // Protocol version (2 bytes, network order)
791 uint16_t version_net = htons(static_cast<uint16_t>(header.version));
792 uint8_t* version_bytes = reinterpret_cast<uint8_t*>(&version_net);
793 aad.push_back(version_bytes[0]);
794 aad.push_back(version_bytes[1]);
795
796 // Epoch (2 bytes, network order)
797 uint16_t epoch_net = htons(header.epoch);
798 uint8_t* epoch_bytes = reinterpret_cast<uint8_t*>(&epoch_net);
799 aad.push_back(epoch_bytes[0]);
800 aad.push_back(epoch_bytes[1]);
801
802 // Sequence number (6 bytes, network order)
803 uint64_t seq_net = htobe64(header.sequence_number);
804 uint8_t* seq_bytes = reinterpret_cast<uint8_t*>(&seq_net);
805 for (int i = 2; i < 8; ++i) { // Only 6 bytes (48-bit sequence number)
806 aad.push_back(seq_bytes[i]);
807 }
808
809 // Length (2 bytes, network order)
810 uint16_t length_net = htons(header.length);
811 uint8_t* length_bytes = reinterpret_cast<uint8_t*>(&length_net);
812 aad.push_back(length_bytes[0]);
813 aad.push_back(length_bytes[1]);
814
815 return Result<std::vector<uint8_t>>(std::move(aad));
816}

893Result<CiphertextRecord> RecordLayer::protect_record_legacy(const PlaintextRecord& plaintext) {
894 // Convert PlaintextRecord to DTLSPlaintext
895 memory::Buffer fragment_copy(plaintext.payload().size());
896 std::memcpy(fragment_copy.mutable_data(), plaintext.payload().data(), plaintext.payload().size());
897
898 DTLSPlaintext dtls_plaintext(
899 plaintext.header().content_type,
900 plaintext.header().version,
901 plaintext.header().epoch,
902 SequenceNumber48(plaintext.header().sequence_number),
903 std::move(fragment_copy)
904 );
905
906 // Protect using new method
907 auto dtls_ciphertext_result = protect_record(dtls_plaintext);
908 if (!dtls_ciphertext_result.is_success()) {
909 return Result<CiphertextRecord>(dtls_ciphertext_result.error());
910 }
911
912 const auto& dtls_ciphertext = dtls_ciphertext_result.value();
913
914 // Convert DTLSCiphertext back to CiphertextRecord for legacy compatibility
915 // Extract ciphertext and tag from encrypted_record
916 const auto& encrypted_record = dtls_ciphertext.get_encrypted_record();
917 size_t tag_size = 16; // Assume GCM
918 size_t ciphertext_size = encrypted_record.size() - tag_size;
919
920 memory::Buffer ciphertext_payload(ciphertext_size);
921 memory::Buffer auth_tag(tag_size);
922
923 std::memcpy(ciphertext_payload.mutable_data(), encrypted_record.data(), ciphertext_size);
924 std::memcpy(auth_tag.mutable_data(), encrypted_record.data() + ciphertext_size, tag_size);
925
926 CiphertextRecord legacy_ciphertext(
927 dtls_ciphertext.get_type(),
928 dtls_ciphertext.get_version(),
929 dtls_ciphertext.get_epoch(),
930 static_cast<uint64_t>(dtls_ciphertext.get_encrypted_sequence_number()),
931 std::move(ciphertext_payload),
932 std::move(auth_tag)
933 );
934
935 // Add connection ID if present
936 if (dtls_ciphertext.has_cid()) {
937 auto cid_vector = dtls_ciphertext.get_connection_id_vector();
938 legacy_ciphertext.set_connection_id(cid_vector);
939 }
940
941 return Result<CiphertextRecord>(std::move(legacy_ciphertext));
942}
