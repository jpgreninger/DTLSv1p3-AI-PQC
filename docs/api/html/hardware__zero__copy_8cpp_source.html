557 auto hw_profile_result = HardwareAccelerationDetector::detect_capabilities();
558 if (!hw_profile_result) {

560 }

562 const auto& profile = hw_profile_result.value();
563 HardwareZeroCopyCrypto::HardwareConfig config;

569 cap.capability == HardwareCapability::ARM_AES) &&
570 cap.available && cap.enabled;
571 });
572
573 bool has_avx = std::any_of(profile.capabilities.begin(), profile.capabilities.end(),
574 [](const auto& cap) {
575 return (cap.capability == HardwareCapability::AVX ||
576 cap.capability == HardwareCapability::AVX2) &&
577 cap.available && cap.enabled;
578 });
579
580 config.enable_simd_batch_ops = has_avx;
581 config.enable_in_place_ops = has_aes;
582 config.simd_batch_size = has_avx ? 8 : 4;
583 config.required_capabilities = has_aes ? HardwareCapability::AES_NI : HardwareCapability::AVX;
584
585 return Result<HardwareZeroCopyCrypto::HardwareConfig>::success(config);

672 auto hw_result = HardwareAccelerationDetector::detect_capabilities();
673 if (!hw_result) {
674 return 64; // Default cache line size
675 }
676
677 const auto& profile = hw_result.value();
678
679 bool has_avx512 = false; // Would need to detect AVX-512
680 bool has_avx2 = std::any_of(profile.capabilities.begin(), profile.capabilities.end(),
681 [](const auto& cap) {
682 return cap.capability == HardwareCapability::AVX2 &&
683 cap.available && cap.enabled;
684 });
685
686 if (has_avx512) {
687 return 64; // AVX-512 prefers 64-byte alignment
688 } else if (has_avx2) {
689 return 32; // AVX2 prefers 32-byte alignment
690 } else {
691 return 16; // SSE prefers 16-byte alignment
692 }
693}

695void copy_optimized(const void* src, void* dest, size_t size) {
696 // Use hardware-accelerated copy if available
697#if defined(__AVX2__)
698 if (size >= 32 && is_aligned(src, 32) && is_aligned(dest, 32)) {
699 // AVX2-optimized copy
700 const __m256i* src_vec = static_cast<const __m256i*>(src);
701 __m256i* dest_vec = static_cast<__m256i*>(dest);
702 size_t vec_count = size / 32;
703
704 for (size_t i = 0; i < vec_count; ++i) {
705 _mm256_store_si256(dest_vec + i, _mm256_load_si256(src_vec + i));
706 }
707
708 // Handle remaining bytes
709 size_t remaining = size % 32;
710 if (remaining > 0) {
711 std::memcpy(static_cast<uint8_t*>(dest) + (vec_count * 32),
712 static_cast<const uint8_t*>(src) + (vec_count * 32),
713 remaining);
714 }
715 return;
716 }
717#endif
718
719 // Fallback to standard copy
720 std::memcpy(dest, src, size);
721}

723void secure_zero(void* ptr, size_t size) {
724#if defined(__AVX2__)
725 if (size >= 32 && is_aligned(ptr, 32)) {
726 // AVX2-optimized zero
727 __m256i* vec_ptr = static_cast<__m256i*>(ptr);
728 __m256i zero = _mm256_setzero_si256();
729 size_t vec_count = size / 32;
730
731 for (size_t i = 0; i < vec_count; ++i) {
732 _mm256_store_si256(vec_ptr + i, zero);
733 }
734
735 // Handle remaining bytes
736 size_t remaining = size % 32;
737 if (remaining > 0) {
738 volatile uint8_t* byte_ptr = static_cast<volatile uint8_t*>(ptr) + (vec_count * 32);
739 for (size_t i = 0; i < remaining; ++i) {
740 byte_ptr[i] = 0;
741 }
742 }
743 return;
744 }
745#endif
746
747 // Fallback to volatile zero
748 volatile uint8_t* byte_ptr = static_cast<volatile uint8_t*>(ptr);
749 for (size_t i = 0; i < size; ++i) {
750 byte_ptr[i] = 0;
751 }
752}
