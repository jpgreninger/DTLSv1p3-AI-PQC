32bool AdaptivePoolSizer::should_expand_pool(const PoolUsagePattern& pattern, size_t current_size) const {
33 if (current_size >= config_.max_pool_size) {
34 return false;
35 }
36
37 // Expand if hit rate is low or peak usage exceeds threshold
38 bool low_hit_rate = pattern.hit_rate < 0.8;
39 bool high_utilization = pattern.peak_concurrent_usage > (current_size * config_.expand_threshold);
40 bool growing_trend = pattern.is_growing && pattern.growth_rate > 0.1;
41
42 return low_hit_rate || high_utilization || growing_trend;
43}

45bool AdaptivePoolSizer::should_shrink_pool(const PoolUsagePattern& pattern, size_t current_size) const {
46 if (current_size <= config_.min_pool_size) {
47 return false;
48 }
49
50 // Shrink if utilization is consistently low
51 bool low_utilization = pattern.peak_concurrent_usage < (current_size * config_.shrink_threshold);
52 bool declining_trend = !pattern.is_growing && pattern.growth_rate < -0.1;
53 bool high_efficiency = pattern.efficiency_score > 0.9; // Already very efficient
54
55 return low_utilization && (declining_trend || high_efficiency);

58size_t AdaptivePoolSizer::predict_future_size(const PoolUsagePattern& pattern,
59 std::chrono::minutes time_horizon) const {
60 if (!config_.enable_predictive_sizing) {
61 return calculate_optimal_size(pattern);
62 }
63
64 // Simple predictive model based on growth rate
65 double time_factor = time_horizon.count() / 60.0; // Convert to hours
66 double predicted_peak = pattern.peak_concurrent_usage * (1.0 + pattern.growth_rate * time_factor);
67
68 // Add buffer for volatility
69 if (pattern.is_volatile) {
70 predicted_peak *= 1.3; // 30% buffer for volatile patterns
71 }
72
73 size_t predicted_size = static_cast<size_t>(std::ceil(predicted_peak));
74 return std::clamp(predicted_size, config_.min_pool_size, config_.max_pool_size);
75}

115size_t AdaptivePoolSizer::calculate_predictive_size(const PoolUsagePattern& pattern) const {
116 // Predictive: use more sophisticated analysis
117
118 // Base prediction on recent trends
119 size_t trend_based_size = predict_future_size(pattern, config_.adaptation_window);
120
121 // Factor in allocation patterns
122 double allocation_factor = 1.0;
123 if (pattern.allocation_rate > 0) {
124 allocation_factor = 1.0 + (pattern.allocation_rate / std::max(1.0, pattern.deallocation_rate) - 1.0) * 0.3;
125 }
126
127 // Consider efficiency
128 double efficiency_factor = 1.0;
129 if (pattern.efficiency_score < 0.8) {
130 efficiency_factor = 1.0 + (0.8 - pattern.efficiency_score);
131 }
132
133 size_t predicted_size = static_cast<size_t>(trend_based_size * allocation_factor * efficiency_factor);
134 return std::clamp(predicted_size, config_.min_pool_size, config_.max_pool_size);
135}

148
149 auto buffer = BufferPool::acquire();
150
151 auto end_time = std::chrono::high_resolution_clock::now();
152 auto acquire_time = std::chrono::duration_cast<std::chrono::nanoseconds>(end_time - start_time);
153
154 // Record allocation
155 record_allocation();
156
157 // Update performance metrics
158 {

160 acquire_times_.push_back(acquire_time);
161 if (!buffer) {

164 }
165
166 // Check for automatic adaptation
167 if (auto_adaptation_enabled_.load()) {

171 if (time_since_adaptation > adaptation_interval_) {
172 adapt_pool_size();
173 last_adaptation_ = now;
174 }
175 }

216void AdaptiveBufferPool::adapt_pool_size() {
217 update_usage_statistics();
218
219 auto pattern = get_usage_pattern();
220 size_t current_size = total_buffers();
221
222 if (sizer_.should_expand_pool(pattern, current_size)) {
223 size_t optimal_size = sizer_.calculate_optimal_size(pattern);
224 if (optimal_size > current_size) {
225 size_t additional = optimal_size - current_size;
226 auto expand_result = expand_pool(additional);
227 if (expand_result) {
228 std::lock_guard<std::mutex> lock(perf_mutex_);
229 performance_metrics_.adaptations_performed++;
230 }
231 }
232 } else if (sizer_.should_shrink_pool(pattern, current_size)) {
233 size_t optimal_size = sizer_.calculate_optimal_size(pattern);
234 if (optimal_size < current_size && optimal_size >= sizer_.get_config().min_pool_size) {
235 auto shrink_result = shrink_pool(optimal_size);
236 if (shrink_result) {
237 std::lock_guard<std::mutex> lock(perf_mutex_);
238 performance_metrics_.adaptations_performed++;
239 }
240 }
241 }

328void AdaptiveBufferPool::calculate_usage_rates() {
329 auto now = std::chrono::steady_clock::now();
330 auto window = sizer_.get_config().adaptation_window;
331 auto cutoff_time = now - window;
332
333 // Calculate allocation rate
334 size_t recent_allocations = std::count_if(allocation_timestamps_.begin(), allocation_timestamps_.end(),
335 [cutoff_time](const auto& timestamp) {
336 return timestamp >= cutoff_time;
337 });
338
339 // Calculate deallocation rate
340 size_t recent_deallocations = std::count_if(deallocation_timestamps_.begin(), deallocation_timestamps_.end(),
341 [cutoff_time](const auto& timestamp) {
342 return timestamp >= cutoff_time;
343 });
344
345 double window_minutes = static_cast<double>(window.count());
346 current_pattern_.allocation_rate = recent_allocations / window_minutes;
347 current_pattern_.deallocation_rate = recent_deallocations / window_minutes;
348}

350void AdaptiveBufferPool::analyze_usage_trends() {
351 if (allocation_timestamps_.size() < 10) {
352 return; // Not enough data for trend analysis
353 }
354
355 // Simple trend analysis: compare first half vs second half of recent data
356 size_t half_point = allocation_timestamps_.size() / 2;
357 size_t first_half_count = half_point;
358 size_t second_half_count = allocation_timestamps_.size() - half_point;
359
360 double first_half_rate = static_cast<double>(first_half_count);
361 double second_half_rate = static_cast<double>(second_half_count);
362
363 current_pattern_.is_growing = second_half_rate > first_half_rate * 1.1; // 10% threshold
364 current_pattern_.growth_rate = (second_half_rate - first_half_rate) / first_half_rate;
365
366 // Check for volatility (high variance in allocation timing)
367 if (allocation_timestamps_.size() >= 5) {
368 std::vector<double> intervals;
369 for (size_t i = 1; i < allocation_timestamps_.size(); ++i) {
370 auto interval = allocation_timestamps_[i] - allocation_timestamps_[i-1];
371 intervals.push_back(std::chrono::duration<double>(interval).count());
372 }
373
374 double mean = std::accumulate(intervals.begin(), intervals.end(), 0.0) / intervals.size();
375 double variance = 0.0;
376 for (double interval : intervals) {
377 variance += std::pow(interval - mean, 2);
378 }
379 variance /= intervals.size();
380
381 double coefficient_of_variation = std::sqrt(variance) / std::max(mean, 0.001);
382 current_pattern_.is_volatile = coefficient_of_variation > 1.0; // High variability
383 }
384}

386void AdaptiveBufferPool::cleanup_old_timestamps() {
387 auto now = std::chrono::steady_clock::now();
388 auto cutoff_time = now - std::chrono::hours(1); // Keep 1 hour of history
389
390 // Remove old allocation timestamps
391 allocation_timestamps_.erase(
392 std::remove_if(allocation_timestamps_.begin(), allocation_timestamps_.end(),
393 [cutoff_time](const auto& timestamp) { return timestamp < cutoff_time; }),
394 allocation_timestamps_.end());
395
396 // Remove old deallocation timestamps
397 deallocation_timestamps_.erase(
398 std::remove_if(deallocation_timestamps_.begin(), deallocation_timestamps_.end(),
399 [cutoff_time](const auto& timestamp) { return timestamp < cutoff_time; }),
400 deallocation_timestamps_.end());
401
402 // Limit buffer lifetime history
403 if (buffer_lifetimes_.size() > 1000) {
404 buffer_lifetimes_.erase(buffer_lifetimes_.begin(), buffer_lifetimes_.begin() + 500);
405 }
