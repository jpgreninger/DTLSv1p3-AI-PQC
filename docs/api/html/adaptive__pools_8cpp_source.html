36bool AdaptivePoolSizer::should_expand_pool(const PoolUsagePattern& pattern, size_t current_size) const {
37 if (current_size >= config_.max_pool_size) {
38 return false;
39 }
40
41 // Expand if hit rate is low or peak usage exceeds threshold
42 bool low_hit_rate = pattern.hit_rate < 0.8;
43 bool high_utilization = pattern.peak_concurrent_usage > (current_size * config_.expand_threshold);
44 bool growing_trend = pattern.is_growing && pattern.growth_rate > 0.1;
45
46 return low_hit_rate || high_utilization || growing_trend;
47}

49bool AdaptivePoolSizer::should_shrink_pool(const PoolUsagePattern& pattern, size_t current_size) const {
50 if (current_size <= config_.min_pool_size) {
51 return false;
52 }
53
54 // Shrink if utilization is consistently low
55 bool low_utilization = pattern.peak_concurrent_usage < (current_size * config_.shrink_threshold);
56 bool declining_trend = !pattern.is_growing && pattern.growth_rate < -0.1;
57 bool high_efficiency = pattern.efficiency_score > 0.9; // Already very efficient
58
59 return low_utilization && (declining_trend || high_efficiency);
60}

62size_t AdaptivePoolSizer::predict_future_size(const PoolUsagePattern& pattern,
63 std::chrono::minutes time_horizon) const {
64 if (!config_.enable_predictive_sizing) {
65 return calculate_optimal_size(pattern);
66 }
67
68 // Simple predictive model based on growth rate
69 double time_factor = time_horizon.count() / 60.0; // Convert to hours
70 double predicted_peak = pattern.peak_concurrent_usage * (1.0 + pattern.growth_rate * time_factor);
71
72 // Add buffer for volatility
73 if (pattern.is_volatile) {
74 predicted_peak *= 1.3; // 30% buffer for volatile patterns
75 }
76
77 size_t predicted_size = static_cast<size_t>(std::ceil(predicted_peak));
78 return std::clamp(predicted_size, config_.min_pool_size, config_.max_pool_size);

87size_t AdaptivePoolSizer::calculate_balanced_size(const PoolUsagePattern& pattern) const {
88 // Balanced: consider peak usage, growth trends, and efficiency

90
91 if (pattern.is_growing) {
92 base_factor += pattern.growth_rate * 0.5;

100 return std::clamp(target_size, config_.min_pool_size, config_.max_pool_size);

119size_t AdaptivePoolSizer::calculate_predictive_size(const PoolUsagePattern& pattern) const {
120 // Predictive: use more sophisticated analysis
121
122 // Base prediction on recent trends
123 size_t trend_based_size = predict_future_size(pattern, config_.adaptation_window);
124
125 // Factor in allocation patterns
126 double allocation_factor = 1.0;
127 if (pattern.allocation_rate > 0) {
128 allocation_factor = 1.0 + (pattern.allocation_rate / std::max(1.0, pattern.deallocation_rate) - 1.0) * 0.3;
129 }
130
131 // Consider efficiency
132 double efficiency_factor = 1.0;
133 if (pattern.efficiency_score < 0.8) {
134 efficiency_factor = 1.0 + (0.8 - pattern.efficiency_score);
135 }
136
137 size_t predicted_size = static_cast<size_t>(trend_based_size * allocation_factor * efficiency_factor);
138 return std::clamp(predicted_size, config_.min_pool_size, config_.max_pool_size);
139}

220void AdaptiveBufferPool::adapt_pool_size() {
221 update_usage_statistics();
222
223 auto pattern = get_usage_pattern();
224 size_t current_size = total_buffers();
225
226 if (sizer_.should_expand_pool(pattern, current_size)) {
227 size_t optimal_size = sizer_.calculate_optimal_size(pattern);
228 if (optimal_size > current_size) {
229 size_t additional = optimal_size - current_size;
230 auto expand_result = expand_pool(additional);
231 if (expand_result) {
232 std::lock_guard<std::mutex> lock(perf_mutex_);
233 performance_metrics_.adaptations_performed++;
234 }
235 }
236 } else if (sizer_.should_shrink_pool(pattern, current_size)) {
237 size_t optimal_size = sizer_.calculate_optimal_size(pattern);
238 if (optimal_size < current_size && optimal_size >= sizer_.get_config().min_pool_size) {
239 auto shrink_result = shrink_pool(optimal_size);
240 if (shrink_result) {
241 std::lock_guard<std::mutex> lock(perf_mutex_);
242 performance_metrics_.adaptations_performed++;
243 }
244 }

332void AdaptiveBufferPool::calculate_usage_rates() {
333 auto now = std::chrono::steady_clock::now();
334 auto window = sizer_.get_config().adaptation_window;
335 auto cutoff_time = now - window;
336
337 // Calculate allocation rate
338 size_t recent_allocations = std::count_if(allocation_timestamps_.begin(), allocation_timestamps_.end(),
339 [cutoff_time](const auto& timestamp) {
340 return timestamp >= cutoff_time;
341 });
342
343 // Calculate deallocation rate
344 size_t recent_deallocations = std::count_if(deallocation_timestamps_.begin(), deallocation_timestamps_.end(),
345 [cutoff_time](const auto& timestamp) {
346 return timestamp >= cutoff_time;
347 });
348
349 double window_minutes = static_cast<double>(window.count());
350 current_pattern_.allocation_rate = recent_allocations / window_minutes;
351 current_pattern_.deallocation_rate = recent_deallocations / window_minutes;
352}

354void AdaptiveBufferPool::analyze_usage_trends() {
355 if (allocation_timestamps_.size() < 10) {
356 return; // Not enough data for trend analysis
357 }
358
359 // Simple trend analysis: compare first half vs second half of recent data
360 size_t half_point = allocation_timestamps_.size() / 2;
361 size_t first_half_count = half_point;
362 size_t second_half_count = allocation_timestamps_.size() - half_point;
363
364 double first_half_rate = static_cast<double>(first_half_count);
365 double second_half_rate = static_cast<double>(second_half_count);
366
367 current_pattern_.is_growing = second_half_rate > first_half_rate * 1.1; // 10% threshold
368 current_pattern_.growth_rate = (second_half_rate - first_half_rate) / first_half_rate;
369
370 // Check for volatility (high variance in allocation timing)
371 if (allocation_timestamps_.size() >= 5) {
372 std::vector<double> intervals;
373 for (size_t i = 1; i < allocation_timestamps_.size(); ++i) {
374 auto interval = allocation_timestamps_[i] - allocation_timestamps_[i-1];
375 intervals.push_back(std::chrono::duration<double>(interval).count());
376 }
377
378 double mean = std::accumulate(intervals.begin(), intervals.end(), 0.0) / intervals.size();
379 double variance = 0.0;
380 for (double interval : intervals) {
381 variance += std::pow(interval - mean, 2);
382 }
383 variance /= intervals.size();
384
385 double coefficient_of_variation = std::sqrt(variance) / std::max(mean, 0.001);
386 current_pattern_.is_volatile = coefficient_of_variation > 1.0; // High variability
387 }
388}

390void AdaptiveBufferPool::cleanup_old_timestamps() {
391 auto now = std::chrono::steady_clock::now();
392 auto cutoff_time = now - std::chrono::hours(1); // Keep 1 hour of history
393
394 // Remove old allocation timestamps
395 allocation_timestamps_.erase(
396 std::remove_if(allocation_timestamps_.begin(), allocation_timestamps_.end(),
397 [cutoff_time](const auto& timestamp) { return timestamp < cutoff_time; }),
398 allocation_timestamps_.end());
399
400 // Remove old deallocation timestamps
401 deallocation_timestamps_.erase(
402 std::remove_if(deallocation_timestamps_.begin(), deallocation_timestamps_.end(),
403 [cutoff_time](const auto& timestamp) { return timestamp < cutoff_time; }),
404 deallocation_timestamps_.end());
405
406 // Limit buffer lifetime history
407 if (buffer_lifetimes_.size() > 1000) {
408 buffer_lifetimes_.erase(buffer_lifetimes_.begin(), buffer_lifetimes_.begin() + 500);
409 }
