6bool AntiReplayCore::is_valid_sequence_number(uint64_t sequence_number, const WindowState& state) {
7 // First packet is always valid

9 return true;

30bool AntiReplayCore::update_window(uint64_t sequence_number, WindowState& state) {
31 // First packet
32 if (state.highest_sequence_number == 0) {
33 state.highest_sequence_number = sequence_number;
34 if (!state.window.empty()) {
35 state.window[0] = true;
36 }
37 state.received_count++;
38 return true;
39 }
40
41 const size_t window_size = state.window.size();

46 // Too old - this is an error condition

48 return false;

123void AntiReplayCore::slide_window(uint64_t new_highest, WindowState& state) {
124 if (new_highest <= state.highest_sequence_number || state.window.empty()) {
125 return;
126 }
127
128 const size_t window_size = state.window.size();
129 const uint64_t shift_amount = new_highest - state.highest_sequence_number;
130
131 if (shift_amount >= window_size) {
132 // Shift is larger than window - clear everything
133 std::fill(state.window.begin(), state.window.end(), false);
134 } else {
135 // Shift window contents
136 const size_t shift_size = static_cast<size_t>(shift_amount);
137
138 // Shift existing bits
139 for (size_t i = window_size - 1; i >= shift_size; --i) {
140 state.window[i] = state.window[i - shift_size];
141 }
142
143 // Clear the newly opened slots
144 std::fill(state.window.begin(), state.window.begin() + shift_size, false);
145 }
146
147 state.highest_sequence_number = new_highest;
148 // Mark the new highest sequence number as received
149 state.window[0] = true;
150}
