81Result<ProtectedRecord> HardwareAcceleratedRecordLayer::protect_record(
82 const PlaintextRecord& plaintext,
83 const ProtectionParams& params) {
84
85 if (!initialized_.load()) {
86 return Result<ProtectedRecord>::error(DTLSError::NOT_INITIALIZED);
87 }
88
89 auto start_time = std::chrono::high_resolution_clock::now();
90
91 // Check if we should use hardware-specific optimization
92 if (is_hardware_acceleration_active()) {
93 auto hw_result = protect_record_aes_gcm_hw(plaintext, params);
94 if (hw_result) {
95 auto end_time = std::chrono::high_resolution_clock::now();
96 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time);
97 update_metrics("protect_record", duration, true, false);
98 return hw_result;
99 }
100 }
101
102 // Build AEAD parameters
103 auto additional_data = build_additional_data(plaintext, params);
104 auto aead_params_result = build_aead_params(params, additional_data);
105 if (!aead_params_result) {
106 return Result<ProtectedRecord>::error(aead_params_result.error());
107 }
108
109 // Encrypt the record
110 crypto::AEADEncryptionParams enc_params;
111 enc_params.key = aead_params_result.value().key;
112 enc_params.nonce = aead_params_result.value().nonce;
113 enc_params.additional_data = aead_params_result.value().additional_data;
114 enc_params.cipher = aead_params_result.value().cipher;
115 enc_params.plaintext = plaintext.payload;
116
117 auto encrypt_result = crypto_provider_->encrypt_aead(enc_params);
118 if (!encrypt_result) {
119 return Result<ProtectedRecord>::error(encrypt_result.error());
120 }
121
122 // Build protected record
123 ProtectedRecord protected_record;
124 protected_record.content_type = ContentType::APPLICATION_DATA; // DTLS 1.3 always uses this
125 protected_record.version = DTLS_V13;
126 protected_record.epoch = params.epoch;
127 protected_record.sequence_number = params.sequence_number;
128
129 const auto& encryption_output = encrypt_result.value();
130 protected_record.payload.reserve(encryption_output.ciphertext.size() + encryption_output.tag.size());
131 protected_record.payload.insert(protected_record.payload.end(),
132 encryption_output.ciphertext.begin(),
133 encryption_output.ciphertext.end());
134 protected_record.payload.insert(protected_record.payload.end(),
135 encryption_output.tag.begin(),
136 encryption_output.tag.end());
137
138 auto end_time = std::chrono::high_resolution_clock::now();
139 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time);
140 update_metrics("protect_record", duration, false, false);
141
142 return Result<ProtectedRecord>::success(std::move(protected_record));
143}

145Result<PlaintextRecord> HardwareAcceleratedRecordLayer::unprotect_record(
146 const ProtectedRecord& protected_record,
147 const ProtectionParams& params) {
148
149 if (!initialized_.load()) {
150 return Result<PlaintextRecord>::error(DTLSError::NOT_INITIALIZED);
151 }
152
153 auto start_time = std::chrono::high_resolution_clock::now();
154
155 // Check if we should use hardware-specific optimization
156 if (is_hardware_acceleration_active()) {
157 auto hw_result = unprotect_record_aes_gcm_hw(protected_record, params);
158 if (hw_result) {
159 auto end_time = std::chrono::high_resolution_clock::now();
160 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time);
161 update_metrics("unprotect_record", duration, true, false);
162 return hw_result;
163 }
164 }
165
166 // Validate minimum payload size (at least tag size)
167 if (protected_record.payload.size() < 16) {
168 return Result<PlaintextRecord>::error(DTLSError::INVALID_RECORD);
169 }
170
171 // Split ciphertext and tag
172 size_t tag_size = 16; // AES-GCM tag size
173 size_t ciphertext_size = protected_record.payload.size() - tag_size;
174
175 std::vector<uint8_t> ciphertext(protected_record.payload.begin(),
176 protected_record.payload.begin() + ciphertext_size);
177 std::vector<uint8_t> tag(protected_record.payload.begin() + ciphertext_size,
178 protected_record.payload.end());
179
180 // Build AEAD parameters for decryption
181 PlaintextRecord temp_plaintext;
182 temp_plaintext.content_type = protected_record.content_type;
183 temp_plaintext.version = protected_record.version;
184 temp_plaintext.epoch = protected_record.epoch;
185 temp_plaintext.sequence_number = protected_record.sequence_number;
186
187 auto additional_data = build_additional_data(temp_plaintext, params);
188 auto aead_params_result = build_aead_params(params, additional_data);
189 if (!aead_params_result) {
190 return Result<PlaintextRecord>::error(aead_params_result.error());
191 }
192
193 // Decrypt the record
194 crypto::AEADDecryptionParams dec_params;
195 dec_params.key = aead_params_result.value().key;
196 dec_params.nonce = aead_params_result.value().nonce;
197 dec_params.additional_data = aead_params_result.value().additional_data;
198 dec_params.cipher = aead_params_result.value().cipher;
199 dec_params.ciphertext = ciphertext;
200 dec_params.tag = tag;
201
202 auto decrypt_result = crypto_provider_->decrypt_aead(dec_params);
203 if (!decrypt_result) {
204 return Result<PlaintextRecord>::error(decrypt_result.error());
205 }
206
207 // Extract inner content type from decrypted payload (DTLS 1.3)
208 const auto& plaintext_data = decrypt_result.value();
209 if (plaintext_data.empty()) {
210 return Result<PlaintextRecord>::error(DTLSError::INVALID_RECORD);
211 }
212
213 // Find the actual content type (last non-zero byte)
214 ContentType inner_content_type = ContentType::INVALID;
215 size_t payload_end = plaintext_data.size();
216
217 for (size_t i = plaintext_data.size(); i > 0; --i) {
218 if (plaintext_data[i - 1] != 0) {
219 inner_content_type = static_cast<ContentType>(plaintext_data[i - 1]);
220 payload_end = i - 1;
221 break;
222 }
223 }
224
225 if (inner_content_type == ContentType::INVALID) {
226 return Result<PlaintextRecord>::error(DTLSError::INVALID_RECORD);
227 }
228
229 // Build plaintext record
230 PlaintextRecord plaintext_record;
231 plaintext_record.content_type = inner_content_type;
232 plaintext_record.version = protected_record.version;
233 plaintext_record.epoch = protected_record.epoch;
234 plaintext_record.sequence_number = protected_record.sequence_number;
235 plaintext_record.payload = std::vector<uint8_t>(plaintext_data.begin(),
236 plaintext_data.begin() + payload_end);
237
238 auto end_time = std::chrono::high_resolution_clock::now();
239 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time);
240 update_metrics("unprotect_record", duration, false, false);
241
242 return Result<PlaintextRecord>::success(std::move(plaintext_record));
243}

251 const std::vector<PlaintextRecord>& plaintexts,
252 const std::vector<ProtectionParams>& params) {
253
254 if (plaintexts.size() != params.size()) {
255 return Result<std::vector<ProtectedRecord>>::error(DTLSError::INVALID_PARAMETER);
256 }
257
258 if (!hardware_config_.enable_batch_processing || plaintexts.size() < 2) {
259 // Fallback to sequential processing

261 results.reserve(plaintexts.size());

264 auto result = protect_record(plaintexts[i], params[i]);
265 if (!result) {

267 }
268 results.push_back(result.value());
269 }

272 }
273
274 // Use SIMD batch processing if available
275 auto start_time = std::chrono::high_resolution_clock::now();
276 auto batch_result = protect_batch_simd(plaintexts, params);
277 auto end_time = std::chrono::high_resolution_clock::now();

279 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time);
280 update_metrics("protect_records_batch", duration, true, true);

439std::vector<uint8_t> HardwareAcceleratedRecordLayer::build_additional_data(
440 const PlaintextRecord& plaintext,
441 const ProtectionParams& params) const {
442
443 // DTLS 1.3 additional data format
444 std::vector<uint8_t> additional_data;
445 additional_data.reserve(13);
446
447 // Sequence number (8 bytes)
448 for (int i = 7; i >= 0; --i) {
449 additional_data.push_back((plaintext.sequence_number >> (i * 8)) & 0xFF);
450 }
451
452 // Content type (1 byte) - always APPLICATION_DATA for DTLS 1.3
453 additional_data.push_back(static_cast<uint8_t>(ContentType::APPLICATION_DATA));
454
455 // Protocol version (2 bytes)
456 additional_data.push_back((plaintext.version >> 8) & 0xFF);
457 additional_data.push_back(plaintext.version & 0xFF);
458
459 // Length (2 bytes) - will be filled by crypto provider
460 additional_data.push_back(0);
461 additional_data.push_back(0);
462
463 return additional_data;
464}

506Result<std::vector<ProtectedRecord>> HardwareAcceleratedRecordLayer::protect_batch_simd(
507 const std::vector<PlaintextRecord>& plaintexts,
508 const std::vector<ProtectionParams>& params) {
509
510 if (!zero_copy_crypto_) {
511 // Fallback to sequential processing
512 std::vector<ProtectedRecord> results;
513 results.reserve(plaintexts.size());
514
515 for (size_t i = 0; i < plaintexts.size(); ++i) {
516 auto result = protect_record(plaintexts[i], params[i]);
517 if (!result) {
518 return Result<std::vector<ProtectedRecord>>::error(result.error());
519 }
520 results.push_back(result.value());
521 }
522
523 return Result<std::vector<ProtectedRecord>>::success(std::move(results));
524 }
525
526 // Prepare batch parameters
527 std::vector<crypto::AEADEncryptionParams> batch_params;
528 batch_params.reserve(plaintexts.size());
529
530 for (size_t i = 0; i < plaintexts.size(); ++i) {
531 auto additional_data = build_additional_data(plaintexts[i], params[i]);
532
533 crypto::AEADEncryptionParams enc_params;
534 enc_params.key = params[i].key;
535 enc_params.nonce = params[i].nonce;
536 enc_params.additional_data = additional_data;
537 enc_params.cipher = params[i].cipher;
538 enc_params.plaintext = plaintexts[i].payload;
539
540 batch_params.push_back(std::move(enc_params));
541 }
542
543 // Use provider's batch encryption
544 auto batch_result = crypto_provider_->batch_encrypt_aead(batch_params);
545 if (!batch_result) {
546 return Result<std::vector<ProtectedRecord>>::error(batch_result.error());
547 }
548
549 // Convert results to protected records
550 std::vector<ProtectedRecord> protected_records;
551 protected_records.reserve(plaintexts.size());
552
553 const auto& encryption_outputs = batch_result.value();
554 for (size_t i = 0; i < plaintexts.size(); ++i) {
555 ProtectedRecord protected_record;
556 protected_record.content_type = ContentType::APPLICATION_DATA;
557 protected_record.version = DTLS_V13;
558 protected_record.epoch = params[i].epoch;
559 protected_record.sequence_number = params[i].sequence_number;
560
561 const auto& encryption_output = encryption_outputs[i];
562 protected_record.payload.reserve(encryption_output.ciphertext.size() + encryption_output.tag.size());
563 protected_record.payload.insert(protected_record.payload.end(),
564 encryption_output.ciphertext.begin(),
565 encryption_output.ciphertext.end());
566 protected_record.payload.insert(protected_record.payload.end(),
567 encryption_output.tag.begin(),
568 encryption_output.tag.end());
569
570 protected_records.push_back(std::move(protected_record));
571 }
572
573 return Result<std::vector<ProtectedRecord>>::success(std::move(protected_records));
574}

629Result<HardwareAcceleratedRecordLayer::HardwareConfig>
630HardwareAcceleratedRecordLayerFactory::get_optimal_config() {
631 auto hw_profile_result = crypto::HardwareAccelerationDetector::detect_capabilities();
632 if (!hw_profile_result) {
633 return Result<HardwareAcceleratedRecordLayer::HardwareConfig>::error(hw_profile_result.error());
634 }
635
636 const auto& profile = hw_profile_result.value();
637 HardwareAcceleratedRecordLayer::HardwareConfig config;
638
639 // Configure based on detected hardware
640 bool has_simd = std::any_of(profile.capabilities.begin(), profile.capabilities.end(),
641 [](const auto& cap) {
642 return (cap.capability == crypto::HardwareCapability::AVX ||
643 cap.capability == crypto::HardwareCapability::AVX2) &&
644 cap.available && cap.enabled;
645 });
646
