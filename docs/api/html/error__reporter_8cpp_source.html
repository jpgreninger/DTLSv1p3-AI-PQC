24Result<void> ErrorReporter::report_error(LogLevel level,
25 DTLSError error,
26 const std::string& category,
27 const std::string& message,
28 std::shared_ptr<ErrorContext> context) {
29 // Check minimum log level
30 if (level < config_.minimum_level) {
31 return make_result();
32 }
33
34 // Check rate limits and update counters atomically
35 uint32_t max_per_second, max_per_minute;
36 {
37 // Get config values under lock
38 std::lock_guard<std::mutex> config_lock(config_mutex_);
39 max_per_second = config_.max_reports_per_second;
40 max_per_minute = config_.max_reports_per_minute;
41 }
42
43 {
44 std::lock_guard<std::mutex> lock(rate_limit_.reset_mutex);
45
46 auto now = std::chrono::steady_clock::now();
47
48 // Reset counters if time window has passed
49 if (now - rate_limit_.second_start >= std::chrono::seconds(1)) {
50 rate_limit_.reports_this_second = 0;
51 rate_limit_.second_start = now;
52 }
53
54 if (now - rate_limit_.minute_start >= std::chrono::minutes(1)) {
55 rate_limit_.reports_this_minute = 0;
56 rate_limit_.minute_start = now;
57 }
58
59 // Check limits BEFORE incrementing
60 if (rate_limit_.reports_this_second >= max_per_second ||
61 rate_limit_.reports_this_minute >= max_per_minute) {
62 stats_.rate_limited_reports++;
63 return make_error<void>(DTLSError::RATE_LIMITED);
64 }
65
66 // Update counters
67 rate_limit_.reports_this_second++;
68 rate_limit_.reports_this_minute++;
69 }
70
71 // Create error report
72 ErrorReport report(level, error, message);
73 report.category = category;
74 report.sensitivity = SensitivityLevel::INTERNAL;
75
76 // Add context information if available
77 if (context) {
78 report.connection_context = anonymize_connection_id(
79 context->get_connection_info().connection_id_hash);
80 }
81
82 // Update statistics
83 update_statistics(report);
84
85 // Write the report
86 auto result = write_report(report);
87 if (!result.is_success()) {
88 stats_.failed_reports++;
89 return result;
90 }
91
92 stats_.total_reports++;
93 return make_result();
94}
