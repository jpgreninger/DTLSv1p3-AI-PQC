482void LeakDetector::periodic_detection_loop() {
483 LeakDetectionConfig local_config;
484
485 while (periodic_thread_running_.load()) {
486 // Get current configuration
487 {
488 std::lock_guard<std::mutex> lock(config_mutex_);
489 local_config = config_;
490 }
491
492 if (!local_config.enable_periodic_checks) {
493 std::this_thread::sleep_for(local_config.check_interval);
494 continue;
495 }
496
497 // Perform leak detection
498 auto leak_report_result = detect_leaks();
499 if (leak_report_result) {
500 auto leak_report = *leak_report_result;
501
502 // If automatic cleanup is enabled and we found leaks
503 if (local_config.enable_automatic_cleanup && leak_report.total_leaks > 0) {
504 cleanup_leaked_resources();
505 }
506
507 // Log critical leaks (in a real implementation, this would use proper logging)
508 if (leak_report.critical_leaks > 0) {
509 std::cerr << "DTLS CRITICAL MEMORY LEAKS DETECTED: " << leak_report.critical_leaks << " leaks\n";
510 }
511 }
512
513 std::this_thread::sleep_for(local_config.check_interval);
514 }

517bool LeakDetector::is_resource_leaked(const ResourceInfo& info, std::chrono::steady_clock::time_point now) const {
518 std::lock_guard<std::mutex> lock(config_mutex_);
519
520 auto age = now - info.allocation_time;
521 auto access_age = now - info.last_access_time;
522
523 if (info.is_critical) {
524 return age > config_.critical_resource_age;
525 }
526
527 // Consider a resource leaked if:
528 // 1. It's older than max_resource_age AND hasn't been accessed recently
529 // 2. Or if it hasn't been accessed for a very long time
530 return (age > config_.max_resource_age && access_age > config_.max_resource_age / 2) ||
531 access_age > config_.max_resource_age * 2;
532}

674bool ResourceCleanupManager::should_cleanup_resource(const ResourceInfo& info, CleanupPolicy policy) const {
675 auto now = std::chrono::steady_clock::now();
676 auto age = now - info.allocation_time;
677 auto access_age = now - info.last_access_time;
678
679 switch (policy) {
680 case CleanupPolicy::CONSERVATIVE:
681 // Only clean obviously leaked resources (very old and unused)
682 return age > std::chrono::hours(2) && access_age > std::chrono::hours(1);
683
684 case CleanupPolicy::MODERATE:
685 // Clean resources that seem to be leaked
686 return age > std::chrono::minutes(30) && access_age > std::chrono::minutes(15);
687
688 case CleanupPolicy::AGGRESSIVE:
689 // Clean all old resources
690 return age > std::chrono::minutes(5) || access_age > std::chrono::minutes(10);
691
692 default:
693 return false;
694 }
695}
