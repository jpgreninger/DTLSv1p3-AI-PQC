772 bool evaluate_condition(const AlertCondition& condition, double current_value) {
773 switch (condition.operator_type) {
774 case AlertCondition::Operator::GREATER_THAN:
775 return current_value > condition.threshold_value;
776 case AlertCondition::Operator::LESS_THAN:
777 return current_value < condition.threshold_value;
778 case AlertCondition::Operator::EQUALS:
779 return std::abs(current_value - condition.threshold_value) < 0.001;
780 case AlertCondition::Operator::NOT_EQUALS:
781 return std::abs(current_value - condition.threshold_value) >= 0.001;
782 case AlertCondition::Operator::RATE_INCREASE:
783 // For rate increase, check if current value increased by more than threshold
784 {
785 auto history_it = metric_history_.find(condition.metric_name);
786 if (history_it != metric_history_.end() && history_it->second.size() >= 2) {
787 double current_rate = current_value;
788 double previous_rate = history_it->second.rbegin()[1].value; // Second-to-last value
789 double rate_change = current_rate - previous_rate;
790 return rate_change > condition.threshold; // Positive change greater than threshold
791 }
792 return false; // Not enough history
793 }
794 case AlertCondition::Operator::RATE_DECREASE:
795 // For rate decrease, check if current value decreased by more than threshold
796 {
797 auto history_it = metric_history_.find(condition.metric_name);
798 if (history_it != metric_history_.end() && history_it->second.size() >= 2) {
799 double current_rate = current_value;
800 double previous_rate = history_it->second.rbegin()[1].value; // Second-to-last value
801 double rate_change = previous_rate - current_rate; // Note: reversed for decrease
802 return rate_change > condition.threshold; // Decrease greater than threshold
803 }
804 return false; // Not enough history
805 }
806 default:
807 return false;
808 }
