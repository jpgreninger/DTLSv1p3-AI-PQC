165std::string to_string(const NetworkAddress& addr) {
166 std::ostringstream oss;
167
168 if (addr.is_ipv4()) {
169 // IPv4 address format: a.b.c.d:port
170 oss << static_cast<int>(addr.address[0]) << "."
171 << static_cast<int>(addr.address[1]) << "."
172 << static_cast<int>(addr.address[2]) << "."
173 << static_cast<int>(addr.address[3]) << ":"
174 << addr.port;
175 } else if (addr.is_ipv6()) {
176 // IPv6 address format: [a:b:c:d:e:f:g:h]:port
177 oss << "[";
178 for (size_t i = 0; i < 16; i += 2) {
179 if (i > 0) oss << ":";
180 uint16_t segment = (static_cast<uint16_t>(addr.address[i]) << 8) |
181 static_cast<uint16_t>(addr.address[i + 1]);
182 oss << std::hex << segment;
183 }
184 oss << "]:" << std::dec << addr.port;
185 } else {
186 oss << "UNKNOWN_ADDRESS_FAMILY:" << addr.port;
187 }
188
189 return oss.str();
190}

277NetworkAddress NetworkAddress::from_ipv4(uint32_t ipv4_addr, uint16_t port_num) {
278 NetworkAddress addr;
279 addr.family = Family::IPv4;
280 addr.port = port_num;
281 addr.address.fill(0);
282 addr.address[0] = static_cast<uint8_t>((ipv4_addr >> 24) & 0xFF);
283 addr.address[1] = static_cast<uint8_t>((ipv4_addr >> 16) & 0xFF);
284 addr.address[2] = static_cast<uint8_t>((ipv4_addr >> 8) & 0xFF);
285 addr.address[3] = static_cast<uint8_t>(ipv4_addr & 0xFF);
286 return addr;

331NetworkAddress::NetworkAddress(const std::string& ip, uint16_t port_num) : port(port_num) {
332 // Simple IP parsing - in production, use a more robust parser
333 if (ip.find(':') != std::string::npos) {
334 // IPv6 - simplified parsing
335 family = Family::IPv6;
336 address.fill(0);
337 // For now, just fill with zeros - real implementation would parse IPv6
338 } else {
339 // IPv4
340 family = Family::IPv4;
341 address.fill(0);
342
343 // Parse IPv4 address
344 std::istringstream iss(ip);
345 std::string token;
346 int i = 0;
347 while (std::getline(iss, token, '.') && i < 4) {
348 address[i] = static_cast<uint8_t>(std::stoi(token));
349 i++;
350 }
351 }

354Result<NetworkAddress> NetworkAddress::from_string(const std::string& address_with_port) {
355 size_t last_colon = address_with_port.rfind(':');
356 if (last_colon == std::string::npos) {
357 return make_error<NetworkAddress>(DTLSError::INVALID_PARAMETER, "No port separator found");
358 }
359
360 std::string ip = address_with_port.substr(0, last_colon);
361 std::string port_str = address_with_port.substr(last_colon + 1);
362
363 try {
364 uint16_t port_num = static_cast<uint16_t>(std::stoi(port_str));
365 NetworkAddress addr(ip, port_num);
366 return make_result(std::move(addr));
367 } catch (const std::exception&) {
368 return make_error<NetworkAddress>(DTLSError::INVALID_PARAMETER, "Invalid port number");
369 }
370}
