75CPUInfo detect_x86_features() {
76 CPUInfo info;
77 uint32_t eax, ebx, ecx, edx;
78
79 // Get vendor string
80 get_cpuid(0, 0, &eax, &ebx, &ecx, &edx);
81 char vendor[13] = {0};
82 memcpy(vendor, &ebx, 4);
83 memcpy(vendor + 4, &edx, 4);
84 memcpy(vendor + 8, &ecx, 4);
85 info.vendor = vendor;
86
87 // Get basic CPU info
88 get_cpuid(1, 0, &eax, &ebx, &ecx, &edx);
89 info.family = (eax >> 8) & 0xF;
90 info.model_id = (eax >> 4) & 0xF;
91 info.stepping = eax & 0xF;
92
93 // Extended family/model for newer CPUs
94 if (info.family == 0xF) {
95 info.family += (eax >> 20) & 0xFF;
96 }
97 if (info.family == 0x6 || info.family == 0xF) {
98 info.model_id += ((eax >> 16) & 0xF) << 4;
99 }
100
101 // Feature detection
102 // SSE features
103 if (edx & (1 << 26)) info.capabilities.insert(HardwareCapability::SSE2);
104 if (ecx & (1 << 0)) info.capabilities.insert(HardwareCapability::SSE3);
105 if (ecx & (1 << 19)) info.capabilities.insert(HardwareCapability::SSE4_1);
106 if (ecx & (1 << 20)) info.capabilities.insert(HardwareCapability::SSE4_2);
107
108 // AES-NI
109 if (ecx & (1 << 25)) info.capabilities.insert(HardwareCapability::AES_NI);
110
111 // PCLMULQDQ
112 if (ecx & (1 << 1)) info.capabilities.insert(HardwareCapability::PCLMULQDQ);
113
114 // AVX
115 if (ecx & (1 << 28)) info.capabilities.insert(HardwareCapability::AVX);
116
117 // Extended features
118 get_cpuid(7, 0, &eax, &ebx, &ecx, &edx);
119 if (ebx & (1 << 5)) info.capabilities.insert(HardwareCapability::AVX2);
120
121 // Get brand string
122 char brand[49] = {0};
123 get_cpuid(0x80000002, 0, &eax, &ebx, &ecx, &edx);
124 memcpy(brand, &eax, 16);
125 get_cpuid(0x80000003, 0, &eax, &ebx, &ecx, &edx);
126 memcpy(brand + 16, &eax, 16);
127 get_cpuid(0x80000004, 0, &eax, &ebx, &ecx, &edx);
128 memcpy(brand + 32, &eax, 16);
129 info.brand = brand;
130
131 return info;
132}

147 info.capabilities.insert(HardwareCapability::ARM_AES);

149 if (line.find("sha1") != std::string::npos) {
150 info.capabilities.insert(HardwareCapability::ARM_SHA1);

152 if (line.find("sha2") != std::string::npos) {
153 info.capabilities.insert(HardwareCapability::ARM_SHA2);
154 }
155 if (line.find("neon") != std::string::npos) {
156 info.capabilities.insert(HardwareCapability::ARM_NEON);
157 }
158 }
159 if (line.find("model name") != std::string::npos) {
160 size_t colon = line.find(':');
161 if (colon != std::string::npos) {
162 info.brand = line.substr(colon + 2);
163 }
164 }
165 }
166#endif
167
168 // Use getauxval for feature detection if available
169#ifdef AT_HWCAP
170 unsigned long hwcap = getauxval(AT_HWCAP);
171 if (hwcap & HWCAP_AES) info.capabilities.insert(HardwareCapability::ARM_AES);
172 if (hwcap & HWCAP_SHA1) info.capabilities.insert(HardwareCapability::ARM_SHA1);
173 if (hwcap & HWCAP_SHA2) info.capabilities.insert(HardwareCapability::ARM_SHA2);
174 if (hwcap & HWCAP_ASIMD) info.capabilities.insert(HardwareCapability::ARM_NEON);
175#endif
176
177 return info;

247Result<HardwareAccelerationProfile> HardwareAccelerationDetector::detect_capabilities() {
248 HardwareAccelerationProfile profile;
249
250 try {
251 // Detect CPU features
252#if defined(__x86_64__) || defined(__i386__)
253 auto cpu_info = detect_x86_features();
254#elif defined(__aarch64__) || defined(__arm__)
255 auto cpu_info = detect_arm_features();
256#else
257 auto cpu_info = detect_generic_features();
258#endif
259
260 profile.cpu_model = cpu_info.brand;
261 profile.platform_name = HardwareAccelerationDetector::get_platform_info();
262 profile.os_version = HardwareAccelerationDetector::get_platform_info();
263
264 // Convert CPU capabilities to hardware capability statuses
265 for (const auto& cap : cpu_info.capabilities) {
266 HardwareCapabilityStatus status;
267 status.capability = cap;
268 status.available = true;
269 status.enabled = true;
270 status.performance_multiplier = 1.0f;
271
272 switch (cap) {
273 case HardwareCapability::AES_NI:
274 status.description = "Intel AES New Instructions";
275 status.performance_multiplier = benchmark_aes_performance();
276 break;
277 case HardwareCapability::AVX:
278 status.description = "Advanced Vector Extensions";
279 status.performance_multiplier = 1.8f;
280 break;
281 case HardwareCapability::AVX2:
282 status.description = "Advanced Vector Extensions 2";
283 status.performance_multiplier = 2.2f;
284 break;
285 case HardwareCapability::SSE2:
286 status.description = "Streaming SIMD Extensions 2";
287 status.performance_multiplier = 1.3f;
288 break;
289 case HardwareCapability::SSE4_1:
290 status.description = "Streaming SIMD Extensions 4.1";
291 status.performance_multiplier = 1.4f;
292 break;
293 case HardwareCapability::SSE4_2:
294 status.description = "Streaming SIMD Extensions 4.2";
295 status.performance_multiplier = 1.5f;
296 break;
297 case HardwareCapability::PCLMULQDQ:
298 status.description = "Carry-less Multiplication";
299 status.performance_multiplier = 2.0f;
300 break;
301 case HardwareCapability::ARM_AES:
302 status.description = "ARM AES Instructions";
303 status.performance_multiplier = 2.8f;
304 break;
305 case HardwareCapability::ARM_SHA1:
306 status.description = "ARM SHA1 Instructions";
307 status.performance_multiplier = 2.2f;
308 break;
309 case HardwareCapability::ARM_SHA2:
310 status.description = "ARM SHA2 Instructions";
311 status.performance_multiplier = 2.4f;
312 break;
313 case HardwareCapability::ARM_NEON:
314 status.description = "ARM NEON SIMD";
315 status.performance_multiplier = 1.9f;
316 break;
317 default:
318 status.description = "Unknown capability";
319 status.performance_multiplier = 1.1f;
320 break;
321 }
322
323 profile.capabilities.push_back(status);
324 }
325
326 // Check for TPM
327 if (detect_tpm_availability()) {
328 HardwareCapabilityStatus tpm_status;
329 tpm_status.capability = HardwareCapability::TPM_2_0;
330 tpm_status.available = true;
331 tpm_status.enabled = true;
332 tpm_status.description = "Trusted Platform Module 2.0";
333 tpm_status.performance_multiplier = 1.0f;
334 profile.capabilities.push_back(tpm_status);
335 }
336
337 // Check for hardware RNG
338 if (HardwareAccelerationDetector::detect_hardware_rng()) {
339 HardwareCapabilityStatus rng_status;
340 rng_status.capability = HardwareCapability::RNG_HARDWARE;
341 rng_status.available = true;
342 rng_status.enabled = true;
343 rng_status.description = "Hardware Random Number Generator";
344 rng_status.performance_multiplier = 5.0f; // RNG is much faster than software
345 profile.capabilities.push_back(rng_status);
346 }
347
348 // Set overall status
349 profile.has_any_acceleration = !profile.capabilities.empty();
350
351 // Calculate overall performance score
352 float total_score = 1.0f;
353 for (const auto& cap : profile.capabilities) {
354 if (cap.enabled) {
355 total_score *= cap.performance_multiplier;
356 }
357 }
358 profile.overall_performance_score = std::min(total_score, 10.0f); // Cap at 10x
359
360 // Generate recommendations
361 std::ostringstream recommendations;
362 if (profile.has_any_acceleration) {
363 recommendations << "Hardware acceleration detected. ";
364 if (std::any_of(profile.capabilities.begin(), profile.capabilities.end(),
365 [](const auto& cap) { return cap.capability == HardwareCapability::AES_NI; })) {
366 recommendations << "Use AES-GCM cipher suites for optimal performance. ";
367 }
368 if (std::any_of(profile.capabilities.begin(), profile.capabilities.end(),
369 [](const auto& cap) { return cap.capability == HardwareCapability::ARM_AES; })) {
370 recommendations << "ARM AES acceleration available - use AES-based cipher suites. ";
371 }
372 } else {
373 recommendations << "No hardware acceleration detected. Consider software optimizations. ";
374 }
375
376 profile.recommendations = recommendations.str();
377
378 return make_result(std::move(profile));
379
380 } catch (const std::exception& e) {
381 return make_error<HardwareAccelerationProfile>(DTLSError::INTERNAL_ERROR);
382 }

747std::vector<CipherSuite> get_hardware_optimized_cipher_suites() {
748 std::vector<CipherSuite> suites;
749
750 auto detection_result = HardwareAccelerationDetector::detect_capabilities();
751 if (!detection_result) {
752 // Fallback to default suites
753 suites.push_back(CipherSuite::TLS_AES_128_GCM_SHA256);
754 return suites;
755 }
756
757 const auto& profile = detection_result.value();
758
759 bool has_aes = std::any_of(profile.capabilities.begin(), profile.capabilities.end(),
760 [](const auto& cap) {
761 return (cap.capability == HardwareCapability::AES_NI ||
762 cap.capability == HardwareCapability::ARM_AES) &&
763 cap.available && cap.enabled;
764 });
765
766 if (has_aes) {
767 // Prefer AES-GCM suites for hardware acceleration
768 suites.push_back(CipherSuite::TLS_AES_256_GCM_SHA384);
769 suites.push_back(CipherSuite::TLS_AES_128_GCM_SHA256);
770 } else {
771 // Use ChaCha20-Poly1305 for better software performance
772 suites.push_back(CipherSuite::TLS_CHACHA20_POLY1305_SHA256);
773 suites.push_back(CipherSuite::TLS_AES_128_GCM_SHA256);
774 }
775
776 return suites;
777}
