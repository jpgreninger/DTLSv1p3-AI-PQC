363 Result<std::shared_ptr<ManagedConnection>> select_weighted_round_robin(
364 const std::vector<std::shared_ptr<ManagedConnection>>& connections) {
365 std::lock_guard<std::mutex> lock(mutex_);
366
367 // Calculate total weight
368 double total_weight = 0.0;
369 for (const auto& conn : connections) {
370 auto it = connection_weights_.find(conn->get_connection_id());
371 if (it != connection_weights_.end()) {
372 total_weight += it->second;
373 } else {
374 total_weight += 1.0; // Default weight
375 }
376 }
377
378 // Generate random value and select connection
379 std::random_device rd;
380 std::mt19937 gen(rd());
381 std::uniform_real_distribution<> dis(0.0, total_weight);
382 double random_value = dis(gen);
383
384 double cumulative_weight = 0.0;
385 for (const auto& conn : connections) {
386 auto it = connection_weights_.find(conn->get_connection_id());
387 double weight = (it != connection_weights_.end()) ? it->second : 1.0;
388 cumulative_weight += weight;
389
390 if (random_value <= cumulative_weight) {
391 return make_result(conn);
392 }
393 }
394
395 // Fallback to first connection
396 return make_result(connections[0]);
397 }

411 Result<std::shared_ptr<ManagedConnection>> select_adaptive(
412 const std::vector<std::shared_ptr<ManagedConnection>>& connections) {
413 // Adaptive strategy combines multiple factors
414 double best_score = std::numeric_limits<double>::max();
415 std::shared_ptr<ManagedConnection> best_connection;
416
417 for (const auto& conn : connections) {
418 auto health = conn->get_health();
419 double load = conn->get_load();
420
421 // Calculate composite score (lower is better)
422 double response_time_score = static_cast<double>(health.last_response_time.count()) / 1000.0;
423 double load_score = load * 100.0;
424 double error_score = health.error_rate * 1000.0;
425
426 double composite_score = response_time_score + load_score + error_score;
427
428 if (composite_score < best_score) {
429 best_score = composite_score;
430 best_connection = conn;
431 }
432 }
433
434 if (best_connection) {
435 return make_result(best_connection);
436 }
437
438 return make_result(connections[0]); // Fallback
