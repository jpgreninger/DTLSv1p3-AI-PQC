23struct MessageFragment {
24 uint16_t message_seq;
25 uint32_t fragment_offset;
26 uint32_t fragment_length;
27 uint32_t total_length;
28 memory::Buffer fragment_data;
29
30 MessageFragment() = default;
31 MessageFragment(uint16_t seq, uint32_t offset, uint32_t frag_len,
32 uint32_t total_len, memory::Buffer data)
33 : message_seq(seq), fragment_offset(offset), fragment_length(frag_len)
34 , total_length(total_len), fragment_data(std::move(data)) {}
35
36 bool is_complete_message() const {
37 return fragment_offset == 0 && fragment_length == total_length;
38 }
39
40 bool is_valid() const {
41 return fragment_offset + fragment_length <= total_length &&
42 fragment_data.size() == fragment_length;
43 }
44};
