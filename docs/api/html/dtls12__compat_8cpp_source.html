64 Result<protocol::PlaintextRecord> process_dtls12_record(
65 const std::vector<uint8_t>& data,
66 const crypto::KeySchedule& keys) override {
67
68 if (data.size() < protocol::RecordHeader::SERIALIZED_SIZE) {
69 return make_error<protocol::PlaintextRecord>(DTLSError::INVALID_RECORD_HEADER);
70 }
71
72 // Parse DTLS 1.2 record header
73 protocol::RecordHeader header;
74 auto header_result = header.deserialize(data);
75 if (!header_result) {
76 return make_error<protocol::PlaintextRecord>(DTLSError::INVALID_RECORD_HEADER);
77 }
78
79 // Validate DTLS 1.2 version
80 if (header.version != protocol::ProtocolVersion::DTLS_1_2) {
81 return make_error<protocol::PlaintextRecord>(DTLSError::UNSUPPORTED_RECORD_VERSION);
82 }
83
84 // Extract payload
85 size_t header_size = protocol::RecordHeader::SERIALIZED_SIZE;
86 if (data.size() < header_size + header.length) {
87 return make_error<protocol::PlaintextRecord>(DTLSError::RECORD_LENGTH_MISMATCH);
88 }
89
90 std::vector<uint8_t> payload(data.begin() + header_size,
91 data.begin() + header_size + header.length);
92
93 // For application data, decrypt using DTLS 1.2 method
94 if (header.content_type == ContentType::APPLICATION_DATA) {
95 auto decrypt_result = decrypt_dtls12_payload(payload, keys, header);
96 if (!decrypt_result) {
97 return make_error<protocol::PlaintextRecord>(decrypt_result.error());
98 }
99 payload = std::move(decrypt_result.value());
100 }
101
102 // Create plaintext record
103 protocol::PlaintextRecord record;
104 record.header() = header;
105 record.payload() = std::move(payload);
106
107 return make_result(std::move(record));
108 }

110 Result<std::vector<uint8_t>> create_dtls12_record(
111 ContentType content_type,
112 const std::vector<uint8_t>& payload,
113 const crypto::KeySchedule& keys) override {
114
115 protocol::RecordHeader header;
116 header.content_type = content_type;
117 header.version = protocol::ProtocolVersion::DTLS_1_2;
118 header.epoch = current_epoch_;
119 header.sequence_number = sequence_number_++;
120 header.length = static_cast<uint16_t>(payload.size());
121
122 // Encrypt application data using DTLS 1.2 method
123 std::vector<uint8_t> final_payload = payload;
124 if (content_type == ContentType::APPLICATION_DATA) {
125 auto encrypt_result = encrypt_dtls12_payload(payload, keys, header);
126 if (!encrypt_result) {
127 return make_error<std::vector<uint8_t>>(encrypt_result.error());
128 }
129 final_payload = std::move(encrypt_result.value());
130 header.length = static_cast<uint16_t>(final_payload.size());
131 }
132
133 // Serialize record
134 std::vector<uint8_t> record_data;
135 record_data.resize(protocol::RecordHeader::SERIALIZED_SIZE + final_payload.size());
136
137 auto serialize_result = header.serialize(record_data);
138 if (!serialize_result) {
139 return make_error<std::vector<uint8_t>>(DTLSError::SERIALIZATION_FAILED);
140 }
141
142 std::copy(final_payload.begin(), final_payload.end(),
143 record_data.begin() + protocol::RecordHeader::SERIALIZED_SIZE);
144
145 return make_result(std::move(record_data));
146 }

155private:
156 Result<std::vector<uint8_t>> encrypt_dtls12_payload(
157 const std::vector<uint8_t>& payload,
158 const crypto::KeySchedule& keys,
159 const protocol::RecordHeader& header) {
160
161 // DTLS 1.2 uses explicit nonce for AEAD
162 std::vector<uint8_t> nonce(8); // 8-byte explicit nonce
163 crypto_provider_->generate_random({nonce.size(), true, {}}, nonce);
164
165 // Construct full nonce (implicit IV + explicit nonce)
166 std::vector<uint8_t> full_nonce(keys.server_write_iv);
167 full_nonce.insert(full_nonce.end(), nonce.begin(), nonce.end());
168
169 // Construct additional data for AEAD
170 std::vector<uint8_t> aad;
171 aad.resize(13);
172 // Sequence number (8 bytes)
173 for (int i = 7; i >= 0; --i) {
174 aad[i] = static_cast<uint8_t>((header.sequence_number >> (8 * (7 - i))) & 0xFF);
175 }
176 aad[8] = static_cast<uint8_t>(header.content_type);
177 aad[9] = static_cast<uint8_t>(header.version >> 8);
178 aad[10] = static_cast<uint8_t>(header.version & 0xFF);
179 aad[11] = static_cast<uint8_t>(payload.size() >> 8);
180 aad[12] = static_cast<uint8_t>(payload.size() & 0xFF);
181
182 // Encrypt using AEAD
183 crypto::AEADParams params;
184 params.key = keys.server_write_key;
185 params.nonce = full_nonce;
186 params.additional_data = aad;
187 params.cipher = AEADCipher::AES_128_GCM; // Default, should be based on cipher suite
188
189 auto encrypt_result = crypto_provider_->aead_encrypt(params, payload);
190 if (!encrypt_result) {
191 return make_error<std::vector<uint8_t>>(encrypt_result.error());
192 }
193
194 // Prepend explicit nonce to ciphertext
195 std::vector<uint8_t> final_payload(nonce);
196 final_payload.insert(final_payload.end(),
197 encrypt_result.value().begin(),
198 encrypt_result.value().end());
199
200 return make_result(std::move(final_payload));
201 }

203 Result<std::vector<uint8_t>> decrypt_dtls12_payload(
204 const std::vector<uint8_t>& encrypted_payload,
205 const crypto::KeySchedule& keys,
206 const protocol::RecordHeader& header) {
207
208 if (encrypted_payload.size() < 8) {
209 return make_error<std::vector<uint8_t>>(DTLSError::DECRYPT_ERROR);
210 }
211
212 // Extract explicit nonce (first 8 bytes)
213 std::vector<uint8_t> explicit_nonce(encrypted_payload.begin(),
214 encrypted_payload.begin() + 8);
215
216 // Construct full nonce
217 std::vector<uint8_t> full_nonce(keys.client_write_iv);
218 full_nonce.insert(full_nonce.end(), explicit_nonce.begin(), explicit_nonce.end());
219
220 // Extract ciphertext (remaining bytes)
221 std::vector<uint8_t> ciphertext(encrypted_payload.begin() + 8,
222 encrypted_payload.end());
223
224 // Construct additional data
225 size_t plaintext_length = ciphertext.size() - 16; // Subtract tag length
226 std::vector<uint8_t> aad;
227 aad.resize(13);
228 for (int i = 7; i >= 0; --i) {
229 aad[i] = static_cast<uint8_t>((header.sequence_number >> (8 * (7 - i))) & 0xFF);
230 }
231 aad[8] = static_cast<uint8_t>(header.content_type);
232 aad[9] = static_cast<uint8_t>(header.version >> 8);
233 aad[10] = static_cast<uint8_t>(header.version & 0xFF);
234 aad[11] = static_cast<uint8_t>(plaintext_length >> 8);
235 aad[12] = static_cast<uint8_t>(plaintext_length & 0xFF);
236
237 // Decrypt using AEAD
238 crypto::AEADParams params;
239 params.key = keys.client_write_key;
240 params.nonce = full_nonce;
241 params.additional_data = aad;
242 params.cipher = AEADCipher::AES_128_GCM;
243
244 return crypto_provider_->aead_decrypt(params, ciphertext);
245 }

255Result<protocol::ClientHello> DTLS12HandshakeConverter::convert_client_hello_from_dtls12(
256 const std::vector<uint8_t>& dtls12_client_hello) {
257
258 protocol::ClientHello client_hello;
259
260 // Parse DTLS 1.2 ClientHello
261 auto deserialize_result = client_hello.deserialize(dtls12_client_hello);
262 if (!deserialize_result) {
263 return make_error<protocol::ClientHello>(DTLSError::DECODE_ERROR);
264 }
265
266 // Convert cipher suites to DTLS 1.3 equivalents
267 std::vector<CipherSuite> dtls13_cipher_suites;
268 for (auto dtls12_suite_value : client_hello.cipher_suites()) {
269 auto dtls12_suite = static_cast<DTLS12CipherSuite>(dtls12_suite_value);
270 auto it = g_dtls12_to_dtls13_cipher_map.find(dtls12_suite);
271 if (it != g_dtls12_to_dtls13_cipher_map.end()) {
272 dtls13_cipher_suites.push_back(it->second);
273 }
274 }
275
276 // Update cipher suites in ClientHello
277 std::vector<uint16_t> cipher_suite_values;
278 for (auto suite : dtls13_cipher_suites) {
279 cipher_suite_values.push_back(static_cast<uint16_t>(suite));
280 }
281 client_hello.set_cipher_suites(cipher_suite_values);
282
283 // Add supported_versions extension for DTLS 1.3
284 protocol::Extension supported_versions_ext;
285 supported_versions_ext.type = static_cast<uint16_t>(ExtensionType::SUPPORTED_VERSIONS);
286
287 // Add DTLS 1.3 and 1.2 as supported versions
288 std::vector<uint8_t> versions_data;
289 versions_data.push_back(4); // Length of version list
290 versions_data.push_back(0xFE); versions_data.push_back(0xFC); // DTLS 1.3
291 versions_data.push_back(0xFE); versions_data.push_back(0xFD); // DTLS 1.2
292
293 supported_versions_ext.data = std::move(versions_data);
294 client_hello.add_extension(supported_versions_ext);
295
296 return make_result(std::move(client_hello));
297}

363// VersionNegotiator implementation
364Result<ProtocolVersion> VersionNegotiator::negotiate_version(
365 const std::vector<uint8_t>& client_hello_data,
366 const DTLS12CompatibilityContext& compat_context) {
367
368 // Parse client hello to check supported versions
369 protocol::ClientHello client_hello;
370 auto parse_result = client_hello.deserialize(client_hello_data);
371 if (!parse_result) {
372 return make_error<ProtocolVersion>(DTLSError::DECODE_ERROR);
373 }
374
375 // Check for supported_versions extension
376 if (client_hello.has_extension(static_cast<uint16_t>(ExtensionType::SUPPORTED_VERSIONS))) {
377 auto extension = client_hello.get_extension(static_cast<uint16_t>(ExtensionType::SUPPORTED_VERSIONS));
378 if (extension) {
379 auto& data = extension->data;
380 if (!data.empty()) {
381 // Check if DTLS 1.3 is supported
382 for (size_t i = 1; i < data.size(); i += 2) {
383 if (i + 1 < data.size()) {
384 uint16_t version = (data[i] << 8) | data[i + 1];
385 if (version == static_cast<uint16_t>(protocol::ProtocolVersion::DTLS_1_3)) {
386 return make_result(protocol::ProtocolVersion::DTLS_1_3);
387 }
388 }
389 }
390 }
391 }
392 }
393
394 // Check legacy version field
395 if (client_hello.legacy_version() == protocol::ProtocolVersion::DTLS_1_3) {
396 return make_result(protocol::ProtocolVersion::DTLS_1_3);
397 } else if (client_hello.legacy_version() == protocol::ProtocolVersion::DTLS_1_2 &&
398 compat_context.enable_dtls12_fallback) {
399 return make_result(protocol::ProtocolVersion::DTLS_1_2);
400 }
401
402 return make_error<ProtocolVersion>(DTLSError::PROTOCOL_VERSION_NOT_SUPPORTED);
403}
