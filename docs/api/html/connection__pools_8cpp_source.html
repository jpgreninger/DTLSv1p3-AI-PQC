41std::unique_ptr<ZeroCopyBuffer> ConnectionPool::acquire_buffer(size_t size) {
42 auto start_time = std::chrono::high_resolution_clock::now();
43
44 std::unique_ptr<ZeroCopyBuffer> buffer;
45
46 // Try to get from size-specific pool first
47 auto pool_it = size_specific_pools_.find(size);
48 if (pool_it != size_specific_pools_.end()) {
49 buffer = pool_it->second->acquire();
50 if (buffer) {
51 std::lock_guard<std::mutex> lock(usage_mutex_);
52 current_stats_.pool_hits++;
53 }
54 }
55
56 // If no buffer from pool, create new one
57 if (!buffer) {
58 // Check if we should create a dedicated pool for this size
59 if (should_create_dedicated_pool(size)) {
60 size_t pool_size = calculate_optimal_pool_size(size);
61 auto new_pool = std::make_unique<BufferPool>(size, pool_size);
62 buffer = new_pool->acquire();
63 size_specific_pools_[size] = std::move(new_pool);
64 } else {
65 buffer = std::make_unique<ZeroCopyBuffer>(size);
66 }
67
68 std::lock_guard<std::mutex> lock(usage_mutex_);
69 current_stats_.pool_misses++;
70 }
71
72 auto end_time = std::chrono::high_resolution_clock::now();
73 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(end_time - start_time);
74
75 // Update statistics
76 {
77 std::lock_guard<std::mutex> lock(usage_mutex_);
78 current_stats_.total_buffers_allocated++;
79 current_stats_.buffers_in_use++;
80 current_memory_usage_ += size;
81
82 allocation_times_.push_back(std::chrono::steady_clock::now());
83 allocation_durations_.push_back(duration);
84
85 // Calculate running average
86 auto total_time = current_stats_.average_acquire_time.count() * (current_stats_.total_buffers_allocated - 1);
87 current_stats_.average_acquire_time = std::chrono::nanoseconds(
88 (total_time + duration.count()) / current_stats_.total_buffers_allocated);
89
90 // Update hit rate
91 size_t total_requests = current_stats_.pool_hits + current_stats_.pool_misses;
92 current_stats_.hit_rate = total_requests > 0 ?
93 static_cast<double>(current_stats_.pool_hits) / total_requests : 0.0;
94
95 // Cleanup old timing data
96 if (allocation_times_.size() > 1000) {
97 allocation_times_.erase(allocation_times_.begin(), allocation_times_.begin() + 500);
98 allocation_durations_.erase(allocation_durations_.begin(), allocation_durations_.begin() + 500);
99 }
100 }
101
102 return buffer;
103}

196void ConnectionPool::optimize_for_characteristics() {
197 // Determine optimal pool configurations based on connection characteristics
198 switch (characteristics_.type) {
199 case ConnectionCharacteristics::ConnectionType::LOW_LATENCY:
200 // Pre-allocate more buffers for low latency
201 for (auto& [size, pool] : size_specific_pools_) {
202 pool->expand_pool(pool->total_buffers() / 2);
203 }
204 break;
205
206 case ConnectionCharacteristics::ConnectionType::HIGH_THROUGHPUT:
207 // Optimize for bulk operations
208 if (!preferred_buffer_sizes().empty()) {
209 for (size_t size : characteristics_.preferred_buffer_sizes) {
210 if (size_specific_pools_.find(size) == size_specific_pools_.end()) {
211 size_specific_pools_[size] = std::make_unique<BufferPool>(size, 16);
212 }
213 }
214 }
215 break;
216
217 case ConnectionCharacteristics::ConnectionType::INTERACTIVE:
218 // Balanced approach with moderate pre-allocation
219 break;
220
221 case ConnectionCharacteristics::ConnectionType::IOT_SENSOR:
222 // Memory-constrained optimization
223 memory_limit_ = 64 * 1024; // 64KB limit for IoT
224 optimization_level_ = 1; // Minimal optimization
225 break;
226
227 default:
228 break;
229 }
230
231 // Update statistics
232 {
233 std::lock_guard<std::mutex> lock(usage_mutex_);
234 current_stats_.adaptations_performed++;
235 }

238void ConnectionPool::adapt_to_usage_pattern() {
239 update_usage_statistics();
240
241 // Analyze buffer size usage patterns
242 if (!characteristics_.buffer_size_usage.empty()) {
243 // Find most frequently used buffer sizes
244 std::vector<std::pair<size_t, size_t>> usage_pairs;
245 for (const auto& [size, count] : characteristics_.buffer_size_usage) {
246 usage_pairs.emplace_back(count, size);
247 }
248
249 std::sort(usage_pairs.rbegin(), usage_pairs.rend()); // Sort by usage count descending
250
251 // Create dedicated pools for top 5 most used sizes
252 for (size_t i = 0; i < std::min(usage_pairs.size(), static_cast<size_t>(5)); ++i) {
253 size_t buffer_size = usage_pairs[i].second;
254 size_t usage_count = usage_pairs[i].first;
255
256 if (size_specific_pools_.find(buffer_size) == size_specific_pools_.end()) {
257 size_t pool_size = std::min(usage_count / 4, static_cast<size_t>(32)); // Quarter of usage, max 32
258 pool_size = std::max(pool_size, static_cast<size_t>(2)); // Minimum 2
259
260 size_specific_pools_[buffer_size] = std::make_unique<BufferPool>(buffer_size, pool_size);
261 }
262 }
263 }
264
265 // Cleanup unused pools
266 cleanup_unused_pools();
267
268 // Predictive pre-allocation
269 if (predictive_enabled_.load()) {
270 predict_and_preallocate();
271 }

354void ConnectionPool::create_specialized_pools() {
355 // Create message pool based on average message size
356 if (characteristics_.average_message_size > 0) {
357 size_t pool_size = std::max(static_cast<size_t>(8), optimization_level_.load() * 2);
358 message_pool_ = std::make_unique<BufferPool>(characteristics_.average_message_size, pool_size);

396size_t ConnectionPool::calculate_optimal_pool_size(size_t buffer_size) const {
397 // Base pool size on optimization level and usage patterns
398 size_t base_size = 2 + optimization_level_.load();
399
400 // Check if we have usage data for this buffer size
401 auto usage_it = characteristics_.buffer_size_usage.find(buffer_size);
402 if (usage_it != characteristics_.buffer_size_usage.end()) {
403 size_t usage_count = usage_it->second;
404 // Size pool to handle quarter of peak usage
405 size_t usage_based_size = std::max(usage_count / 4, static_cast<size_t>(1));
406 base_size = std::max(base_size, usage_based_size);
407 }
408
409 // Consider connection type
410 switch (characteristics_.type) {
411 case ConnectionCharacteristics::ConnectionType::LOW_LATENCY:
412 base_size *= 2; // More pre-allocation for low latency
413 break;
414 case ConnectionCharacteristics::ConnectionType::HIGH_THROUGHPUT:
415 base_size *= 3; // Even more for high throughput
416 break;
417 case ConnectionCharacteristics::ConnectionType::IOT_SENSOR:
418 base_size = std::min(base_size, static_cast<size_t>(2)); // Constrain for IoT
419 break;
420 default:
421 break;
422 }
423
424 // Cap based on memory limits
425 size_t max_buffers = memory_limit_.load() / buffer_size;
426 return std::min(base_size, std::max(max_buffers, static_cast<size_t>(1)));
427}

429bool ConnectionPool::should_create_dedicated_pool(size_t buffer_size) const {
430 // Don't create pools for very large buffers
431 if (buffer_size > 64 * 1024) {
432 return false;
433 }
434
435 // Don't create pools if we're at memory limit
436 if (current_memory_usage_.load() > memory_limit_.load() * 0.8) {
437 return false;
438 }
439
440 // Create pool if we've seen this size enough times
441 auto usage_it = characteristics_.buffer_size_usage.find(buffer_size);
442 if (usage_it != characteristics_.buffer_size_usage.end()) {
443 return usage_it->second >= 5; // At least 5 allocations of this size
444 }
445
446 return false;
447}

496Result<void> ConnectionMemoryManager::create_connection_pool(
497 void* connection_id, const ConnectionCharacteristics& characteristics) {
498
499 if (!connection_id) {
500 return Result<void>(DTLSError::INVALID_PARAMETER);
501 }
502
503 std::lock_guard<std::mutex> lock(connections_mutex_);
504
505 if (connection_pools_.find(connection_id) != connection_pools_.end()) {
506 return Result<void>(DTLSError::ALREADY_INITIALIZED);
507 }
508
509 // Check global limits
510 if (connection_pools_.size() >= 10000) { // Limit to 10k connections
511 return Result<void>(DTLSError::RESOURCE_EXHAUSTED);
512 }
513
514 try {
515 auto pool = std::make_unique<ConnectionPool>(connection_id, characteristics);
516 pool->optimize_for_characteristics();
517
518 connection_pools_[connection_id] = std::move(pool);
519 connection_characteristics_[connection_id] = characteristics;
520
521 // Update system statistics
522 {
523 std::lock_guard<std::mutex> stats_lock(stats_mutex_);
524 system_stats_.total_connections++;
525 system_stats_.active_connections++;
526 }
527
528 return Result<void>();
529
530 } catch (const std::exception&) {
531 return Result<void>(DTLSError::OUT_OF_MEMORY);
532 }
