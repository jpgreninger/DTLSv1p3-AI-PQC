72Result<memory::Buffer> CookieManager::generate_cookie(const ClientInfo& client_info) {
73 std::lock_guard<std::mutex> lock(mutex_);
74
75 if (!initialized_) {
76 return Result<memory::Buffer>(DTLSError::NOT_INITIALIZED);
77 }
78
79 // Check if client has too many active cookies
80 if (client_has_too_many_cookies(client_info)) {
81 // Remove oldest cookies for this client
82 remove_client_cookies(client_info);
83 }
84
85 // Generate timestamp (using microseconds for uniqueness)
86 auto now = std::chrono::steady_clock::now();
87 auto timestamp = std::chrono::duration_cast<std::chrono::microseconds>(
88 now.time_since_epoch()).count();
89
90 // Generate HMAC-based cookie
91 auto cookie_result = generate_hmac_cookie(client_info, timestamp);
92 if (!cookie_result.is_success()) {
93 return cookie_result;
94 }
95
96 memory::Buffer cookie = std::move(cookie_result.value());
97
98 // Track the cookie
99 std::string cookie_key = generate_cookie_key(cookie);
100 CookieEntry entry(client_info);
101 active_cookies_[cookie_key] = entry;
102
103 // Add to client mapping
104 std::string client_id = client_info.get_client_id();
105 client_cookie_mapping_[client_id].push_back(cookie_key);
106
107 ++stats_.cookies_generated;
108 stats_.active_cookies = active_cookies_.size();
109
110 return Result<memory::Buffer>(std::move(cookie));
111}

113CookieManager::CookieValidationResult
114CookieManager::validate_cookie(const memory::Buffer& cookie,
115 const ClientInfo& client_info) {
116 std::lock_guard<std::mutex> lock(mutex_);
117
118 // Always perform all validation steps for constant timing
119 // Use bitmasks to track validation results without early returns
120
121 ++stats_.cookies_validated;
122
123 // Initialize all variables we'll need (constant-time)
124 auto now = std::chrono::steady_clock::now();
125 auto now_microseconds = std::chrono::duration_cast<std::chrono::microseconds>(
126 now.time_since_epoch()).count();
127 uint64_t timestamp = 0;
128 std::string cookie_key;
129 auto it = active_cookies_.end();
130
131 // Validation flags (all checks performed regardless of early failures)
132 volatile bool initialized_ok = initialized_;
133 volatile bool format_valid = true;
134 volatile bool hmac_valid = true;
135 volatile bool not_expired = true;
136 volatile bool cookie_found = false;
137 volatile bool client_matches = false;
138 volatile bool not_consumed = true;
139
140 // Always check format (constant-time)
141 format_valid = is_valid_cookie_format(cookie);
142
143 // Always try HMAC verification (constant-time)
144 hmac_valid = verify_hmac_cookie(cookie, client_info, timestamp);
145
146 // Always calculate age (constant-time)
147 auto cookie_age_microseconds = now_microseconds - timestamp;
148 auto cookie_age = cookie_age_microseconds / 1000000;
149 not_expired = (cookie_age <= config_.cookie_lifetime.count());
150
151 // Always generate cookie key and lookup (constant-time)
152 cookie_key = generate_cookie_key(cookie);
153 it = active_cookies_.find(cookie_key);
154 cookie_found = (it != active_cookies_.end());
155
156 // Always check client info if cookie found (constant-time comparison)
157 if (cookie_found) {
158 client_matches = (it->second.client_info == client_info);
159 not_consumed = !it->second.consumed;
160 }
161
162 // Determine result based on all validation flags (constant-time logic)
163 CookieValidationResult result;
164
165 if (!initialized_ok) {
166 result = CookieValidationResult::INVALID;
167 } else if (!format_valid) {
168 ++stats_.validation_failures;
169 result = CookieValidationResult::INVALID;
170 } else if (!hmac_valid) {
171 ++stats_.validation_failures;
172 result = CookieValidationResult::CLIENT_MISMATCH;
173 } else if (!not_expired) {
174 ++stats_.cookies_expired;
175 result = CookieValidationResult::EXPIRED;
176 } else if (!cookie_found) {
177 ++stats_.validation_failures;
178 result = CookieValidationResult::NOT_FOUND;
179 } else if (!client_matches) {
180 ++stats_.validation_failures;
181 result = CookieValidationResult::CLIENT_MISMATCH;
182 } else if (!not_consumed) {
183 ++stats_.replay_attempts;
184 result = CookieValidationResult::REPLAY_DETECTED;
185 } else {
186 // Success case - always perform these operations for constant timing
187 result = CookieValidationResult::VALID;
188 }
189
190 // Always perform update operations if we have a valid iterator (constant-time)
191 if (cookie_found && result == CookieValidationResult::VALID) {
192 it->second.last_access_time = now;
193 ++it->second.usage_count;
194 it->second.consumed = true;
195
196 std::string client_id = client_info.get_client_id();
197 authenticated_clients_[client_id] = now;
198 }
199
200 return result;
201}

240void CookieManager::cleanup_expired_cookies() {
241 std::lock_guard<std::mutex> lock(mutex_);
242
243 auto now = std::chrono::steady_clock::now();
244
245 // Only run cleanup if enough time has passed
246 if ((now - last_cleanup_) < config_.cleanup_interval) {
247 return;
248 }
249
250 // Remove expired cookies
251 auto cookie_it = active_cookies_.begin();
252 while (cookie_it != active_cookies_.end()) {
253 if (cookie_it->second.is_expired(config_.cookie_lifetime)) {
254 std::string client_id = cookie_it->second.client_info.get_client_id();
255
256 // Remove from client mapping
257 auto client_it = client_cookie_mapping_.find(client_id);
258 if (client_it != client_cookie_mapping_.end()) {
259 auto& cookie_list = client_it->second;
260 cookie_list.erase(
261 std::remove(cookie_list.begin(), cookie_list.end(), cookie_it->first),
262 cookie_list.end());
263
264 if (cookie_list.empty()) {
265 client_cookie_mapping_.erase(client_it);
266 }
267 }
268
269 cookie_it = active_cookies_.erase(cookie_it);
270 } else {
271 ++cookie_it;
272 }
273 }
274
275 stats_.active_cookies = active_cookies_.size();
276 last_cleanup_ = now;

310Result<memory::Buffer> CookieManager::generate_hmac_cookie(
311 const ClientInfo& client_info, uint64_t timestamp) const {
312
313 // Calculate required cookie size
314 size_t total_size = 4 + 1 + 8 + HMAC_SIZE; // MAGIC + VERSION + TIMESTAMP + HMAC
315
316 memory::Buffer cookie(total_size);
317 cookie.resize(total_size);
318 uint8_t* data = reinterpret_cast<uint8_t*>(cookie.mutable_data());
319 size_t offset = 0;
320
321 // Write magic number
322 uint32_t magic_be = htonl(COOKIE_MAGIC);
323 std::memcpy(data + offset, &magic_be, 4);
324 offset += 4;
325
326 // Write version
327 data[offset] = COOKIE_VERSION;
328 offset += 1;
329
330 // Write timestamp
331 uint64_t timestamp_be = htobe64(timestamp);
332 std::memcpy(data + offset, &timestamp_be, 8);
333 offset += 8;
334
335 // Prepare HMAC input (everything except HMAC itself)
336 std::vector<uint8_t> hmac_input;
337 hmac_input.reserve(offset + client_info.client_address.size() + 2 + client_info.client_hello_data.size());
338
339 // Add cookie prefix (magic + version + timestamp)
340 hmac_input.insert(hmac_input.end(), data, data + offset);
341
342 // Add client address
343 hmac_input.insert(hmac_input.end(),
344 client_info.client_address.begin(),
345 client_info.client_address.end());
346
347 // Add client port
348 uint16_t port_be = htons(client_info.client_port);
349 hmac_input.insert(hmac_input.end(),
350 reinterpret_cast<const uint8_t*>(&port_be),
351 reinterpret_cast<const uint8_t*>(&port_be) + 2);
352
353 // Add client hello data
354 hmac_input.insert(hmac_input.end(),
355 client_info.client_hello_data.begin(),
356 client_info.client_hello_data.end());
357
358 // Calculate HMAC-SHA256
359 unsigned int hmac_len = 0;
360 unsigned char hmac_result[EVP_MAX_MD_SIZE];
361
362 if (!HMAC(EVP_sha256(),
363 secret_key_.data(), static_cast<int>(secret_key_.size()),
364 hmac_input.data(), hmac_input.size(),
365 hmac_result, &hmac_len)) {
366 return Result<memory::Buffer>(DTLSError::CRYPTO_PROVIDER_ERROR);
367 }
368
369 if (hmac_len != HMAC_SIZE) {
370 return Result<memory::Buffer>(DTLSError::CRYPTO_PROVIDER_ERROR);
371 }
372
373 // Write HMAC
374 std::memcpy(data + offset, hmac_result, HMAC_SIZE);
375
376 return Result<memory::Buffer>(std::move(cookie));
377}

379bool CookieManager::verify_hmac_cookie(const memory::Buffer& cookie,
380 const ClientInfo& client_info,
381 uint64_t& timestamp) const {
382 if (cookie.size() < 4 + 1 + 8 + HMAC_SIZE) {
383 return false;
384 }
385
386 const uint8_t* data = reinterpret_cast<const uint8_t*>(cookie.data());
387 size_t offset = 0;
388
389 // Verify magic number
390 uint32_t magic;
391 std::memcpy(&magic, data + offset, 4);
392 if (ntohl(magic) != COOKIE_MAGIC) {
393 return false;
394 }
395 offset += 4;
396
397 // Verify version
398 if (data[offset] != COOKIE_VERSION) {
399 return false;
400 }
401 offset += 1;
402
403 // Extract timestamp
404 uint64_t timestamp_be;
405 std::memcpy(&timestamp_be, data + offset, 8);
406 timestamp = be64toh(timestamp_be);
407 offset += 8;
408
409 // Prepare HMAC input (everything except HMAC)
410 size_t hmac_input_size = offset + client_info.client_address.size() + 2 + client_info.client_hello_data.size();
411 std::vector<uint8_t> hmac_input;
412 hmac_input.reserve(hmac_input_size);
413
414 // Add cookie prefix
415 hmac_input.insert(hmac_input.end(), data, data + offset);
416
417 // Add client info
418 hmac_input.insert(hmac_input.end(),
419 client_info.client_address.begin(),
420 client_info.client_address.end());
421
422 uint16_t port_be = htons(client_info.client_port);
423 hmac_input.insert(hmac_input.end(),
424 reinterpret_cast<const uint8_t*>(&port_be),
425 reinterpret_cast<const uint8_t*>(&port_be) + 2);
426
427 hmac_input.insert(hmac_input.end(),
428 client_info.client_hello_data.begin(),
429 client_info.client_hello_data.end());
430
431 // Calculate expected HMAC
432 unsigned int hmac_len = 0;
433 unsigned char expected_hmac[EVP_MAX_MD_SIZE];
434
435 if (!HMAC(EVP_sha256(),
436 secret_key_.data(), static_cast<int>(secret_key_.size()),
437 hmac_input.data(), hmac_input.size(),
438 expected_hmac, &hmac_len)) {
439 return false;
440 }
441
442 if (hmac_len != HMAC_SIZE) {
443 return false;
444 }
445
446 // Compare HMACs using constant-time comparison
447 return CRYPTO_memcmp(data + offset, expected_hmac, HMAC_SIZE) == 0;
448}

453 SHA256(reinterpret_cast<const unsigned char*>(cookie.data()),

464bool CookieManager::client_has_too_many_cookies(const ClientInfo& client_info) const {
465 std::string client_id = client_info.get_client_id();
466 auto it = client_cookie_mapping_.find(client_id);
467
468 if (it == client_cookie_mapping_.end()) {
469 return false;
470 }
471
472 // Count valid (non-expired, non-consumed) cookies
473 uint32_t valid_cookies = 0;
474 for (const auto& cookie_key : it->second) {
475 auto cookie_it = active_cookies_.find(cookie_key);
476 if (cookie_it != active_cookies_.end() &&
477 !cookie_it->second.consumed &&
478 !cookie_it->second.is_expired(config_.cookie_lifetime)) {
479 ++valid_cookies;
480 }
481 }
482
483 return valid_cookies >= config_.max_cookies_per_client;
484}

486void CookieManager::remove_client_cookies(const ClientInfo& client_info) {
487 std::string client_id = client_info.get_client_id();
488 auto client_it = client_cookie_mapping_.find(client_id);
489
490 if (client_it == client_cookie_mapping_.end()) {
491 return;
492 }
493
494 // Remove oldest cookies first
495 auto& cookie_list = client_it->second;
496 std::vector<std::pair<std::string, std::chrono::steady_clock::time_point>> cookie_times;
497
498 for (const auto& cookie_key : cookie_list) {
499 auto cookie_it = active_cookies_.find(cookie_key);
500 if (cookie_it != active_cookies_.end()) {
501 cookie_times.emplace_back(cookie_key, cookie_it->second.creation_time);
502 }
503 }
504
505 // Sort by creation time (oldest first)
506 std::sort(cookie_times.begin(), cookie_times.end(),
507 [](const auto& a, const auto& b) {
508 return a.second < b.second;
509 });
510
511 // Remove oldest cookies until we're under the limit
512 size_t cookies_to_remove = cookie_times.size() - (config_.max_cookies_per_client - 1);
513 for (size_t i = 0; i < cookies_to_remove && i < cookie_times.size(); ++i) {
514 const auto& cookie_key = cookie_times[i].first;
515 active_cookies_.erase(cookie_key);
516 cookie_list.erase(
517 std::remove(cookie_list.begin(), cookie_list.end(), cookie_key),
518 cookie_list.end());
519 }
520
521 if (cookie_list.empty()) {
522 client_cookie_mapping_.erase(client_it);
523 }
524}

554 if (cookie.size() < MIN_COOKIE_SIZE || cookie.size() > MAX_COOKIE_SIZE) {
555 return false;
556 }

559 return false; // Too small for our cookie format

570

573 return false;
574 }
