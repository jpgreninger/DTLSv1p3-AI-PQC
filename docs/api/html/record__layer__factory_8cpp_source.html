301DTLSCiphertext MockRecordLayer::create_mock_ciphertext(const DTLSPlaintext& plaintext) {
302 DTLSCiphertext ciphertext;

304 ciphertext.version = ProtocolVersion::DTLS_1_2;
305 ciphertext.epoch = current_epoch_;
306 ciphertext.encrypted_sequence_number = SequenceNumber48(current_sequence_number_);
307 ciphertext.length = plaintext.fragment.size() + 16; // Add mock auth tag
308
309 // Mock encrypted content - create a new buffer with plaintext + mock tag
310 memory::ZeroCopyBuffer encrypted_buffer(plaintext.fragment.size() + 16);
311 if (plaintext.fragment.size() > 0) {

313 }
314 // Add mock auth tag (16 bytes of 0xAA)

318 ciphertext.encrypted_record = memory::Buffer(std::move(encrypted_buffer));
