183 status.total_memory_usage = memory_stats.current_bytes_allocated;
184 status.direct_memory_usage = status.total_memory_usage - status.pool_memory_usage;
185 status.active_allocations = memory_stats.current_allocations;
186 status.peak_allocations = memory_stats.peak_allocations;

406size_t optimize_memory_pools() {
407 auto& pool_manager = GlobalPoolManager::instance();
408 auto pool_stats = pool_manager.get_all_statistics();
409
410 size_t optimized_count = 0;
411
412 for (const auto& stats : pool_stats) {
413 // If utilization is very low, shrink the pool
414 if (stats.utilization_ratio < 0.1 && stats.total_buffers > 4) {
415 auto& pool = pool_manager.get_pool(stats.buffer_size);
416 size_t target_size = std::max(static_cast<size_t>(4), stats.total_buffers / 2);
417
418 auto shrink_result = pool.shrink_pool(target_size);
419 if (shrink_result) {
420 optimized_count++;
421 }
422 }
423 // If utilization is very high, expand the pool
424 else if (stats.utilization_ratio > 0.9 && stats.allocation_failures > 0) {
425 auto& pool = pool_manager.get_pool(stats.buffer_size);
426 size_t additional = std::max(static_cast<size_t>(4), stats.total_buffers / 4);
427
428 auto expand_result = pool.expand_pool(additional);
429 if (expand_result) {
430 optimized_count++;
431 }
432 }
433 }
434
435 return optimized_count;
436}

442Result<void> preload_pools(const std::vector<std::pair<size_t, size_t>>& usage_hints) {
443 auto& pool_manager = GlobalPoolManager::instance();
444
445 for (const auto& [buffer_size, expected_count] : usage_hints) {
446 if (buffer_size == 0 || expected_count == 0) {
447 continue;
448 }
449
450 // Create or expand pool to expected size
451 auto create_result = pool_manager.create_pool(buffer_size, expected_count);
452 if (!create_result && create_result.error() != DTLSError::ALREADY_INITIALIZED) {
453 return create_result;
454 }
455
456 // If pool already exists, expand it if needed
457 if (create_result.error() == DTLSError::ALREADY_INITIALIZED) {
458 auto& pool = pool_manager.get_pool(buffer_size);
459 if (pool.total_buffers() < expected_count) {
460 size_t additional = expected_count - pool.total_buffers();
461 auto expand_result = pool.expand_pool(additional);
462 if (!expand_result) {
463 return expand_result;
464 }
465 }
466 }
467 }
468
469 return Result<void>();
470}

472size_t predict_memory_usage(std::chrono::minutes time_horizon) {
473 // Simple prediction based on current usage trends
474 auto memory_stats = utils::MemoryStatsCollector::instance().get_statistics();
475 size_t current_usage = memory_stats.current_bytes_allocated;
476
477 // Calculate usage rate (bytes per minute)
478 auto uptime = std::chrono::steady_clock::now() - memory_stats.start_time;
479 auto uptime_minutes = std::chrono::duration_cast<std::chrono::minutes>(uptime);
480
481 if (uptime_minutes.count() == 0) {
482 return current_usage; // No trend data available
483 }
484
485 double usage_rate = static_cast<double>(current_usage) / uptime_minutes.count();
486 size_t predicted_usage = current_usage + static_cast<size_t>(usage_rate * time_horizon.count());
487
488 return predicted_usage;
489}

619Result<std::vector<BufferPtr>> split_buffer(const BufferView& source,
620 const std::vector<size_t>& split_points) {
621 std::vector<BufferPtr> result;
622
623 if (split_points.empty()) {
624 auto clone_result = clone_buffer(source);
625 if (clone_result) {
626 result.push_back(std::move(*clone_result));
627 } else {
628 return Result<std::vector<BufferPtr>>(clone_result.error());
629 }
630 return Result<std::vector<BufferPtr>>(std::move(result));
631 }
632
633 size_t offset = 0;
634 for (size_t split_point : split_points) {
635 if (split_point <= offset || split_point > source.size()) {
636 return Result<std::vector<BufferPtr>>(DTLSError::INVALID_PARAMETER);
637 }
638
639 auto slice = source.slice(offset, split_point - offset);
640 auto clone_result = clone_buffer(slice);
641 if (!clone_result) {
642 return Result<std::vector<BufferPtr>>(clone_result.error());
643 }
644
645 result.push_back(std::move(*clone_result));
646 offset = split_point;
647 }
648
649 // Add remaining data
650 if (offset < source.size()) {
651 auto slice = source.slice(offset, source.size() - offset);
652 auto clone_result = clone_buffer(slice);
653 if (!clone_result) {
654 return Result<std::vector<BufferPtr>>(clone_result.error());
655 }
656 result.push_back(std::move(*clone_result));
657 }
658
659 return Result<std::vector<BufferPtr>>(std::move(result));
660}
