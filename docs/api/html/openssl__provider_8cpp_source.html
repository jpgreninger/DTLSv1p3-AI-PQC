159Result<std::vector<uint8_t>> OpenSSLProvider::generate_random(const RandomParams& params) {
160 DTLS_CRYPTO_TIMER("openssl_generate_random");
161
162 if (!pimpl_->initialized_) {
163 return Result<std::vector<uint8_t>>(DTLSError::NOT_INITIALIZED);
164 }
165
166 // Validate parameters according to RFC 9147 requirements
167 if (params.length == 0) {
168 return Result<std::vector<uint8_t>>(DTLSError::INVALID_PARAMETER);
169 }
170
171 // For DTLS v1.3, enforce 32-byte random for ClientHello/ServerHello compliance
172 if (params.length > static_cast<size_t>(std::numeric_limits<int>::max())) {
173 return Result<std::vector<uint8_t>>(DTLSError::INVALID_PARAMETER);
174 }
175
176 // Check if cryptographically secure random is requested (required for DTLS v1.3)
177 if (params.cryptographically_secure) {
178 // Verify OpenSSL PRNG is properly seeded for FIPS compliance
179 if (RAND_status() != 1) {
180 // Attempt to seed the PRNG if not already seeded
181 if (RAND_poll() != 1) {
182 return Result<std::vector<uint8_t>>(DTLSError::RANDOM_GENERATION_FAILED);
183 }
184 }
185 }
186
187 std::vector<uint8_t> random_bytes(params.length);
188
189 // Use RAND_bytes for cryptographically secure generation (RFC 9147 requirement)
190 int result;
191 if (params.cryptographically_secure) {
192 result = RAND_bytes(random_bytes.data(), static_cast<int>(params.length));
193 } else {
194 // Use RAND_pseudo_bytes for non-cryptographic use (deprecated in OpenSSL 1.1.0+)
195 // Fall back to RAND_bytes for better security
196 result = RAND_bytes(random_bytes.data(), static_cast<int>(params.length));
197 }
198
199 if (result != 1) {
200 // Enhanced error reporting - get specific OpenSSL error
201 unsigned long openssl_error = ERR_get_error();
202
203 // Clear the error and secure zero the buffer
204 secure_cleanup(random_bytes);
205
206 // Map specific OpenSSL random generation errors
207 if (openssl_error != 0) {
208 return Result<std::vector<uint8_t>>(map_openssl_error_detailed());
209 } else {
210 return Result<std::vector<uint8_t>>(DTLSError::RANDOM_GENERATION_FAILED);
211 }
212 }
213
214 // If additional entropy is provided, mix it in using HKDF-Extract pattern
215 if (!params.additional_entropy.empty()) {
216 // Create a simple entropy mixing using XOR (for additional randomness)
217 // Note: This is a lightweight approach - for production, consider HKDF-Extract
218 size_t entropy_pos = 0;
219 for (size_t i = 0; i < random_bytes.size() && entropy_pos < params.additional_entropy.size(); ++i) {
220 random_bytes[i] ^= params.additional_entropy[entropy_pos];
221 entropy_pos = (entropy_pos + 1) % params.additional_entropy.size();
222 }
223 }
224
225 // Validate the generated random for basic entropy (simple statistical check)
226 if (params.cryptographically_secure && params.length >= 16) {
227 if (!validate_random_entropy(random_bytes)) {
228 secure_cleanup(random_bytes);
229 return Result<std::vector<uint8_t>>(DTLSError::RANDOM_GENERATION_FAILED);
230 }
231 }
232
233 return Result<std::vector<uint8_t>>(std::move(random_bytes));
234}

238 if (!pimpl_->initialized_) {
239 return Result<std::vector<uint8_t>>(DTLSError::NOT_INITIALIZED);
240 }
241
242 if (params.secret.empty() || params.output_length == 0) {
243 return Result<std::vector<uint8_t>>(DTLSError::INVALID_PARAMETER);
244 }
245
246 // Get the hash algorithm
247 const EVP_MD* md = nullptr;
248 switch (params.hash_algorithm) {
249 case HashAlgorithm::SHA256:
250 md = EVP_sha256();
251 break;
252 case HashAlgorithm::SHA384:
253 md = EVP_sha384();
254 break;
255 case HashAlgorithm::SHA512:
256 md = EVP_sha512();
257 break;
258 default:
259 return Result<std::vector<uint8_t>>(DTLSError::OPERATION_NOT_SUPPORTED);
260 }
261
262 if (!md) {
263 return Result<std::vector<uint8_t>>(DTLSError::CRYPTO_PROVIDER_ERROR);
264 }
265
266 // Create KDF context
267 EVP_PKEY_CTX* pctx = EVP_PKEY_CTX_new_id(EVP_PKEY_HKDF, nullptr);
268 if (!pctx) {
269 return Result<std::vector<uint8_t>>(DTLSError::OUT_OF_MEMORY);
270 }
271
272 std::vector<uint8_t> output(params.output_length);
273 size_t output_len = params.output_length;
274
275 int result = 1;
276
277 // Initialize HKDF
278 if (result == 1) {
279 result = EVP_PKEY_derive_init(pctx);
280 }
281
282 // Set hash algorithm
283 if (result == 1) {
284 result = EVP_PKEY_CTX_set_hkdf_md(pctx, md);
285 }
286
287 // Set input key material (IKM)
288 if (result == 1) {
289 result = EVP_PKEY_CTX_set1_hkdf_key(pctx, params.secret.data(),
290 static_cast<int>(params.secret.size()));
291 }
292
293 // Set salt if provided
294 if (result == 1 && !params.salt.empty()) {
295 result = EVP_PKEY_CTX_set1_hkdf_salt(pctx, params.salt.data(),
296 static_cast<int>(params.salt.size()));
297 }
298
299 // Set info if provided
300 if (result == 1 && !params.info.empty()) {
301 result = EVP_PKEY_CTX_add1_hkdf_info(pctx, params.info.data(),
302 static_cast<int>(params.info.size()));
303 }
304
305 // Derive the key
306 if (result == 1) {
307 result = EVP_PKEY_derive(pctx, output.data(), &output_len);
308 }
309
310 EVP_PKEY_CTX_free(pctx);
311
312 if (result != 1) {
313 return Result<std::vector<uint8_t>>(DTLSError::KEY_DERIVATION_FAILED);
314 }
315
316 output.resize(output_len);
317 return Result<std::vector<uint8_t>>(std::move(output));

321 if (!pimpl_->initialized_) {
322 return Result<std::vector<uint8_t>>(DTLSError::NOT_INITIALIZED);

324
325 // Validate parameters
326 if (params.secret.empty()) {
327 return Result<std::vector<uint8_t>>(DTLSError::INVALID_PARAMETER);
328 }
329
330 if (params.output_length == 0 || params.output_length > 4096) {
331 return Result<std::vector<uint8_t>>(DTLSError::INVALID_PARAMETER);
332 }
333
334 // Default iteration count (PBKDF2 doesn't use iterations from KeyDerivationParams)
335 uint32_t iterations = 10000;
336
337 // Validate iteration count
338 if (iterations < 1000 || iterations > 1000000) {
339 return Result<std::vector<uint8_t>>(DTLSError::INVALID_PARAMETER);
340 }
341
342 // Determine hash function
343 const EVP_MD* md = nullptr;
344 switch (params.hash_algorithm) {

346 md = EVP_sha1();
347 break;
348 case HashAlgorithm::SHA256:
349 md = EVP_sha256();
350 break;
351 case HashAlgorithm::SHA384:
352 md = EVP_sha384();
353 break;
354 case HashAlgorithm::SHA512:
355 md = EVP_sha512();
356 break;

359 }
360
361 // Prepare output buffer
362 std::vector<uint8_t> output(params.output_length);
363
364 // Use salt if provided, otherwise use empty salt

370 reinterpret_cast<const char*>(params.secret.data()),
371 static_cast<int>(params.secret.size()),
372 salt_ptr,
373 salt_len,
374 static_cast<int>(iterations),
375 md,
376 static_cast<int>(params.output_length),
377 output.data()
378 );
379
380 if (result != 1) {

382 secure_cleanup(output);

393

395 return Result<std::vector<uint8_t>>(DTLSError::NOT_INITIALIZED);

399 return Result<std::vector<uint8_t>>(DTLSError::INVALID_PARAMETER);
400 }
401
402 // Validate AEAD parameters using new helper function
403 auto validation_result = validate_aead_params(params.cipher, params.key, params.nonce);
404 if (!validation_result) {
405 return Result<std::vector<uint8_t>>(validation_result.error());
406 }
407
408 // Get the cipher algorithm and tag length using helper functions
409 const EVP_CIPHER* cipher = nullptr;
410 size_t tag_length = get_aead_tag_length(params.cipher);
411
412 switch (params.cipher) {
413 case AEADCipher::AES_128_GCM:
414 cipher = EVP_aes_128_gcm();
415 break;
416 case AEADCipher::AES_256_GCM:
417 cipher = EVP_aes_256_gcm();
418 break;
419 case AEADCipher::CHACHA20_POLY1305:
420 cipher = EVP_chacha20_poly1305();
421 break;
422 case AEADCipher::AES_128_CCM:
423 case AEADCipher::AES_128_CCM_8:
424 cipher = EVP_aes_128_ccm();
425 break;
426 default:
427 return Result<std::vector<uint8_t>>(DTLSError::OPERATION_NOT_SUPPORTED);
428 }
429
430 if (!cipher) {
431 return Result<std::vector<uint8_t>>(DTLSError::CRYPTO_PROVIDER_ERROR);
432 }
433
434 // Create cipher context
435 EVP_CIPHER_CTX* ctx = EVP_CIPHER_CTX_new();
436 if (!ctx) {
437 return Result<std::vector<uint8_t>>(DTLSError::OUT_OF_MEMORY);
438 }
439
440 // Output buffer: plaintext + tag
441 std::vector<uint8_t> ciphertext(plaintext.size() + tag_length);
442 int outlen = 0;
443 int result = 1;
444
445 // Initialize encryption
446 if (result == 1) {
447 result = EVP_EncryptInit_ex(ctx, cipher, nullptr, nullptr, nullptr);
448 }
449
450 // Set nonce length (for GCM/CCM)
451 if (result == 1 && (params.cipher == AEADCipher::AES_128_GCM ||
452 params.cipher == AEADCipher::AES_256_GCM ||
453 params.cipher == AEADCipher::AES_128_CCM ||
454 params.cipher == AEADCipher::AES_128_CCM_8)) {
455 result = EVP_CIPHER_CTX_ctrl(ctx, EVP_CTRL_AEAD_SET_IVLEN,
456 static_cast<int>(params.nonce.size()), nullptr);
457 }
458
459 // For CCM, set tag length
460 if (result == 1 && (params.cipher == AEADCipher::AES_128_CCM ||
461 params.cipher == AEADCipher::AES_128_CCM_8)) {
462 result = EVP_CIPHER_CTX_ctrl(ctx, EVP_CTRL_AEAD_SET_TAG,
463 static_cast<int>(tag_length), nullptr);
464 }
465
466 // Set key and nonce
467 if (result == 1) {
468 result = EVP_EncryptInit_ex(ctx, nullptr, nullptr, params.key.data(), params.nonce.data());
469 }
470
471 // For CCM, set plaintext length
472 if (result == 1 && (params.cipher == AEADCipher::AES_128_CCM ||
473 params.cipher == AEADCipher::AES_128_CCM_8)) {
474 result = EVP_EncryptUpdate(ctx, nullptr, &outlen, nullptr, static_cast<int>(plaintext.size()));
475 }
476
477 // Set additional authenticated data (AAD)
478 if (result == 1 && !params.additional_data.empty()) {
479 result = EVP_EncryptUpdate(ctx, nullptr, &outlen,
480 params.additional_data.data(),
481 static_cast<int>(params.additional_data.size()));
482 }
483
484 // Encrypt plaintext
485 if (result == 1) {
486 result = EVP_EncryptUpdate(ctx, ciphertext.data(), &outlen,
487 plaintext.data(), static_cast<int>(plaintext.size()));
488 }
489
490 // Finalize encryption
491 int final_len = 0;
492 if (result == 1) {
493 result = EVP_EncryptFinal_ex(ctx, ciphertext.data() + outlen, &final_len);
494 }
495
496 // Get authentication tag
497 if (result == 1) {
498 result = EVP_CIPHER_CTX_ctrl(ctx, EVP_CTRL_AEAD_GET_TAG,
499 static_cast<int>(tag_length),
500 ciphertext.data() + plaintext.size());
501 }
502
503 EVP_CIPHER_CTX_free(ctx);
504
505 if (result != 1) {
506 DTLSError error = map_openssl_error_detailed();
507 return Result<std::vector<uint8_t>>(error);
508 }
509
510 ciphertext.resize(plaintext.size() + tag_length);
511 return Result<std::vector<uint8_t>>(std::move(ciphertext));
512}

679 case AEADCipher::AES_128_CCM_8:
680 cipher = EVP_aes_128_ccm();
681 break;
682 default:
683 return Result<AEADEncryptionOutput>(DTLSError::OPERATION_NOT_SUPPORTED);
684 }
685
686 if (!cipher) {
687 return Result<AEADEncryptionOutput>(DTLSError::CRYPTO_PROVIDER_ERROR);
688 }
689
690 // Create cipher context
691 EVP_CIPHER_CTX* ctx = EVP_CIPHER_CTX_new();
692 if (!ctx) {
693 return Result<AEADEncryptionOutput>(DTLSError::OUT_OF_MEMORY);
694 }
695
696 // Output buffers
697 std::vector<uint8_t> ciphertext(params.plaintext.size());
698 std::vector<uint8_t> tag(tag_length);
699 int outlen = 0;
700 int result = 1;
701
702 // Initialize encryption
703 if (result == 1) {
704 result = EVP_EncryptInit_ex(ctx, cipher, nullptr, nullptr, nullptr);
705 }
706
707 // Set nonce length (for GCM/CCM)
708 if (result == 1 && (params.cipher == AEADCipher::AES_128_GCM ||
709 params.cipher == AEADCipher::AES_256_GCM ||
710 params.cipher == AEADCipher::AES_128_CCM ||
711 params.cipher == AEADCipher::AES_128_CCM_8)) {
712 result = EVP_CIPHER_CTX_ctrl(ctx, EVP_CTRL_AEAD_SET_IVLEN,
713 static_cast<int>(params.nonce.size()), nullptr);
714 }
715
716 // For CCM, set tag length
717 if (result == 1 && (params.cipher == AEADCipher::AES_128_CCM ||
718 params.cipher == AEADCipher::AES_128_CCM_8)) {
719 result = EVP_CIPHER_CTX_ctrl(ctx, EVP_CTRL_AEAD_SET_TAG,
720 static_cast<int>(tag_length), nullptr);
721 }
722
723 // Set key and nonce
724 if (result == 1) {
725 result = EVP_EncryptInit_ex(ctx, nullptr, nullptr, params.key.data(), params.nonce.data());
726 }
727
728 // For CCM, set plaintext length
729 if (result == 1 && (params.cipher == AEADCipher::AES_128_CCM ||
730 params.cipher == AEADCipher::AES_128_CCM_8)) {
731 result = EVP_EncryptUpdate(ctx, nullptr, &outlen, nullptr, static_cast<int>(params.plaintext.size()));
732 }
733
734 // Set additional authenticated data (AAD)
735 if (result == 1 && !params.additional_data.empty()) {
736 result = EVP_EncryptUpdate(ctx, nullptr, &outlen,
737 params.additional_data.data(),
738 static_cast<int>(params.additional_data.size()));
739 }
740
741 // Encrypt plaintext
742 if (result == 1) {
743 result = EVP_EncryptUpdate(ctx, ciphertext.data(), &outlen,
744 params.plaintext.data(), static_cast<int>(params.plaintext.size()));
745 }
746
747 // Finalize encryption
748 int final_len = 0;
749 if (result == 1) {
750 result = EVP_EncryptFinal_ex(ctx, ciphertext.data() + outlen, &final_len);
751 }
752
753 // Get authentication tag
754 if (result == 1) {
755 result = EVP_CIPHER_CTX_ctrl(ctx, EVP_CTRL_AEAD_GET_TAG,
756 static_cast<int>(tag_length),
757 tag.data());
758 }
759
760 EVP_CIPHER_CTX_free(ctx);
761
762 if (result != 1) {
763 DTLSError error = map_openssl_error_detailed();
764 return Result<AEADEncryptionOutput>(error);
765 }
766
767 ciphertext.resize(outlen + final_len);
768
769 AEADEncryptionOutput output;
770 output.ciphertext = std::move(ciphertext);
771 output.tag = std::move(tag);
772
773 return Result<AEADEncryptionOutput>(std::move(output));
774}

776Result<std::vector<uint8_t>> OpenSSLProvider::decrypt_aead(const AEADDecryptionParams& params) {
777 if (!pimpl_->initialized_) {
778 return Result<std::vector<uint8_t>>(DTLSError::NOT_INITIALIZED);
779 }
780
781 if (params.ciphertext.empty() || params.tag.empty()) {
782 return Result<std::vector<uint8_t>>(DTLSError::INVALID_PARAMETER);
783 }
784
785 // Validate AEAD parameters using new helper function
786 auto validation_result = validate_aead_params(params.cipher, params.key, params.nonce);
787 if (!validation_result) {
788 return Result<std::vector<uint8_t>>(validation_result.error());
789 }
790
791 // Get the cipher algorithm and validate tag length
792 const EVP_CIPHER* cipher = nullptr;
793 size_t expected_tag_length = get_aead_tag_length(params.cipher);
794
795 switch (params.cipher) {
796 case AEADCipher::AES_128_GCM:
797 cipher = EVP_aes_128_gcm();
798 break;
799 case AEADCipher::AES_256_GCM:
800 cipher = EVP_aes_256_gcm();
801 break;
802 case AEADCipher::CHACHA20_POLY1305:
803 cipher = EVP_chacha20_poly1305();
804 break;
805 case AEADCipher::AES_128_CCM:
806 case AEADCipher::AES_128_CCM_8:
807 cipher = EVP_aes_128_ccm();
808 break;
809 default:
810 return Result<std::vector<uint8_t>>(DTLSError::OPERATION_NOT_SUPPORTED);
811 }
812
813 if (!cipher || params.tag.size() != expected_tag_length) {
814 return Result<std::vector<uint8_t>>(DTLSError::INVALID_PARAMETER);
815 }
816
817 // Create cipher context
818 EVP_CIPHER_CTX* ctx = EVP_CIPHER_CTX_new();
819 if (!ctx) {
820 return Result<std::vector<uint8_t>>(DTLSError::OUT_OF_MEMORY);
821 }
822
823 // Output buffer
824 std::vector<uint8_t> plaintext(params.ciphertext.size());
825 int outlen = 0;
826 int result = 1;
827
828 // Initialize decryption
829 if (result == 1) {
830 result = EVP_DecryptInit_ex(ctx, cipher, nullptr, nullptr, nullptr);
831 }
832
833 // Set nonce length (for GCM/CCM)
834 if (result == 1 && (params.cipher == AEADCipher::AES_128_GCM ||
835 params.cipher == AEADCipher::AES_256_GCM ||
836 params.cipher == AEADCipher::AES_128_CCM ||
837 params.cipher == AEADCipher::AES_128_CCM_8)) {
838 result = EVP_CIPHER_CTX_ctrl(ctx, EVP_CTRL_AEAD_SET_IVLEN,
839 static_cast<int>(params.nonce.size()), nullptr);
840 }
841
842 // For CCM, set tag length and tag
843 if (result == 1 && (params.cipher == AEADCipher::AES_128_CCM ||
844 params.cipher == AEADCipher::AES_128_CCM_8)) {
845 result = EVP_CIPHER_CTX_ctrl(ctx, EVP_CTRL_AEAD_SET_TAG,
846 static_cast<int>(expected_tag_length),
847 const_cast<uint8_t*>(params.tag.data()));
848 }
849
850 // Set key and nonce
851 if (result == 1) {
852 result = EVP_DecryptInit_ex(ctx, nullptr, nullptr, params.key.data(), params.nonce.data());
853 }
854
855 // For CCM, set ciphertext length
856 if (result == 1 && (params.cipher == AEADCipher::AES_128_CCM ||
857 params.cipher == AEADCipher::AES_128_CCM_8)) {
858 result = EVP_DecryptUpdate(ctx, nullptr, &outlen, nullptr, static_cast<int>(params.ciphertext.size()));
859 }
860
861 // Set additional authenticated data (AAD)
862 if (result == 1 && !params.additional_data.empty()) {
863 result = EVP_DecryptUpdate(ctx, nullptr, &outlen,
864 params.additional_data.data(),
865 static_cast<int>(params.additional_data.size()));
866 }
867
868 // Decrypt ciphertext
869 if (result == 1) {
870 result = EVP_DecryptUpdate(ctx, plaintext.data(), &outlen,
871 params.ciphertext.data(), static_cast<int>(params.ciphertext.size()));
872 }
873
874 // For GCM, set the tag for verification
875 if (result == 1 && (params.cipher == AEADCipher::AES_128_GCM ||
876 params.cipher == AEADCipher::AES_256_GCM ||
877 params.cipher == AEADCipher::CHACHA20_POLY1305)) {
878 result = EVP_CIPHER_CTX_ctrl(ctx, EVP_CTRL_AEAD_SET_TAG,
879 static_cast<int>(expected_tag_length),
880 const_cast<uint8_t*>(params.tag.data()));
881 }
882
883 // Finalize decryption (this verifies the tag)
884 int final_len = 0;
885 if (result == 1) {
886 result = EVP_DecryptFinal_ex(ctx, plaintext.data() + outlen, &final_len);
887 }
888
889 EVP_CIPHER_CTX_free(ctx);
890
891 if (result != 1) {
892 DTLSError error = map_openssl_error_detailed();
893 // For AEAD decryption failures, prefer DECRYPT_ERROR for authentication failures
894 if (error == DTLSError::CRYPTO_PROVIDER_ERROR) {
895 error = DTLSError::DECRYPT_ERROR;
896 }
897 return Result<std::vector<uint8_t>>(error);
898 }
899
900 plaintext.resize(outlen + final_len);
901 return Result<std::vector<uint8_t>>(std::move(plaintext));
902}

997// MAC validation with timing-attack resistance (RFC 9147 Section 5.2)
998Result<bool> OpenSSLProvider::verify_hmac(const MACValidationParams& params) {
999 if (!pimpl_->initialized_) {
1000 return Result<bool>(DTLSError::NOT_INITIALIZED);
1001 }
1002
1003 // Input validation
1004 if (params.key.empty() || params.data.empty() || params.expected_mac.empty()) {
1005 return Result<bool>(DTLSError::INVALID_PARAMETER);
1006 }
1007
1008 // Check maximum data length if specified
1009 if (params.max_data_length > 0 && params.data.size() > params.max_data_length) {
1010 return Result<bool>(DTLSError::INVALID_PARAMETER);
1011 }
1012
1013 // Compute HMAC using existing method
1014 HMACParams hmac_params;
1015 hmac_params.key = params.key;
1016 hmac_params.data = params.data;
1017 hmac_params.algorithm = params.algorithm;
1018
1019 auto computed_hmac_result = compute_hmac(hmac_params);
1020 if (!computed_hmac_result) {
1021 return Result<bool>(computed_hmac_result.error());
1022 }
1023
1024 const auto& computed_hmac = computed_hmac_result.value();
1025
1026 // Always use constant-time comparison to prevent timing attacks
1027 bool is_valid = false;
1028
1029 // Ensure both MACs are always computed with same-size expectations
1030 // This prevents timing attacks based on MAC length validation
1031 bool size_match = (computed_hmac.size() == params.expected_mac.size());
1032
1033 if (params.constant_time_required || true) { // Always use constant-time for security
1034 // Use OpenSSL's CRYPTO_memcmp if available and sizes match
1035 if (size_match) {
1036#if OPENSSL_VERSION_NUMBER >= 0x1010000fL
1037 // OpenSSL 1.1.0+ has CRYPTO_memcmp
1038 is_valid = (CRYPTO_memcmp(computed_hmac.data(), params.expected_mac.data(), computed_hmac.size()) == 0);
1039#else
1040 // Fallback to our enhanced constant-time comparison
1041 is_valid = constant_time_compare(computed_hmac, params.expected_mac);
1042#endif
1043 } else {
1044 // Always use our constant-time comparison for size mismatches
1045 is_valid = constant_time_compare(computed_hmac, params.expected_mac);
1046 }
1047 } else {
1048 // This branch should never be used in production
1049 is_valid = size_match && (computed_hmac == params.expected_mac);
1050 }
1051
1052 return Result<bool>(is_valid);
1053}

1055// DTLS v1.3 record MAC validation (RFC 9147 Section 4.2.1)
1056Result<bool> OpenSSLProvider::validate_record_mac(const RecordMACParams& params) {
1057 if (!pimpl_->initialized_) {
1058 return Result<bool>(DTLSError::NOT_INITIALIZED);
1059 }
1060
1061 // Input validation
1062 if (params.mac_key.empty() || params.record_header.empty() ||
1063 params.plaintext.empty() || params.expected_mac.empty()) {
1064 return Result<bool>(DTLSError::INVALID_PARAMETER);
1065 }
1066
1067 // Construct MAC input according to RFC 9147 Section 4.2.1:
1068 // MAC(MAC_key, seq_num_encrypted || DTLSCiphertext.type ||
1069 // DTLSCiphertext.version || DTLSCiphertext.epoch ||
1070 // DTLSCiphertext.sequence_number || DTLSCiphertext.length ||
1071 // DTLSInnerPlaintext)
1072
1073 std::vector<uint8_t> mac_input;
1074 mac_input.reserve(8 + params.record_header.size() + params.plaintext.size());
1075
1076 // Add encrypted sequence number (8 bytes)
1077 // For now, we'll use the raw sequence number - proper implementation would
1078 // encrypt it using the sequence_number_key
1079 uint64_t seq_num = static_cast<uint64_t>(params.sequence_number);
1080 for (int i = 7; i >= 0; --i) {
1081 mac_input.push_back(static_cast<uint8_t>((seq_num >> (i * 8)) & 0xFF));
1082 }
1083
1084 // Add record header components
1085 mac_input.insert(mac_input.end(), params.record_header.begin(), params.record_header.end());
1086
1087 // Add plaintext
1088 mac_input.insert(mac_input.end(), params.plaintext.begin(), params.plaintext.end());
1089
1090 // Compute HMAC
1091 HMACParams hmac_params;
1092 hmac_params.key = params.mac_key;
1093 hmac_params.data = mac_input;
1094 hmac_params.algorithm = params.mac_algorithm;
1095
1096 auto computed_mac_result = compute_hmac(hmac_params);
1097 if (!computed_mac_result) {
1098 return Result<bool>(computed_mac_result.error());
1099 }
1100
1101 const auto& computed_mac = computed_mac_result.value();
1102
1103 // Always use constant-time comparison to prevent timing attacks
1104 bool is_valid = false;
1105 bool size_match = (computed_mac.size() == params.expected_mac.size());
1106
1107 if (size_match) {
1108#if OPENSSL_VERSION_NUMBER >= 0x1010000fL
1109 // OpenSSL 1.1.0+ has CRYPTO_memcmp
1110 is_valid = (CRYPTO_memcmp(computed_mac.data(), params.expected_mac.data(), computed_mac.size()) == 0);
1111#else
1112 // Fallback to our enhanced constant-time comparison
1113 is_valid = constant_time_compare(computed_mac, params.expected_mac);
1114#endif
1115 } else {
1116 // Always use our constant-time comparison for size mismatches
1117 is_valid = constant_time_compare(computed_mac, params.expected_mac);
1118 }
1119
1120 return Result<bool>(is_valid);
1121}

1254// Random entropy validation for RFC 9147 compliance
1255bool OpenSSLProvider::validate_random_entropy(const std::vector<uint8_t>& random_data) const {
1256 if (random_data.empty()) {
1257 return false;
1258 }
1259
1260 // Simple entropy checks for DTLS v1.3 random values
1261 // These are basic statistical tests, not comprehensive entropy analysis
1262
1263 // 1. Check for all-zero bytes (obvious failure)
1264 bool all_zero = std::all_of(random_data.begin(), random_data.end(),
1265 [](uint8_t byte) { return byte == 0; });
1266 if (all_zero) {
1267 return false;
1268 }
1269
1270 // 2. Check for all-same bytes
1271 bool all_same = std::all_of(random_data.begin(), random_data.end(),
1272 [&](uint8_t byte) { return byte == random_data[0]; });
1273 if (all_same) {
1274 return false;
1275 }
1276
1277 // 3. Basic frequency analysis - no byte value should occur more than 75% of the time
1278 if (random_data.size() >= 16) {
1279 std::array<size_t, 256> byte_count{};
1280 for (uint8_t byte : random_data) {
1281 byte_count[byte]++;
1282 }
1283
1284 size_t max_frequency = *std::max_element(byte_count.begin(), byte_count.end());
1285 double frequency_ratio = static_cast<double>(max_frequency) / random_data.size();
1286
1287 if (frequency_ratio > 0.75) {
1288 return false;
1289 }
1290 }
1291
1292 // 4. Simple runs test - check for excessive runs of consecutive same bits
1293 if (random_data.size() >= 8) {
1294 size_t max_run = 0;
1295 size_t current_run = 1;
1296 uint8_t prev_bit = random_data[0] & 1;
1297
1298 for (size_t i = 1; i < random_data.size(); ++i) {
1299 for (int bit = 0; bit < 8; ++bit) {
1300 uint8_t current_bit = (random_data[i] >> bit) & 1;
1301 if (current_bit == prev_bit) {
1302 current_run++;
1303 } else {
1304 max_run = std::max(max_run, current_run);
1305 current_run = 1;
1306 prev_bit = current_bit;
1307 }
1308 }
1309 }
1310 max_run = std::max(max_run, current_run);
1311
1312 // For 32-byte random (256 bits), max run should be < 32 consecutive bits

1571 return Result<bool>(DTLSError::INVALID_PARAMETER);

1631 EVP_MD_CTX_free(ctx);

1641 params.scheme == SignatureScheme::RSA_PSS_RSAE_SHA384 ||
1642 params.scheme == SignatureScheme::RSA_PSS_RSAE_SHA512) {
1643
1644 result = EVP_PKEY_CTX_set_rsa_padding(pkey_ctx, RSA_PKCS1_PSS_PADDING);
1645 if (result == 1) {

1652 }
1653 }
1654
1655 if (result != 1) {
1656 EVP_MD_CTX_free(ctx);
1657 return Result<bool>(map_openssl_error_detailed());
1658 }
1659 }
1660
1661 // Additional security validations before verification
1662 if (result == 1) {
1663 // For ECDSA signatures, perform ASN.1 format validation
1664 if (utils::is_ecdsa_signature(params.scheme)) {
1665 auto asn1_validation_result = utils::validate_ecdsa_asn1_signature(signature, *params.public_key);
1666 if (!asn1_validation_result) {
1667 EVP_MD_CTX_free(ctx);
1668 return Result<bool>(asn1_validation_result.error());
1669 }
1670 if (!*asn1_validation_result) {
1671 EVP_MD_CTX_free(ctx);
1672 return Result<bool>(false); // Invalid ASN.1 format
1673 }
1674 }
1675
1676 // For EdDSA signatures, perform additional format validation
1677 if (params.scheme == SignatureScheme::ED25519 || params.scheme == SignatureScheme::ED448) {
1678 size_t expected_length = (params.scheme == SignatureScheme::ED25519) ? 64 : 114;
1679 if (signature.size() != expected_length || signature[0] == 0x00) {
1680 EVP_MD_CTX_free(ctx);
1681 return Result<bool>(false);
1682 }
1683 }
1684 }
1685
1686 // Update with data to verify
1687 if (result == 1) {
1688 result = EVP_DigestVerifyUpdate(ctx, params.data.data(), params.data.size());
1689 if (result != 1) {
1690 EVP_MD_CTX_free(ctx);
1691 return Result<bool>(map_openssl_error_detailed());
1692 }
1693 }
1694

1696 // Record start time for timing attack mitigation
1697 auto verification_start = std::chrono::high_resolution_clock::now();
1698
1699 if (result == 1) {
1700 result = EVP_DigestVerifyFinal(ctx, signature.data(), signature.size());
1701 }
1702
1703 // Add minimal delay to make timing more consistent (timing attack mitigation)
1704 auto verification_end = std::chrono::high_resolution_clock::now();
1705 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(
1706 verification_end - verification_start);
1707
1708 // Ensure minimum verification time to reduce timing information leakage
1709 const auto min_verification_time = std::chrono::microseconds(10);
1710 if (duration < min_verification_time) {
1711 std::this_thread::sleep_for(min_verification_time - duration);
1712 }
1713
1714 EVP_MD_CTX_free(ctx);
1715
1716 // Interpret the verification result according to OpenSSL conventions
1717 if (result == 1) {
1718 return Result<bool>(true); // Signature verification succeeded
1719 } else if (result == 0) {
1720 return Result<bool>(false); // Signature verification failed (signature invalid)
1721 } else {
1722 // Error occurred during verification (result < 0)
1723 return Result<bool>(map_openssl_error_detailed());
1724 }
1725}

1916 if (!pimpl_->initialized_) {

1918 }
1919
1920 // Cast to OpenSSL public key
1921 const auto* openssl_key = dynamic_cast<const OpenSSLPublicKey*>(&key);
1922 if (!openssl_key) {
1923 return Result<size_t>(DTLSError::INVALID_PARAMETER);
1924 }
1925
1926 EVP_PKEY* pkey = openssl_key->native_key();
1927 if (!pkey) {
1928 return Result<size_t>(DTLSError::INVALID_PARAMETER);
1929 }
1930
1931 // Get signature length based on key type and scheme
1932 int key_type = EVP_PKEY_base_id(pkey);
1933 size_t signature_length = 0;
1934
1935 switch (scheme) {
1936 // RSA signatures - length depends on key size
1937 case SignatureScheme::RSA_PKCS1_SHA256:
1938 case SignatureScheme::RSA_PKCS1_SHA384:
1939 case SignatureScheme::RSA_PKCS1_SHA512:
1940 case SignatureScheme::RSA_PSS_RSAE_SHA256:
1941 case SignatureScheme::RSA_PSS_RSAE_SHA384:
1942 case SignatureScheme::RSA_PSS_RSAE_SHA512:
1943 if (key_type == EVP_PKEY_RSA) {
1944 signature_length = static_cast<size_t>(EVP_PKEY_size(pkey));
1945 } else {
1946 return Result<size_t>(DTLSError::INVALID_PARAMETER);
1947 }
1948 break;
1949
1950 // ECDSA signatures - variable length, estimate maximum
1951 case SignatureScheme::ECDSA_SECP256R1_SHA256:
1952 if (key_type == EVP_PKEY_EC) {
1953 signature_length = 72; // Maximum for P-256 (2 * 32 + DER overhead)
1954 } else {
1955 return Result<size_t>(DTLSError::INVALID_PARAMETER);
1956 }
1957 break;
1958
1959 case SignatureScheme::ECDSA_SECP384R1_SHA384:
1960 if (key_type == EVP_PKEY_EC) {
1961 signature_length = 104; // Maximum for P-384 (2 * 48 + DER overhead)
1962 } else {
1963 return Result<size_t>(DTLSError::INVALID_PARAMETER);
1964 }
1965 break;
1966
1967 case SignatureScheme::ECDSA_SECP521R1_SHA512:
1968 if (key_type == EVP_PKEY_EC) {
1969 signature_length = 139; // Maximum for P-521 (2 * 66 + DER overhead)
1970 } else {
1971 return Result<size_t>(DTLSError::INVALID_PARAMETER);
1972 }
1973 break;
1974
1975 // EdDSA signatures - fixed length
1976 case SignatureScheme::ED25519:
1977 if (key_type == EVP_PKEY_ED25519) {
1978 signature_length = 64; // Ed25519 signature is always 64 bytes
1979 } else {
1980 return Result<size_t>(DTLSError::INVALID_PARAMETER);
1981 }
1982 break;
1983
1984 case SignatureScheme::ED448:
1985 if (key_type == EVP_PKEY_ED448) {
1986 signature_length = 114; // Ed448 signature is always 114 bytes
1987 } else {
1988 return Result<size_t>(DTLSError::INVALID_PARAMETER);
1989 }
1990 break;
1991
1992 default:
1993 return Result<size_t>(DTLSError::OPERATION_NOT_SUPPORTED);
1994 }
1995
1996 return Result<size_t>(signature_length);
1997}

2022Result<std::vector<uint8_t>> OpenSSLProvider::sign_handshake_transcript(
2023 const std::vector<uint8_t>& transcript_hash,
2024 SignatureScheme scheme,
2025 const PrivateKey& private_key,
2026 bool is_server) const {
2027
2028 if (!pimpl_->initialized_) {
2029 return Result<std::vector<uint8_t>>(DTLSError::NOT_INITIALIZED);
2030 }

2032 if (transcript_hash.empty()) {
2033 return Result<std::vector<uint8_t>>(DTLSError::INVALID_PARAMETER);
2034 }
2035
2036 // DTLS v1.3 handshake signature includes context strings per RFC 9147
2037 std::string context_string;
2038 if (is_server) {
2039 context_string = "TLS 1.3, server CertificateVerify";
2040 } else {
2041 context_string = "TLS 1.3, client CertificateVerify";
2042 }
2043

2045 std::vector<uint8_t> content;
2046
2047 // Add 64 space (0x20) characters
2048 content.insert(content.end(), 64, 0x20);
2049
2050 // Add context string
2051 content.insert(content.end(), context_string.begin(), context_string.end());
2052
2053 // Add separator byte (0x00)
2054 content.push_back(0x00);
2055
2056 // Add transcript hash
2057 content.insert(content.end(), transcript_hash.begin(), transcript_hash.end());
2058
2059 // Create signature parameters
2060 SignatureParams params;
2061 params.data = content;
2062 params.scheme = scheme;
2063 params.private_key = &private_key;
2064
2065 // Use the main sign_data method (cast away const since we're not modifying state)
2066 return const_cast<OpenSSLProvider*>(this)->sign_data(params);
2067}

2357Result<std::vector<uint8_t>> OpenSSLProvider::perform_key_exchange(const KeyExchangeParams& params) {
2358 if (!pimpl_->initialized_) {
2359 return Result<std::vector<uint8_t>>(DTLSError::NOT_INITIALIZED);
2360 }
2361
2362 if (!params.private_key || params.peer_public_key.empty()) {
2363 return Result<std::vector<uint8_t>>(DTLSError::INVALID_PARAMETER);
2364 }
2365
2366 // Cast to OpenSSL private key
2367 const auto* openssl_private_key = dynamic_cast<const OpenSSLPrivateKey*>(params.private_key);
2368 if (!openssl_private_key) {
2369 return Result<std::vector<uint8_t>>(DTLSError::INVALID_PARAMETER);
2370 }
2371
2372 EVP_PKEY* private_key = openssl_private_key->native_key();
2373 if (!private_key) {
2374 return Result<std::vector<uint8_t>>(DTLSError::INVALID_PARAMETER);
2375 }
2376
2377 // Create peer public key from raw bytes
2378 EVP_PKEY* peer_key = nullptr;
2379
2380 switch (params.group) {
2381 case NamedGroup::X25519: {
2382 peer_key = EVP_PKEY_new_raw_public_key(EVP_PKEY_X25519, nullptr,
2383 params.peer_public_key.data(),
2384 params.peer_public_key.size());
2385 break;
2386 }
2387 case NamedGroup::X448: {
2388 peer_key = EVP_PKEY_new_raw_public_key(EVP_PKEY_X448, nullptr,
2389 params.peer_public_key.data(),
2390 params.peer_public_key.size());
2391 break;
2392 }
2393 case NamedGroup::SECP256R1:
2394 case NamedGroup::SECP384R1:
2395 case NamedGroup::SECP521R1: {
2396 // For ECDH, we need to reconstruct the EC public key
2397 int curve_nid;
2398 switch (params.group) {
2399 case NamedGroup::SECP256R1: curve_nid = NID_X9_62_prime256v1; break;
2400 case NamedGroup::SECP384R1: curve_nid = NID_secp384r1; break;
2401 case NamedGroup::SECP521R1: curve_nid = NID_secp521r1; break;
2402 default: curve_nid = NID_X9_62_prime256v1; break;
2403 }
2404
2405 EC_KEY* ec_key = EC_KEY_new_by_curve_name(curve_nid);
2406 if (!ec_key) {
2407 return Result<std::vector<uint8_t>>(DTLSError::KEY_EXCHANGE_FAILED);
2408 }
2409
2410 const EC_GROUP* group = EC_KEY_get0_group(ec_key);
2411 EC_POINT* point = EC_POINT_new(group);
2412
2413 if (point && EC_POINT_oct2point(group, point, params.peer_public_key.data(),
2414 params.peer_public_key.size(), nullptr) == 1) {
2415 EC_KEY_set_public_key(ec_key, point);
2416 peer_key = EVP_PKEY_new();
2417 if (peer_key) {
2418 EVP_PKEY_assign_EC_KEY(peer_key, ec_key);
2419 } else {
2420 EC_KEY_free(ec_key);
2421 }
2422 } else {
2423 EC_KEY_free(ec_key);
2424 }
2425
2426 if (point) EC_POINT_free(point);
2427 break;
2428 }
2429 default:
2430 return Result<std::vector<uint8_t>>(DTLSError::OPERATION_NOT_SUPPORTED);
2431 }
2432
2433 if (!peer_key) {
2434 return Result<std::vector<uint8_t>>(DTLSError::KEY_EXCHANGE_FAILED);
2435 }
2436
2437 // Perform key derivation
2438 EVP_PKEY_CTX* ctx = EVP_PKEY_CTX_new(private_key, nullptr);
2439 if (!ctx) {
2440 EVP_PKEY_free(peer_key);
2441 return Result<std::vector<uint8_t>>(DTLSError::OUT_OF_MEMORY);
2442 }
2443
2444 int result = 1;
2445 if (result == 1) {
2446 result = EVP_PKEY_derive_init(ctx);
2447 }
2448
2449 if (result == 1) {
2450 result = EVP_PKEY_derive_set_peer(ctx, peer_key);
2451 }
2452
2453 // Determine output length
2454 size_t shared_secret_len = 0;
2455 if (result == 1) {
2456 result = EVP_PKEY_derive(ctx, nullptr, &shared_secret_len);
2457 }
2458
2459 std::vector<uint8_t> shared_secret;
2460 if (result == 1 && shared_secret_len > 0) {
2461 shared_secret.resize(shared_secret_len);
2462 result = EVP_PKEY_derive(ctx, shared_secret.data(), &shared_secret_len);
2463 }
2464
2465 EVP_PKEY_CTX_free(ctx);
2466 EVP_PKEY_free(peer_key);
2467
2468 if (result != 1) {
2469 return Result<std::vector<uint8_t>>(DTLSError::KEY_EXCHANGE_FAILED);
2470 }
2471
2472 shared_secret.resize(shared_secret_len);
2473 return Result<std::vector<uint8_t>>(std::move(shared_secret));
2474}

2476Result<bool> OpenSSLProvider::validate_certificate_chain(const CertValidationParams& params) {
2477 if (!pimpl_->initialized_) {
2478 return Result<bool>(DTLSError::NOT_INITIALIZED);
2479 }
2480
2481 if (!params.chain || params.chain->certificate_count() == 0) {
2482 return Result<bool>(DTLSError::INVALID_PARAMETER);
2483 }
2484
2485 // Create X509_STORE for certificate verification
2486 X509_STORE* store = X509_STORE_new();
2487 if (!store) {
2488 return Result<bool>(DTLSError::OUT_OF_MEMORY);
2489 }
2490
2491 // Load default trusted certificates
2492 X509_STORE_set_default_paths(store);
2493
2494 // Parse the certificate chain
2495 std::vector<X509*> cert_chain;
2496
2497 for (size_t i = 0; i < params.chain->certificate_count(); ++i) {
2498 auto cert_data = params.chain->certificate_at(i);
2499 BIO* bio = BIO_new_mem_buf(cert_data.data(), static_cast<int>(cert_data.size()));
2500 if (!bio) {
2501 X509_STORE_free(store);
2502 for (X509* cert : cert_chain) {
2503 X509_free(cert);
2504 }
2505 return Result<bool>(DTLSError::OUT_OF_MEMORY);
2506 }
2507
2508 X509* cert = nullptr;
2509
2510 // Try DER format first
2511 cert = d2i_X509_bio(bio, nullptr);
2512 if (!cert) {
2513 // Reset BIO and try PEM format
2514 BIO_reset(bio);
2515 cert = PEM_read_bio_X509(bio, nullptr, nullptr, nullptr);
2516 }
2517
2518 BIO_free(bio);
2519
2520 if (!cert) {
2521 X509_STORE_free(store);
2522 for (X509* c : cert_chain) {
2523 X509_free(c);
2524 }
2525 return Result<bool>(DTLSError::CERTIFICATE_UNKNOWN);
2526 }
2527
2528 cert_chain.push_back(cert);
2529 }
2530
2531 if (cert_chain.empty()) {

2533 return Result<bool>(DTLSError::CERTIFICATE_UNKNOWN);
2534 }
2535
2536 // Create X509_STORE_CTX for verification
2537 X509_STORE_CTX* ctx = X509_STORE_CTX_new();
2538 if (!ctx) {
2539 X509_STORE_free(store);
2540 for (X509* cert : cert_chain) {
2541 X509_free(cert);
2542 }
2543 return Result<bool>(DTLSError::OUT_OF_MEMORY);
2544 }
2545
2546 // Create certificate stack for intermediate certificates
2547 STACK_OF(X509)* untrusted = nullptr;
2548 if (cert_chain.size() > 1) {
2549 untrusted = sk_X509_new_null();
2550 if (!untrusted) {
2551 X509_STORE_CTX_free(ctx);
2552 X509_STORE_free(store);
2553 for (X509* cert : cert_chain) {
2554 X509_free(cert);
2555 }
2556 return Result<bool>(DTLSError::OUT_OF_MEMORY);
2557 }
2558
2559 // Add intermediate certificates to the untrusted stack
2560 for (size_t i = 1; i < cert_chain.size(); ++i) {
2561 sk_X509_push(untrusted, cert_chain[i]);
2562 }
2563 }
2564
2565 // Initialize verification context
2566 int init_result = X509_STORE_CTX_init(ctx, store, cert_chain[0], untrusted);
2567 if (init_result != 1) {
2568 if (untrusted) {
2569 sk_X509_free(untrusted);
2570 }
2571 X509_STORE_CTX_free(ctx);
2572 X509_STORE_free(store);
2573 for (X509* cert : cert_chain) {
2574 X509_free(cert);
2575 }
2576 return Result<bool>(DTLSError::CRYPTO_PROVIDER_ERROR);
2577 }
2578
2579 // Set verification flags based on parameters
2580 unsigned long flags = 0;
2581 if (!params.check_revocation) {
2582 flags |= X509_V_FLAG_NO_CHECK_TIME;
2583 }
2584
2585 if (flags != 0) {
2586 X509_STORE_CTX_set_flags(ctx, flags);
2587 }
2588
2589 // Perform certificate chain verification
2590 int verify_result = X509_verify_cert(ctx);
2591 bool is_valid = (verify_result == 1);
2592
2593 // Get verification error if validation failed
2594 if (!is_valid) {
2595 int error = X509_STORE_CTX_get_error(ctx);
2596 // Log the specific verification error for debugging
2597 // const char* error_string = X509_verify_cert_error_string(error);
2598 (void)error; // Suppress unused variable warning
2599 }
2600
2601 // Cleanup
2602 if (untrusted) {
2603 sk_X509_free(untrusted);
2604 }
2605 X509_STORE_CTX_free(ctx);
2606 X509_STORE_free(store);
2607 for (X509* cert : cert_chain) {
2608 X509_free(cert);
2609 }
2610
2611 return Result<bool>(is_valid);

3662// Hybrid Key Exchange Implementation (draft-kwiatkowski-tls-ecdhe-mlkem-03)
3663Result<HybridKeyExchangeResult> OpenSSLProvider::perform_hybrid_key_exchange(const HybridKeyExchangeParams& params) {
3664 if (!pimpl_->initialized_) {
3665 return Result<HybridKeyExchangeResult>(DTLSError::NOT_INITIALIZED);
3666 }
3667
3668 if (!hybrid_pqc::is_hybrid_pqc_group(params.hybrid_group)) {
3669 return Result<HybridKeyExchangeResult>(DTLSError::OPERATION_NOT_SUPPORTED);
3670 }
3671
3672 HybridKeyExchangeResult result;
3673
3674 // Get the classical ECDHE group and ML-KEM parameter set
3675 auto classical_group = hybrid_pqc::get_classical_group(params.hybrid_group);
3676 auto mlkem_param_set = hybrid_pqc::get_mlkem_parameter_set(params.hybrid_group);
3677
3678 // Perform classical ECDHE key exchange
3679 KeyExchangeParams classical_params;
3680 classical_params.group = classical_group;
3681 classical_params.peer_public_key = params.classical_peer_public_key;
3682 classical_params.private_key = params.classical_private_key;
3683
3684 auto classical_result = perform_key_exchange(classical_params);
3685 if (!classical_result) {
3686 return Result<HybridKeyExchangeResult>(classical_result.error());
3687 }
3688 result.classical_shared_secret = *classical_result;
3689
3690 // Perform ML-KEM operation (encapsulation or decapsulation)
3691 if (params.is_encapsulation) {
3692 // Client side: encapsulate with server's ML-KEM public key
3693 MLKEMEncapParams encap_params;
3694 encap_params.parameter_set = mlkem_param_set;
3695 encap_params.public_key = params.pq_peer_public_key;
3696
3697 auto encap_result = mlkem_encapsulate(encap_params);
3698 if (!encap_result) {
3699 return Result<HybridKeyExchangeResult>(encap_result.error());
3700 }
3701
3702 result.pq_ciphertext = encap_result->ciphertext;
3703 result.pq_shared_secret = encap_result->shared_secret;
3704 } else {
3705 // Server side: decapsulate client's ML-KEM ciphertext
3706 MLKEMDecapParams decap_params;
3707 decap_params.parameter_set = mlkem_param_set;
3708 decap_params.private_key = params.pq_private_key;
3709 decap_params.ciphertext = params.pq_peer_public_key; // Actually ciphertext in this case
3710
3711 auto decap_result = mlkem_decapsulate(decap_params);
3712 if (!decap_result) {
3713 return Result<HybridKeyExchangeResult>(decap_result.error());
3714 }
3715
3716 result.pq_shared_secret = *decap_result;
3717 }
3718
3719 // Combine the shared secrets using HKDF as specified in draft-kwiatkowski-tls-ecdhe-mlkem-03
3720 // combined_shared_secret = HKDF-Extract(salt=0, IKM = ecdh_ss || mlkem_ss)
3721
3722 std::vector<uint8_t> combined_ikm;
3723 combined_ikm.insert(combined_ikm.end(), result.classical_shared_secret.begin(), result.classical_shared_secret.end());
3724 combined_ikm.insert(combined_ikm.end(), result.pq_shared_secret.begin(), result.pq_shared_secret.end());
3725
3726 // Use HKDF-Extract with empty salt (zero-filled)
3727 KeyDerivationParams hkdf_params;
3728 hkdf_params.secret = combined_ikm;
3729 hkdf_params.salt = std::vector<uint8_t>(32, 0); // SHA-256 output length
3730 hkdf_params.info.clear();
3731 hkdf_params.output_length = 32; // Output 32 bytes
