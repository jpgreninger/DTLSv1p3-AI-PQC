23Result<uint64_t> ResourceManager::allocate_connection_resources(
24 const NetworkAddress& source_address,
25 size_t memory_estimate) {
26
27 std::string source_key = address_to_key(source_address);
28
29 // Check if allocation would succeed
30 if (!can_allocate(source_address, ResourceType::CONNECTION_SLOT, 1) ||
31 !can_allocate(source_address, ResourceType::CONNECTION_MEMORY, memory_estimate)) {
32
33 record_allocation_failure(ResourceResult::CONNECTION_LIMIT_EXCEEDED);
34 return make_error<uint64_t>(DTLSError::RESOURCE_EXHAUSTED,
35 "Connection allocation would exceed limits");
36 }
37
38 // Generate allocation ID
39 uint64_t allocation_id = generate_allocation_id();
40
41 // Create allocation record
42 auto allocation = std::make_unique<ResourceAllocation>(
43 ResourceType::CONNECTION_SLOT, memory_estimate, source_key);
44
45 // Update source data
46 auto* source_data = get_or_create_source_data(source_address);
47 if (!source_data) {
48 record_allocation_failure(ResourceResult::RESOURCE_UNAVAILABLE);
49 return make_error<uint64_t>(DTLSError::RESOURCE_EXHAUSTED,
50 "Cannot create source data");
51 }
52
53 {
54 std::lock_guard<std::mutex> lock(source_data->allocation_mutex);
55 source_data->total_memory += memory_estimate;
56 source_data->connection_count++;
57 source_data->allocation_ids.insert(allocation_id);
58 source_data->last_activity = std::chrono::steady_clock::now();
59 }
60
61 // Update global counters
62 total_allocated_memory_ += memory_estimate;
63 connection_memory_ += memory_estimate;
64 total_connections_++;
65
66 // Store allocation
67 {
68 std::unique_lock<std::shared_mutex> lock(allocations_mutex_);
69 allocations_[allocation_id] = std::move(allocation);
70 }
71
72 // Update statistics
73 {
74 std::lock_guard<std::mutex> lock(stats_mutex_);
75 stats_.total_allocated_memory = total_allocated_memory_.load();
76 stats_.connection_memory = connection_memory_.load();
77 stats_.total_connections = total_connections_.load();
78 stats_.active_connections++;
79 if (stats_.total_allocated_memory > stats_.peak_memory_usage) {
80 stats_.peak_memory_usage = stats_.total_allocated_memory;
81 }
82 }
83
84 update_pressure_level();
85
86 return make_result(allocation_id);

215Result<void> ResourceManager::release_resources(uint64_t allocation_id) {
216 ResourceAllocation* allocation = nullptr;
217
218 {
219 std::shared_lock<std::shared_mutex> lock(allocations_mutex_);
220 auto it = allocations_.find(allocation_id);
221 if (it == allocations_.end()) {
222 return make_error<void>(DTLSError::MESSAGE_NOT_FOUND, "Allocation ID not found");
223 }
224 allocation = it->second.get();
225 }
226
227 // Update source data
228 {
229 std::unique_lock<std::shared_mutex> source_lock(source_data_mutex_);
230 auto source_it = source_data_.find(allocation->source_key);
231 if (source_it != source_data_.end()) {
232 auto& source_data = source_it->second;
233 std::lock_guard<std::mutex> lock(source_data->allocation_mutex);
234
235 source_data->total_memory -= allocation->amount;
236 source_data->allocation_ids.erase(allocation_id);
237
238 switch (allocation->type) {
239 case ResourceType::CONNECTION_SLOT:
240 case ResourceType::CONNECTION_MEMORY:
241 if (source_data->connection_count > 0) {
242 source_data->connection_count--;
243 }
244 connection_memory_ -= allocation->amount;
245 if (stats_.active_connections > 0) {
246 std::lock_guard<std::mutex> stats_lock(stats_mutex_);
247 stats_.active_connections--;
248 }
249 break;
250 case ResourceType::HANDSHAKE_SLOT:
251 case ResourceType::HANDSHAKE_MEMORY:
252 if (source_data->handshake_count > 0) {
253 source_data->handshake_count--;
254 }
255 handshake_memory_ -= allocation->amount;
256 if (pending_handshakes_ > 0) {
257 pending_handshakes_--;
258 }
259 {
260 std::lock_guard<std::mutex> stats_lock(stats_mutex_);
261 stats_.completed_handshakes++;
262 stats_.pending_handshakes = pending_handshakes_.load();
263 }
264 break;
265 case ResourceType::BUFFER_MEMORY:
266 source_data->buffer_memory -= allocation->amount;
267 buffer_memory_ -= allocation->amount;
268 break;
269 }
270 }
271 }
272
273 // Update global counters
274 total_allocated_memory_ -= allocation->amount;
275
276 // Remove allocation
277 {
278 std::unique_lock<std::shared_mutex> lock(allocations_mutex_);
279 allocations_.erase(allocation_id);
280 }
281
282 // Update statistics
283 {
284 std::lock_guard<std::mutex> lock(stats_mutex_);
285 stats_.total_allocated_memory = total_allocated_memory_.load();
286 stats_.connection_memory = connection_memory_.load();
287 stats_.handshake_memory = handshake_memory_.load();
288 stats_.buffer_memory = buffer_memory_.load();
289 }
290
291 update_pressure_level();
292
293 return make_result();
294}

315bool ResourceManager::can_allocate(const NetworkAddress& source_address,
316 ResourceType type,
317 size_t amount) const {
318
319 std::string source_key = address_to_key(source_address);
320
321 // Check global limits first
322 switch (type) {
323 case ResourceType::CONNECTION_SLOT:
324 if (total_connections_.load() >= config_.max_total_connections) {
325 return false;
326 }
327 break;
328 case ResourceType::HANDSHAKE_SLOT:
329 if (pending_handshakes_.load() >= config_.max_pending_handshakes) {
330 return false;
331 }
332 break;
333 case ResourceType::CONNECTION_MEMORY:
334 case ResourceType::HANDSHAKE_MEMORY:
335 case ResourceType::BUFFER_MEMORY:
336 if (total_allocated_memory_.load() + amount > config_.max_total_memory) {
337 return false;
338 }
339 break;
340 }
341
342 // Check memory limits
343 if (!check_memory_limits(source_key, amount)) {
344 return false;
345 }
346
347 // Check connection limits
348 if (!check_connection_limits(source_key, type)) {
349 return false;
350 }
351
352 return true;
353}

465 auto now = std::chrono::steady_clock::now();
466
467 // Only check periodically
468 if (now - last_health_check_.load() < std::chrono::seconds{10}) {
469 return current_pressure_.load();
470 }
471
472 update_pressure_level();
473
474 // Trigger cleanup if under pressure
475 auto pressure = current_pressure_.load();

496 std::vector<NetworkAddress> high_usage_sources;
497
498 std::shared_lock<std::shared_mutex> lock(source_data_mutex_);
499 for (const auto& [key, source_data] : source_data_) {
500 double memory_usage = static_cast<double>(source_data->total_memory.load()) /
501 config_.max_memory_per_connection;
502 double connection_usage = static_cast<double>(source_data->connection_count.load()) /
503 config_.max_connections_per_source;
504
505 if (memory_usage > threshold || connection_usage > threshold) {
506 // Parse key back to NetworkAddress (simplified)
507 // In a real implementation, you'd want a more robust method
508 try {
509 size_t colon_pos = key.find(':');
510 if (colon_pos != std::string::npos) {
511 std::string ip = key.substr(0, colon_pos);
512 uint16_t port = static_cast<uint16_t>(std::stoi(key.substr(colon_pos + 1)));
513 high_usage_sources.emplace_back(ip, port);
514 }
515 } catch (...) {
516 // Skip malformed keys
517 }
518 }
519 }
520
521 return high_usage_sources;

642void ResourceManager::update_pressure_level() {
643 double memory_usage = get_memory_usage_percentage();
644 double connection_usage = get_connection_usage_percentage();
645
646 PressureLevel new_pressure = PressureLevel::NORMAL;
647
648 if (memory_usage >= config_.memory_critical_threshold ||
649 connection_usage >= config_.connection_critical_threshold) {
650 new_pressure = PressureLevel::CRITICAL;
651 } else if (memory_usage >= config_.memory_warning_threshold ||
652 connection_usage >= config_.connection_warning_threshold) {
653 new_pressure = PressureLevel::WARNING;
654 }
655
656 // Check for emergency conditions
657 if (memory_usage >= 0.99 || connection_usage >= 0.99) {
658 new_pressure = PressureLevel::EMERGENCY;
659 }
660
661 auto old_pressure = current_pressure_.exchange(new_pressure);
662
663 if (new_pressure > old_pressure) {
664 std::lock_guard<std::mutex> lock(stats_mutex_);
665 stats_.pressure_events++;
666 stats_.last_pressure_event = std::chrono::steady_clock::now();
667 stats_.current_pressure = new_pressure;
668 }
669}

671size_t ResourceManager::cleanup_expired_allocations() {
672 size_t cleaned = 0;
673 auto now = std::chrono::steady_clock::now();
674 std::vector<uint64_t> to_cleanup;
675
676 {
677 std::shared_lock<std::shared_mutex> lock(allocations_mutex_);
678 for (const auto& [id, allocation] : allocations_) {
679 auto age = now - allocation->last_activity;
680 bool should_cleanup = false;
681
682 switch (allocation->type) {
683 case ResourceType::CONNECTION_SLOT:
684 case ResourceType::CONNECTION_MEMORY:
685 should_cleanup = age > config_.connection_timeout;
686 break;
687 case ResourceType::HANDSHAKE_SLOT:
688 case ResourceType::HANDSHAKE_MEMORY:
689 should_cleanup = age > config_.handshake_timeout;
690 break;
691 case ResourceType::BUFFER_MEMORY:
692 should_cleanup = age > (config_.connection_timeout * 2);
693 break;
694 }
695
696 if (should_cleanup) {
697 to_cleanup.push_back(id);
698 }
699 }
700 }
701
702 for (uint64_t id : to_cleanup) {
703 if (release_resources(id).is_success()) {
704 cleaned++;
705 }
706 }
707
708 return cleaned;
709}
