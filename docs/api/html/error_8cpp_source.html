146bool is_fatal_error(DTLSError error) {
147 // Determine if an error is fatal and should terminate the connection
148 switch (error) {
149 // Non-fatal errors that allow retry or recovery
150 case DTLSError::SUCCESS:
151 case DTLSError::TIMEOUT:
152 case DTLSError::OPERATION_ABORTED:
153 case DTLSError::RESOURCE_UNAVAILABLE:
154 case DTLSError::CONNECTION_TIMEOUT:
155 case DTLSError::NETWORK_UNREACHABLE:
156 case DTLSError::HOST_UNREACHABLE:
157 case DTLSError::PORT_UNREACHABLE:
158 case DTLSError::USER_CANCELED:
159 case DTLSError::RATE_LIMITED:
160 case DTLSError::SERVICE_UNAVAILABLE:
161 case DTLSError::MAINTENANCE_MODE:
162 // RFC 9147: Invalid records should be silently discarded (non-fatal)
163 case DTLSError::INVALID_RECORD_HEADER:
164 return false;
165
166 // Fatal errors that require connection termination
167 case DTLSError::PROTOCOL_VERSION_NOT_SUPPORTED:
168 case DTLSError::HANDSHAKE_FAILURE:
169 case DTLSError::CERTIFICATE_VERIFY_FAILED:
170 case DTLSError::CERTIFICATE_EXPIRED:
171 case DTLSError::CERTIFICATE_REVOKED:
172 case DTLSError::UNKNOWN_CA:
173 case DTLSError::ACCESS_DENIED:
174 case DTLSError::DECODE_ERROR:
175 case DTLSError::ILLEGAL_PARAMETER:
176 case DTLSError::DECRYPT_ERROR:
177 case DTLSError::BAD_RECORD_MAC:
178 case DTLSError::SIGNATURE_VERIFICATION_FAILED:
179 case DTLSError::CIPHER_SUITE_NOT_SUPPORTED:
180 case DTLSError::KEY_EXCHANGE_FAILED:
181 case DTLSError::INSUFFICIENT_SECURITY:
182 case DTLSError::REPLAY_ATTACK_DETECTED:
183 case DTLSError::TAMPERING_DETECTED:
184 case DTLSError::SECURITY_POLICY_VIOLATION:
185 case DTLSError::AUTHENTICATION_FAILED:
186 case DTLSError::AUTHORIZATION_FAILED:
187 case DTLSError::PERMISSION_DENIED:
188 return true;
189
190 default:
191 // Conservative approach: treat unknown errors as fatal
192 return true;
193 }
