32// Handshake fragment information
33struct HandshakeFragment {
34 uint8_t message_type{0};
35 uint32_t message_length{0};
36 uint32_t fragment_offset{0};
37 uint32_t fragment_length{0};
38 uint16_t message_sequence{0};
39 std::shared_ptr<ZeroCopyBuffer> fragment_data;
40 std::chrono::steady_clock::time_point received_time;
41
42 bool is_complete() const {
43 return fragment_offset == 0 && fragment_length == message_length;
44 }
45
46 bool is_valid() const {
47 return fragment_data && fragment_data->size() == fragment_length &&
48 fragment_offset + fragment_length <= message_length;
49 }
50};
