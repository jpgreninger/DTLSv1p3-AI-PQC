9Result<void> SequenceNumber48::serialize_to_buffer(uint8_t* buffer) const {
10 if (!buffer) {
11 return make_error<void>(DTLSError::INTERNAL_ERROR, "Null buffer provided");
12 }
13
14 // Store 48-bit value in big-endian format
15 uint64_t val = value;
16 buffer[0] = (val >> 40) & 0xFF;
17 buffer[1] = (val >> 32) & 0xFF;
18 buffer[2] = (val >> 24) & 0xFF;
19 buffer[3] = (val >> 16) & 0xFF;
20 buffer[4] = (val >> 8) & 0xFF;
21 buffer[5] = val & 0xFF;
22
23 return make_result();
24}

28 return make_error<SequenceNumber48>(DTLSError::INTERNAL_ERROR, "Null buffer provided");
29 }
30
31 // Read 48-bit value from big-endian format
32 uint64_t val = 0;
33 val |= (static_cast<uint64_t>(buffer[0]) << 40);
34 val |= (static_cast<uint64_t>(buffer[1]) << 32);
35 val |= (static_cast<uint64_t>(buffer[2]) << 24);
36 val |= (static_cast<uint64_t>(buffer[3]) << 16);
37 val |= (static_cast<uint64_t>(buffer[4]) << 8);
38 val |= static_cast<uint64_t>(buffer[5]);

85Result<size_t> DTLSPlaintext::serialize(memory::Buffer& buffer) const {
86 size_t total_size = HEADER_SIZE + fragment.size();
87
88 // Ensure buffer has enough space
89 auto resize_result = buffer.resize(total_size);
90 if (!resize_result.is_success()) {
91 return make_error<size_t>(DTLSError::INTERNAL_ERROR, "Failed to resize buffer");
92 }
93
94 uint8_t* data = reinterpret_cast<uint8_t*>(buffer.mutable_data());
95 size_t offset = 0;
96
97 // Serialize header fields
98 data[offset++] = static_cast<uint8_t>(type);
99
100 // Protocol version (big-endian uint16_t)
101 uint16_t version_raw = static_cast<uint16_t>(version);
102 data[offset++] = (version_raw >> 8) & 0xFF;
103 data[offset++] = version_raw & 0xFF;
104
105 // Epoch (big-endian uint16_t)
106 data[offset++] = (epoch >> 8) & 0xFF;
107 data[offset++] = epoch & 0xFF;
108
109 // Sequence number (48-bit, big-endian)
110 auto seq_result = sequence_number.serialize_to_buffer(&data[offset]);
111 if (!seq_result.is_success()) {
112 return make_error<size_t>(seq_result.error(), "Failed to serialize sequence number");
113 }
114 offset += SequenceNumber48::SERIALIZED_SIZE;
115
116 // Length (big-endian uint16_t)
117 data[offset++] = (length >> 8) & 0xFF;
118 data[offset++] = length & 0xFF;
119
120 // Fragment data
121 if (fragment.size() > 0) {
122 std::memcpy(&data[offset], fragment.data(), fragment.size());
123 offset += fragment.size();
124 }
125
126 return make_result(offset);
127}

129Result<DTLSPlaintext> DTLSPlaintext::deserialize(const memory::Buffer& buffer, size_t offset) {
130 if (buffer.size() < offset + HEADER_SIZE) {
131 return make_error<DTLSPlaintext>(DTLSError::DECODE_ERROR, "Buffer too small for DTLS header");
132 }
133
134 const uint8_t* data = reinterpret_cast<const uint8_t*>(buffer.data());
135 size_t pos = offset;
136
137 // Deserialize header fields
138 ContentType content_type = static_cast<ContentType>(data[pos++]);
139
140 // Protocol version (big-endian uint16_t)
141 uint16_t version_raw = (static_cast<uint16_t>(data[pos]) << 8) | data[pos + 1];
142 ProtocolVersion proto_version = static_cast<ProtocolVersion>(version_raw);
143 pos += 2;
144
145 // Epoch (big-endian uint16_t)
146 uint16_t epoch_val = (static_cast<uint16_t>(data[pos]) << 8) | data[pos + 1];
147 pos += 2;
148
149 // Sequence number (48-bit, big-endian)
150 auto seq_result = SequenceNumber48::deserialize_from_buffer(&data[pos]);
151 if (!seq_result.is_success()) {
152 return make_error<DTLSPlaintext>(seq_result.error(), "Failed to deserialize sequence number");
153 }
154 SequenceNumber48 seq_num = seq_result.value();
155 pos += SequenceNumber48::SERIALIZED_SIZE;
156
157 // Length (big-endian uint16_t)
158 uint16_t fragment_length = (static_cast<uint16_t>(data[pos]) << 8) | data[pos + 1];
159 pos += 2;
160
161 // Validate length
162 if (fragment_length > MAX_FRAGMENT_LENGTH) {
163 return make_error<DTLSPlaintext>(DTLSError::RECORD_OVERFLOW, "Fragment length exceeds maximum");
164 }
165
166 if (buffer.size() < pos + fragment_length) {
167 return make_error<DTLSPlaintext>(DTLSError::DECODE_ERROR, "Buffer too small for fragment");
168 }
169
170 // Extract fragment
171 memory::Buffer fragment_buf;
172 if (fragment_length > 0) {
173 fragment_buf = memory::Buffer(fragment_length);
174 auto resize_result = fragment_buf.resize(fragment_length);
175 if (!resize_result.is_success()) {
176 return make_error<DTLSPlaintext>(DTLSError::INTERNAL_ERROR, "Failed to allocate fragment buffer");
177 }
178 std::memcpy(fragment_buf.mutable_data(), &data[pos], fragment_length);
179 }
180
181 DTLSPlaintext record(content_type, proto_version, epoch_val, seq_num, std::move(fragment_buf));
182 record.length = fragment_length; // Ensure length matches what was deserialized
183
184 return make_result(std::move(record));
185}

275Result<size_t> DTLSCiphertext::serialize(memory::Buffer& buffer) const {
276 size_t total_size = HEADER_SIZE + encrypted_record.size();
277 if (has_connection_id) {
278 total_size += 1 + connection_id_length; // Length byte + CID
279 }
280
281 // Ensure buffer has enough space
282 auto resize_result = buffer.resize(total_size);
283 if (!resize_result.is_success()) {
284 return make_error<size_t>(DTLSError::INTERNAL_ERROR, "Failed to resize buffer");
285 }
286
287 uint8_t* data = reinterpret_cast<uint8_t*>(buffer.mutable_data());
288 size_t offset = 0;
289
290 // Serialize header fields (same as DTLSPlaintext)
291 data[offset++] = static_cast<uint8_t>(type);
292
293 // Protocol version (big-endian uint16_t)
294 uint16_t version_raw = static_cast<uint16_t>(version);
295 data[offset++] = (version_raw >> 8) & 0xFF;
296 data[offset++] = version_raw & 0xFF;
297
298 // Epoch (big-endian uint16_t)
299 data[offset++] = (epoch >> 8) & 0xFF;
300 data[offset++] = epoch & 0xFF;
301
302 // Encrypted sequence number (48-bit, big-endian)
303 auto seq_result = encrypted_sequence_number.serialize_to_buffer(&data[offset]);
304 if (!seq_result.is_success()) {
305 return make_error<size_t>(seq_result.error(), "Failed to serialize encrypted sequence number");
306 }
307 offset += SequenceNumber48::SERIALIZED_SIZE;
308
309 // Length (big-endian uint16_t) - includes CID if present
310 uint16_t serialized_length = length;
311 if (has_connection_id) {
312 serialized_length += 1 + connection_id_length;
313 }
314 data[offset++] = (serialized_length >> 8) & 0xFF;
315 data[offset++] = serialized_length & 0xFF;
316
317 // Connection ID if present
318 if (has_connection_id) {
319 data[offset++] = connection_id_length;
320 if (connection_id_length > 0) {
321 std::memcpy(&data[offset], connection_id.data(), connection_id_length);
322 offset += connection_id_length;
323 }
324 }
325
326 // Encrypted record data
327 if (encrypted_record.size() > 0) {
328 std::memcpy(&data[offset], encrypted_record.data(), encrypted_record.size());
329 offset += encrypted_record.size();
330 }

332 return make_result(offset);

335Result<DTLSCiphertext> DTLSCiphertext::deserialize(const memory::Buffer& buffer, size_t offset) {
336 if (buffer.size() < offset + HEADER_SIZE) {
337 return make_error<DTLSCiphertext>(DTLSError::DECODE_ERROR, "Buffer too small for DTLS header");
338 }
339
340 const uint8_t* data = reinterpret_cast<const uint8_t*>(buffer.data());
341 size_t pos = offset;
342
343 // Deserialize header fields (same as DTLSPlaintext)
344 ContentType content_type = static_cast<ContentType>(data[pos++]);
345
346 // Protocol version (big-endian uint16_t)
347 uint16_t version_raw = (static_cast<uint16_t>(data[pos]) << 8) | data[pos + 1];
348 ProtocolVersion proto_version = static_cast<ProtocolVersion>(version_raw);
349 pos += 2;
350
351 // Epoch (big-endian uint16_t)
352 uint16_t epoch_val = (static_cast<uint16_t>(data[pos]) << 8) | data[pos + 1];
353 pos += 2;
354
355 // Encrypted sequence number (48-bit, big-endian)
356 auto seq_result = SequenceNumber48::deserialize_from_buffer(&data[pos]);
357 if (!seq_result.is_success()) {
358 return make_error<DTLSCiphertext>(seq_result.error(), "Failed to deserialize encrypted sequence number");
359 }
360 SequenceNumber48 encrypted_seq_num = seq_result.value();
361 pos += SequenceNumber48::SERIALIZED_SIZE;
362
363 // Length (big-endian uint16_t)
364 uint16_t total_length = (static_cast<uint16_t>(data[pos]) << 8) | data[pos + 1];
365 pos += 2;
366
367 // Validate total buffer size
368 if (buffer.size() < pos + total_length) {
369 return make_error<DTLSCiphertext>(DTLSError::DECODE_ERROR, "Buffer too small for encrypted record");
370 }
371
372 DTLSCiphertext record;
373 record.type = content_type;
374 record.version = proto_version;
375 record.epoch = epoch_val;
376 record.encrypted_sequence_number = encrypted_seq_num;
377
378 // Check if Connection ID is present (heuristic: if type is not application_data for encrypted record)
379 // In practice, we'd need additional context to determine CID presence
380 // For now, we'll assume no CID unless we can detect it
381
382 size_t encrypted_data_length = total_length;
383
384 // Extract encrypted record
385 if (encrypted_data_length > 0) {
386 record.encrypted_record = memory::Buffer(encrypted_data_length);
387 auto resize_result = record.encrypted_record.resize(encrypted_data_length);
388 if (!resize_result.is_success()) {
389 return make_error<DTLSCiphertext>(DTLSError::INTERNAL_ERROR, "Failed to allocate encrypted record buffer");
390 }
391 std::memcpy(record.encrypted_record.mutable_data(), &data[pos], encrypted_data_length);
392 }
393
394 record.length = static_cast<uint16_t>(record.encrypted_record.size());
395
396 return make_result(std::move(record));
397}

405 // Check protocol version (should be DTLS v1.3)
