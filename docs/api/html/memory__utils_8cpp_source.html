32void MemoryStatsCollector::record_allocation(size_t size, const std::string& location) {
33 stats_.total_allocations.fetch_add(1, std::memory_order_relaxed);
34 stats_.total_bytes_allocated.fetch_add(size, std::memory_order_relaxed);
35
36 size_t current_allocs = stats_.current_allocations.fetch_add(1, std::memory_order_relaxed) + 1;
37 size_t current_bytes = stats_.current_bytes_allocated.fetch_add(size, std::memory_order_relaxed) + size;
38
39 // Update peak allocations
40 size_t current_peak_allocs = stats_.peak_allocations.load();
41 while (current_allocs > current_peak_allocs &&
42 !stats_.peak_allocations.compare_exchange_weak(current_peak_allocs, current_allocs)) {
43 // Retry if another thread updated peak
44 }
45
46 // Update peak bytes
47 size_t current_peak_bytes = stats_.peak_bytes_allocated.load();
48 while (current_bytes > current_peak_bytes &&
49 !stats_.peak_bytes_allocated.compare_exchange_weak(current_peak_bytes, current_bytes)) {
50 // Retry if another thread updated peak
51 }
52
53 // Track individual allocations if enabled
54 if (tracking_enabled_.load()) {
55 std::lock_guard<std::mutex> lock(tracking_mutex_);
56
57 AllocationInfo info;
58 info.size = size;
59 info.allocation_time = std::chrono::steady_clock::now();
60 info.location = location;
61 info.thread_id = std::this_thread::get_id();
62
63 // Note: In a real implementation, we'd need the actual pointer
64 // For now, we use a placeholder
65 active_allocations_[reinterpret_cast<void*>(current_allocs)] = info;
66 }
67}

177 if (!a || !b) return false;

179 const uint8_t* ptr_a = static_cast<const uint8_t*>(a);
180 const uint8_t* ptr_b = static_cast<const uint8_t*>(b);

406void MemoryDebugger::add_buffer_checksum(const ZeroCopyBuffer& buffer) {
407 if (!buffer.data() || buffer.size() == 0) {
408 return;
409 }
410
411 // Simple checksum calculation
412 uint32_t checksum = 0;
413 const uint8_t* data = reinterpret_cast<const uint8_t*>(buffer.data());
414
415 for (size_t i = 0; i < buffer.size(); ++i) {
416 checksum = (checksum << 1) ^ data[i];
417 }
418
419 std::lock_guard<std::mutex> lock(checksums_mutex_);
420 buffer_checksums_[buffer.data()] = checksum;
421}
422
423bool MemoryDebugger::verify_buffer_checksum(const ZeroCopyBuffer& buffer) const {
424 if (!buffer.data() || buffer.size() == 0) {
425 return true; // Empty buffers are always valid
426 }
427
428 std::lock_guard<std::mutex> lock(checksums_mutex_);
429 auto it = buffer_checksums_.find(buffer.data());
430 if (it == buffer_checksums_.end()) {
431 return false; // No checksum recorded
432 }
433
434 // Recalculate checksum
435 uint32_t checksum = 0;
436 const uint8_t* data = reinterpret_cast<const uint8_t*>(buffer.data());
437
438 for (size_t i = 0; i < buffer.size(); ++i) {
439 checksum = (checksum << 1) ^ data[i];
440 }
441
442 return checksum == it->second;
443}

574// Memory health check
575Result<MemoryHealthReport> perform_memory_health_check() {
576 MemoryHealthReport report;
577 report.check_time = std::chrono::steady_clock::now();
578
579 auto stats = MemoryStatsCollector::instance().get_statistics();
580
581 report.total_memory_usage = stats.current_bytes_allocated;
582 report.active_allocations = stats.current_allocations;
583 report.memory_leaks = MemoryStatsCollector::instance().get_leak_count();
584
585 // Calculate fragmentation ratio (simplified)
586 auto pool_stats = GlobalPoolManager::instance().get_all_statistics();
587 size_t total_pool_memory = 0;
588 size_t available_pool_memory = 0;
589
590 for (const auto& pool_stat : pool_stats) {
591 total_pool_memory += pool_stat.total_buffers * pool_stat.buffer_size;
592 available_pool_memory += pool_stat.available_buffers * pool_stat.buffer_size;
593 }
594
595 if (total_pool_memory > 0) {
596 report.fragmentation_ratio = 1.0 - (static_cast<double>(available_pool_memory) / total_pool_memory);
597 }
598
599 // Check for issues
600 auto config = get_memory_configuration();
601
602 if (report.total_memory_usage > config.warning_threshold) {
603 report.issues.push_back("Memory usage exceeds warning threshold");
604 }
605
606 if (report.memory_leaks > 0) {
607 report.issues.push_back("Memory leaks detected: " + std::to_string(report.memory_leaks));
608 report.overall_healthy = false;
609 }
610
611 if (report.fragmentation_ratio > 0.5) {
612 report.issues.push_back("High memory fragmentation detected");
613 }
614
615 if (stats.allocation_failures > 0) {
616 report.issues.push_back("Allocation failures detected: " + std::to_string(stats.allocation_failures));
617 report.overall_healthy = false;
618 }
619
620 return Result<MemoryHealthReport>(std::move(report));
621}
