37Result<size_t> RecordHeader::serialize(memory::Buffer& buffer) const {
38 if (buffer.capacity() < SERIALIZED_SIZE) {
39 return Result<size_t>(DTLSError::INSUFFICIENT_BUFFER_SIZE);
40 }
41
42 if (!is_valid()) {
43 return Result<size_t>(DTLSError::INVALID_RECORD_HEADER);
44 }
45
46 // Resize buffer to ensure we have space
47 auto resize_result = buffer.resize(SERIALIZED_SIZE);
48 if (!resize_result.is_success()) {
49 return Result<size_t>(resize_result.error());
50 }
51
52 std::byte* data = buffer.mutable_data();
53 size_t offset = 0;
54
55 // Content type (1 byte)
56 data[offset++] = static_cast<std::byte>(static_cast<uint8_t>(content_type));
57
58 // Protocol version (2 bytes, network byte order)
59 uint16_t version_net = htons(static_cast<uint16_t>(version));
60 copy_to_byte_buffer(data + offset, &version_net, 2);
61 offset += 2;
62
63 // Epoch (2 bytes, network byte order)
64 uint16_t epoch_net = htons(epoch);
65 copy_to_byte_buffer(data + offset, &epoch_net, 2);
66 offset += 2;
67
68 // Sequence number (6 bytes, network byte order - only lower 48 bits)
69 uint64_t seq_net = htobe64(sequence_number);
70 copy_to_byte_buffer(data + offset, reinterpret_cast<uint8_t*>(&seq_net) + 2, 6);
71 offset += 6;
72
73 // Length (2 bytes, network byte order)
74 uint16_t length_net = htons(length);
75 copy_to_byte_buffer(data + offset, &length_net, 2);
76 offset += 2;
77
78 return Result<size_t>(SERIALIZED_SIZE);
79}

81Result<RecordHeader> RecordHeader::deserialize(const memory::Buffer& buffer, size_t offset) {
82 if (buffer.size() < offset + SERIALIZED_SIZE) {
83 return Result<RecordHeader>(DTLSError::INSUFFICIENT_BUFFER_SIZE);
84 }
85
86 const std::byte* data = buffer.data() + offset;
87 RecordHeader header;
88 size_t pos = 0;
89
90 // Content type (1 byte)
91 header.content_type = static_cast<ContentType>(static_cast<uint8_t>(data[pos++]));
92
93 // Protocol version (2 bytes, network byte order)
94 uint16_t version_net;
95 copy_from_byte_buffer(&version_net, data + pos, 2);
96 header.version = static_cast<ProtocolVersion>(ntohs(version_net));
97 pos += 2;
98
99 // Epoch (2 bytes, network byte order)
100 uint16_t epoch_net;
101 copy_from_byte_buffer(&epoch_net, data + pos, 2);
102 header.epoch = ntohs(epoch_net);
103 pos += 2;
104
105 // Sequence number (6 bytes, network byte order)
106 uint64_t seq_net = 0;
107 copy_from_byte_buffer(reinterpret_cast<uint8_t*>(&seq_net) + 2, data + pos, 6);
108 header.sequence_number = be64toh(seq_net);
109 pos += 6;
110
111 // Length (2 bytes, network byte order)
112 uint16_t length_net;
113 copy_from_byte_buffer(&length_net, data + pos, 2);
114 header.length = ntohs(length_net);
115 pos += 2;
116
117 if (!header.is_valid()) {
118 return Result<RecordHeader>(DTLSError::INVALID_RECORD_HEADER);
119 }
120
121 return Result<RecordHeader>(std::move(header));
122}

347Result<CiphertextRecord> CiphertextRecord::deserialize(const memory::Buffer& buffer, size_t offset) {
348 if (buffer.size() < offset + RecordHeader::SERIALIZED_SIZE) {
349 return Result<CiphertextRecord>(DTLSError::INSUFFICIENT_BUFFER_SIZE);
350 }
351
352 // Deserialize header first
353 auto header_result = RecordHeader::deserialize(buffer, offset);
354 if (!header_result.is_success()) {
355 return Result<CiphertextRecord>(header_result.error());
356 }
357
358 RecordHeader header = header_result.value();
359 offset += RecordHeader::SERIALIZED_SIZE;
360
361 if (buffer.size() < offset + header.length) {
362 return Result<CiphertextRecord>(DTLSError::INSUFFICIENT_BUFFER_SIZE);
363 }
364
365 CiphertextRecord record;
366 record.header_ = header;
367
368 // For now, assume no connection ID and that the payload includes auth tag
369 // In a real implementation, we'd need additional context to determine CID presence
370 size_t remaining_length = header.length;
371
372 // For AEAD ciphers, typically last 16 bytes are auth tag
373 constexpr size_t AUTH_TAG_SIZE = 16;
374 if (remaining_length >= AUTH_TAG_SIZE) {
375 size_t encrypted_payload_size = remaining_length - AUTH_TAG_SIZE;
376
377 // Extract encrypted payload
378 if (encrypted_payload_size > 0) {
379 record.encrypted_payload_ = memory::Buffer(encrypted_payload_size);
380 auto resize_result = record.encrypted_payload_.resize(encrypted_payload_size);
381 if (!resize_result.is_success()) {
382 return Result<CiphertextRecord>(DTLSError::INTERNAL_ERROR);
383 }
384 copy_from_byte_buffer(record.encrypted_payload_.mutable_data(),
385 buffer.data() + offset, encrypted_payload_size);
386 offset += encrypted_payload_size;
387 }
388
389 // Extract authentication tag
390 record.authentication_tag_ = memory::Buffer(AUTH_TAG_SIZE);
391 auto resize_result = record.authentication_tag_.resize(AUTH_TAG_SIZE);
392 if (!resize_result.is_success()) {
393 return Result<CiphertextRecord>(DTLSError::INTERNAL_ERROR);
394 }
395 copy_from_byte_buffer(record.authentication_tag_.mutable_data(),
396 buffer.data() + offset, AUTH_TAG_SIZE);
397 } else {
398 // No auth tag, entire payload is encrypted data
399 record.encrypted_payload_ = memory::Buffer(remaining_length);
400 auto resize_result = record.encrypted_payload_.resize(remaining_length);
401 if (!resize_result.is_success()) {
402 return Result<CiphertextRecord>(DTLSError::INTERNAL_ERROR);
403 }
404 copy_from_byte_buffer(record.encrypted_payload_.mutable_data(),
405 buffer.data() + offset, remaining_length);
406 }
407
408 return Result<CiphertextRecord>(std::move(record));
409}
