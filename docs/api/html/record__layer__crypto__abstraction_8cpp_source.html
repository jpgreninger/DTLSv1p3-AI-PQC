150Result<DTLSCiphertext> RecordLayerWithCryptoAbstraction::protect_record(const DTLSPlaintext& plaintext) {
151 std::lock_guard<std::mutex> lock(state_mutex_);
152
153 if (!initialized_) {
154 return Result<DTLSCiphertext>(DTLSError::NOT_INITIALIZED);
155 }
156
157 auto* ops = crypto_operations();
158 if (!ops) {
159 return Result<DTLSCiphertext>(DTLSError::NOT_INITIALIZED);
160 }
161
162 // Validate plaintext
163 if (plaintext.fragment.size() > DTLSPlaintext::MAX_FRAGMENT_LENGTH) {
164 return Result<DTLSCiphertext>(DTLSError::INVALID_PARAMETER);
165 }
166
167 // Get current epoch keys
168 if (current_keys_.client_write_key.empty() || current_keys_.server_write_key.empty()) {
169 return Result<DTLSCiphertext>(DTLSError::STATE_MACHINE_ERROR);
170 }
171
172 // Determine which keys to use based on connection role
173 // For simplicity, we'll use client keys for encryption and server keys for decryption
174 // In a real implementation, this would be determined by the connection role
175 const auto& write_key = current_keys_.client_write_key;
176 const auto& write_iv = current_keys_.client_write_iv;
177 const auto& seq_num_key = current_keys_.client_sequence_number_key;
178
179 if (write_key.empty() || write_iv.empty()) {
180 return Result<DTLSCiphertext>(DTLSError::STATE_MACHINE_ERROR);
181 }
182
183 // Encrypt sequence number (RFC 9147 Section 4.2.3)
184 SequenceNumber48 encrypted_seq_num;
185 if (!seq_num_key.empty() && seq_num_key.size() >= 16) {
186 // Use AES-128-ECB to encrypt sequence number
187 crypto::AEADParams seq_encrypt_params;
188 seq_encrypt_params.cipher = AEADCipher::AES_128_GCM; // Use GCM for consistency
189 seq_encrypt_params.key = seq_num_key;
190 seq_encrypt_params.nonce = std::vector<uint8_t>(12, 0); // Zero nonce for sequence number encryption
191 seq_encrypt_params.additional_data = std::vector<uint8_t>(); // Empty AAD
192
193 // Convert sequence number to 6 bytes
194 std::vector<uint8_t> seq_bytes(8, 0);
195 auto seq_num_value = static_cast<uint64_t>(plaintext.sequence_number);
196 for (int i = 5; i >= 0; --i) {
197 seq_bytes[7-i] = static_cast<uint8_t>((seq_num_value >> (i * 8)) & 0xFF);
198 }
199
200 auto encrypt_result = ops->aead_encrypt(
201 std::vector<uint8_t>(seq_bytes.begin(), seq_bytes.begin() + 6),
202 seq_encrypt_params.key,
203 seq_encrypt_params.nonce,
204 seq_encrypt_params.additional_data,
205 seq_encrypt_params.cipher);
206 if (encrypt_result.is_success()) {
207 const auto& encrypted_output = encrypt_result.value();
208 if (encrypted_output.ciphertext.size() >= 6) {
209 std::array<uint8_t, 6> seq_array = {encrypted_output.ciphertext[0], encrypted_output.ciphertext[1], encrypted_output.ciphertext[2],
210 encrypted_output.ciphertext[3], encrypted_output.ciphertext[4], encrypted_output.ciphertext[5]};
211 auto seq_result = SequenceNumber48::deserialize_from_buffer(seq_array.data());
212 if (seq_result.is_success()) {
213 encrypted_seq_num = seq_result.value();
214 } else {
215 encrypted_seq_num = plaintext.sequence_number; // Fallback
216 }
217 } else {
218 encrypted_seq_num = plaintext.sequence_number; // Fallback
219 }
220 } else {
221 encrypted_seq_num = plaintext.sequence_number; // Fallback
222 }
223 } else {
224 encrypted_seq_num = plaintext.sequence_number; // Fallback
225 }
226
227 // Prepare AEAD encryption
228 crypto::AEADParams aead_params;
229 aead_params.cipher = AEADCipher::AES_128_GCM; // Default cipher suite
230 aead_params.key = write_key;
231
232 // Construct nonce per RFC 9147 Section 5.3
233 // Nonce = write_iv XOR seq_num (padded to IV length)
234 aead_params.nonce = write_iv;
235 auto seq_for_nonce = static_cast<uint64_t>(plaintext.sequence_number);
236 size_t nonce_size = std::min(aead_params.nonce.size(), static_cast<size_t>(8));
237 for (size_t i = 0; i < nonce_size; ++i) {
238 size_t iv_index = aead_params.nonce.size() - 1 - i;
239 aead_params.nonce[iv_index] ^= static_cast<uint8_t>((seq_for_nonce >> (i * 8)) & 0xFF);
240 }
241
242 // Construct Additional Authenticated Data (AAD)
243 // AAD = type || version || epoch || encrypted_sequence_number || length
244 std::vector<uint8_t> aad;
245 aad.reserve(13);
246 aad.push_back(static_cast<uint8_t>(plaintext.type));
247 aad.push_back(static_cast<uint8_t>(static_cast<uint16_t>(plaintext.version) >> 8));
248 aad.push_back(static_cast<uint8_t>(static_cast<uint16_t>(plaintext.version) & 0xFF));
249 aad.push_back(static_cast<uint8_t>(plaintext.epoch >> 8));
250 aad.push_back(static_cast<uint8_t>(plaintext.epoch & 0xFF));
251
252 // Serialize encrypted sequence number to bytes
253 std::array<uint8_t, 6> encrypted_seq_bytes;
254 auto serialize_result = encrypted_seq_num.serialize_to_buffer(encrypted_seq_bytes.data());
255 if (serialize_result.is_success()) {
256 aad.insert(aad.end(), encrypted_seq_bytes.begin(), encrypted_seq_bytes.end());
257 }
258
259 uint16_t content_length = static_cast<uint16_t>(plaintext.fragment.size());
260 aad.push_back(static_cast<uint8_t>(content_length >> 8));
261 aad.push_back(static_cast<uint8_t>(content_length & 0xFF));
262
263 aead_params.additional_data = std::move(aad);
264
265 // Encrypt the fragment
266 std::vector<uint8_t> plaintext_data(plaintext.fragment.size());
267 std::memcpy(plaintext_data.data(), plaintext.fragment.data(), plaintext.fragment.size());
268
269 auto encrypt_result = ops->aead_encrypt(
270 plaintext_data,
271 aead_params.key,
272 aead_params.nonce,
273 aead_params.additional_data,
274 aead_params.cipher);
275 if (encrypt_result.is_error()) {
276 return Result<DTLSCiphertext>(encrypt_result.error());
277 }
278
279 // Construct DTLSCiphertext
280 DTLSCiphertext ciphertext;
281 ciphertext.type = ContentType::APPLICATION_DATA; // Always application_data for encrypted records
282 ciphertext.version = plaintext.version;
283 ciphertext.epoch = plaintext.epoch;
284 ciphertext.encrypted_sequence_number = encrypted_seq_num;
285
286 // Combine ciphertext and tag
287 const auto& encrypted_output = encrypt_result.value();
288 std::vector<uint8_t> combined_data;
289 combined_data.reserve(encrypted_output.ciphertext.size() + encrypted_output.tag.size());
290 combined_data.insert(combined_data.end(), encrypted_output.ciphertext.begin(), encrypted_output.ciphertext.end());
291 combined_data.insert(combined_data.end(), encrypted_output.tag.begin(), encrypted_output.tag.end());
292
293 ciphertext.encrypted_record = memory::Buffer(reinterpret_cast<const std::byte*>(combined_data.data()), combined_data.size());
294 ciphertext.length = static_cast<uint16_t>(ciphertext.encrypted_record.size());
295
296 // Add connection ID if enabled
297 if (!local_connection_id_.empty()) {
298 ciphertext.has_connection_id = true;
299 ciphertext.connection_id_length = static_cast<uint8_t>(
300 std::min(local_connection_id_.size(), static_cast<size_t>(DTLSCiphertext::MAX_CONNECTION_ID_LENGTH)));
301 std::copy_n(local_connection_id_.begin(), ciphertext.connection_id_length,
302 ciphertext.connection_id.begin());
303 } else {
304 ciphertext.has_connection_id = false;
305 ciphertext.connection_id_length = 0;
306 }
307
308 return Result<DTLSCiphertext>(std::move(ciphertext));
309}

311Result<DTLSPlaintext> RecordLayerWithCryptoAbstraction::unprotect_record(const DTLSCiphertext& ciphertext) {
312 std::lock_guard<std::mutex> lock(state_mutex_);
313
314 if (!initialized_) {
315 return Result<DTLSPlaintext>(DTLSError::NOT_INITIALIZED);
316 }
317
318 auto* ops = crypto_operations();
319 if (!ops) {
320 return Result<DTLSPlaintext>(DTLSError::NOT_INITIALIZED);
321 }
322
323 // Validate ciphertext
324 if (ciphertext.encrypted_record.size() == 0 ||
325 ciphertext.encrypted_record.size() > DTLSCiphertext::MAX_ENCRYPTED_RECORD_LENGTH) {
326 return Result<DTLSPlaintext>(DTLSError::INVALID_PARAMETER);
327 }
328
329 // Get current epoch keys
330 if (current_keys_.client_write_key.empty() || current_keys_.server_write_key.empty()) {
331 return Result<DTLSPlaintext>(DTLSError::STATE_MACHINE_ERROR);
332 }
333
334 // Determine which keys to use based on connection role (opposite of protect_record)
335 // For simplicity, we'll use server keys for decryption
336 const auto& read_key = current_keys_.server_write_key;
337 const auto& read_iv = current_keys_.server_write_iv;
338 const auto& peer_seq_num_key = current_keys_.server_sequence_number_key;
339
340 if (read_key.empty() || read_iv.empty()) {
341 return Result<DTLSPlaintext>(DTLSError::STATE_MACHINE_ERROR);
342 }
343
344 // Decrypt sequence number (RFC 9147 Section 4.2.3)
345 SequenceNumber48 decrypted_seq_num;
346 if (!peer_seq_num_key.empty() && peer_seq_num_key.size() >= 16) {
347 // Use AES-128-ECB to decrypt sequence number
348 crypto::AEADParams seq_decrypt_params;
349 seq_decrypt_params.cipher = AEADCipher::AES_128_GCM; // Use GCM for consistency
350 seq_decrypt_params.key = peer_seq_num_key;
351 seq_decrypt_params.nonce = std::vector<uint8_t>(12, 0); // Zero nonce for sequence number decryption
352 seq_decrypt_params.additional_data = std::vector<uint8_t>(); // Empty AAD
353
354 // Convert encrypted sequence number to bytes
355 std::array<uint8_t, 6> encrypted_seq_bytes;
356 auto serialize_result = ciphertext.encrypted_sequence_number.serialize_to_buffer(encrypted_seq_bytes.data());
357 if (serialize_result.is_success()) {
358 std::vector<uint8_t> encrypted_data(encrypted_seq_bytes.begin(), encrypted_seq_bytes.end());
359
360 // For sequence number encryption, we need to "decrypt" by using the same encryption operation
361 // since we're using AES-ECB conceptually (though implemented with GCM)
362 auto decrypt_result = ops->aead_encrypt(
363 encrypted_data,
364 seq_decrypt_params.key,
365 seq_decrypt_params.nonce,
366 seq_decrypt_params.additional_data,
367 seq_decrypt_params.cipher);
368 if (decrypt_result.is_success()) {
369 const auto& decrypted_output = decrypt_result.value();
370 if (decrypted_output.ciphertext.size() >= 6) {
371 auto seq_result = SequenceNumber48::deserialize_from_buffer(decrypted_output.ciphertext.data());
372 if (seq_result.is_success()) {
373 decrypted_seq_num = seq_result.value();
374 } else {
375 decrypted_seq_num = ciphertext.encrypted_sequence_number; // Fallback
376 }
377 } else {
378 decrypted_seq_num = ciphertext.encrypted_sequence_number; // Fallback
379 }
380 } else {
381 decrypted_seq_num = ciphertext.encrypted_sequence_number; // Fallback
382 }
383 } else {
384 decrypted_seq_num = ciphertext.encrypted_sequence_number; // Fallback
385 }
386 } else {
387 decrypted_seq_num = ciphertext.encrypted_sequence_number; // Fallback
388 }
389
390 // Prepare AEAD decryption
391 crypto::AEADParams aead_params;
392 aead_params.cipher = AEADCipher::AES_128_GCM; // Default cipher suite
393 aead_params.key = read_key;
394
395 // Construct nonce per RFC 9147 Section 5.3
396 // Nonce = read_iv XOR seq_num (padded to IV length)
397 aead_params.nonce = read_iv;
398 auto seq_for_nonce = static_cast<uint64_t>(decrypted_seq_num);
399 size_t nonce_size = std::min(aead_params.nonce.size(), static_cast<size_t>(8));
400 for (size_t i = 0; i < nonce_size; ++i) {
401 size_t iv_index = aead_params.nonce.size() - 1 - i;
402 aead_params.nonce[iv_index] ^= static_cast<uint8_t>((seq_for_nonce >> (i * 8)) & 0xFF);
403 }
404
405 // Construct Additional Authenticated Data (AAD)
406 // AAD = type || version || epoch || encrypted_sequence_number || length
407 std::vector<uint8_t> aad;
408 aad.reserve(13);
409 aad.push_back(static_cast<uint8_t>(ciphertext.type));
410 aad.push_back(static_cast<uint8_t>(static_cast<uint16_t>(ciphertext.version) >> 8));
411 aad.push_back(static_cast<uint8_t>(static_cast<uint16_t>(ciphertext.version) & 0xFF));
412 aad.push_back(static_cast<uint8_t>(ciphertext.epoch >> 8));
413 aad.push_back(static_cast<uint8_t>(ciphertext.epoch & 0xFF));
414
415 // Serialize encrypted sequence number to bytes
416 std::array<uint8_t, 6> encrypted_seq_bytes;
417 auto serialize_result = ciphertext.encrypted_sequence_number.serialize_to_buffer(encrypted_seq_bytes.data());
418 if (serialize_result.is_success()) {
419 aad.insert(aad.end(), encrypted_seq_bytes.begin(), encrypted_seq_bytes.end());
420 }
421
422 aad.push_back(static_cast<uint8_t>(ciphertext.length >> 8));
423 aad.push_back(static_cast<uint8_t>(ciphertext.length & 0xFF));
424
425 aead_params.additional_data = std::move(aad);
426
427 // Decrypt the record - separate ciphertext and tag
428 const auto& encrypted_data = ciphertext.encrypted_record;
429 if (encrypted_data.size() < 16) { // Minimum tag size for GCM
430 return Result<DTLSPlaintext>(DTLSError::DECRYPT_ERROR);
431 }
432
433 // Assume last 16 bytes are the tag for GCM
434 size_t ciphertext_size = encrypted_data.size() - 16;
435 std::vector<uint8_t> ciphertext_only(ciphertext_size);
436 std::vector<uint8_t> tag(16);
437
438 std::memcpy(ciphertext_only.data(), encrypted_data.data(), ciphertext_size);
439 std::memcpy(tag.data(), encrypted_data.data() + ciphertext_size, 16);
440
441 auto decrypt_result = ops->aead_decrypt(
442 ciphertext_only,
443 tag,
444 aead_params.key,
445 aead_params.nonce,
446 aead_params.additional_data,
447 aead_params.cipher);
448 if (decrypt_result.is_error()) {
449 return Result<DTLSPlaintext>(decrypt_result.error());
450 }
451
452 // Extract the inner content type and actual content
453 const auto& decrypted_data = decrypt_result.value();
454 if (decrypted_data.empty()) {
455 return Result<DTLSPlaintext>(DTLSError::DECRYPT_ERROR);
456 }
457
458 // Per RFC 9147, the last byte of decrypted content is the inner content type
459 ContentType inner_content_type = static_cast<ContentType>(decrypted_data.back());
460
461 // Remove padding zeros and content type byte
462 size_t content_end = decrypted_data.size() - 1;
463 while (content_end > 0 && decrypted_data[content_end - 1] == 0) {
464 content_end--;
465 }
466
467 if (content_end == 0) {
468 return Result<DTLSPlaintext>(DTLSError::DECRYPT_ERROR);
469 }
470
471 // Create the plaintext record
472 DTLSPlaintext plaintext;
473 plaintext.type = inner_content_type;
474 plaintext.version = ciphertext.version;
475 plaintext.epoch = ciphertext.epoch;
476 plaintext.sequence_number = decrypted_seq_num;
477 std::vector<uint8_t> fragment_data(decrypted_data.begin(), decrypted_data.begin() + content_end);
478 plaintext.fragment = memory::Buffer(reinterpret_cast<const std::byte*>(fragment_data.data()), fragment_data.size());
479 plaintext.length = static_cast<uint16_t>(plaintext.fragment.size());
480
481 return Result<DTLSPlaintext>(std::move(plaintext));
482}

551Result<void> RecordLayerWithCryptoAbstraction::switch_crypto_operations(
552 std::unique_ptr<crypto::ICryptoOperations> new_crypto_ops) {
553 std::lock_guard<std::mutex> lock(state_mutex_);
