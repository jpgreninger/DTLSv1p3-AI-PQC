553 const std::vector<uint8_t>& secret,
554 const std::string& label,
555 const std::vector<uint8_t>& context,
556 size_t length,
557 HashAlgorithm hash_algo) {
558
559 KeyDerivationParams params;
560 params.secret = secret;
561 params.output_length = length;
562 params.hash_algorithm = hash_algo;
563
564 // Create HKDF-Expand-Label info structure
565 std::vector<uint8_t> info;
566
567 // Length (2 bytes, big-endian)
568 info.push_back(static_cast<uint8_t>((length >> 8) & 0xFF));
569 info.push_back(static_cast<uint8_t>(length & 0xFF));
570
571 // Label length + "tls13 " + label

573 info.push_back(static_cast<uint8_t>(full_label.length()));
574 info.insert(info.end(), full_label.begin(), full_label.end());
575
576 // Context length + context
577 info.push_back(static_cast<uint8_t>(context.size()));
578 info.insert(info.end(), context.begin(), context.end());
579
580 params.info = std::move(info);
581 return params;
582}

1138Result<std::unique_ptr<ICryptoOperations>>
1139CryptoOperationsFactory::create_cached_operations(const std::string& provider_name) {
1140 if (!caching_enabled_) {
1141 return create_operations(provider_name);
1142 }
1143
1144 ProviderSelection criteria;
1145 criteria.preferred_provider = provider_name;
1146 std::string cache_key = generate_cache_key(provider_name, criteria);
1147
1148 {
1149 std::shared_lock<std::shared_mutex> lock(factory_mutex_);
1150 auto it = operation_cache_.find(cache_key);
1151 if (it != operation_cache_.end()) {
1152 it->second.last_used = std::chrono::steady_clock::now();
1153 it->second.use_count++;
1154 cache_hit_count_++;
1155
1156 // Create a new instance based on the cached one
1157 return create_operations(provider_name);
1158 }
1159 }
1160
1161 cache_miss_count_++;
1162 auto result = create_operations(provider_name);
1163 if (result.is_success()) {
1164 std::unique_lock<std::shared_mutex> lock(factory_mutex_);
1165 if (operation_cache_.size() >= cache_size_limit_) {
1166 cleanup_expired_cache_entries();
1167 }
1168
1169 factory_stats_.cached_created++;
1170
1171 // Note: We don't actually cache the operations themselves due to
1172 // thread safety concerns. Instead, we cache the fact that this
1173 // configuration works and can be quickly recreated.
1174 CachedOperations cached;
1175 cached.created_time = std::chrono::steady_clock::now();
1176 cached.last_used = cached.created_time;
1177 cached.use_count = 1;
1178 operation_cache_[cache_key] = std::move(cached);
1179 }
1180
1181 return result;
1182}

1285
1286Result<std::string> ProviderCapabilityMatcher::find_best_provider(
1287 const ProviderSelection& criteria) {
1288
1289 auto& factory = ProviderFactory::instance();
1290 auto available_providers = factory.available_providers();
1291
1292 if (available_providers.empty()) {
1293 return Result<std::string>(DTLSError::CRYPTO_PROVIDER_NOT_AVAILABLE);
1294 }
1295
1296 std::string best_provider;
1297 double best_score = -1.0;
1298
1299 for (const auto& provider_name : available_providers) {
1300 auto capabilities_result = factory.get_capabilities(provider_name);
1301 if (!capabilities_result.is_success()) {
1302 continue;
1303 }
1304
1305 auto capabilities = capabilities_result.value();
1306 if (!meets_basic_requirements(capabilities, criteria)) {
1307 continue;
1308 }
1309
1310 double score = calculate_compatibility_score(capabilities, criteria);
1311 if (score > best_score) {
1312 best_score = score;
1313 best_provider = provider_name;
1314 }
1315 }
1316
1317 if (best_provider.empty()) {
1318 return Result<std::string>(DTLSError::CRYPTO_PROVIDER_NOT_AVAILABLE);
1319 }
1320
1321 return Result<std::string>(best_provider);
1322}
1323
1324double ProviderCapabilityMatcher::calculate_compatibility_score(
1325 const ProviderCapabilities& capabilities,
1326 const ProviderSelection& criteria) {
1327
1328 if (!meets_basic_requirements(capabilities, criteria)) {
1329 return 0.0;
1330 }
1331
1332 double score = 0.5; // Base score for meeting basic requirements
1333
1334 // Add feature score
1335 score += 0.3 * calculate_feature_score(capabilities, criteria);
1336
1337 // Add performance score

1343bool ProviderCapabilityMatcher::meets_basic_requirements(
1344 const ProviderCapabilities& capabilities,
1345 const ProviderSelection& criteria) {
1346
1347 // Check hardware acceleration requirement
1348 if (criteria.require_hardware_acceleration && !capabilities.hardware_acceleration) {

1350 }
1351
1352 // Check FIPS compliance requirement
1353 if (criteria.require_fips_compliance && !capabilities.fips_mode) {
1354 return false;
1355 }
1356
1357 // Check cipher suite support
1358 for (auto required_suite : criteria.required_cipher_suites) {
1359 auto it = std::find(capabilities.supported_cipher_suites.begin(),
1360 capabilities.supported_cipher_suites.end(), required_suite);
1361 if (it == capabilities.supported_cipher_suites.end()) {
1362 return false;
1363 }
1364 }
1365
1366 // Check named group support
1367 for (auto required_group : criteria.required_groups) {
1368 auto it = std::find(capabilities.supported_groups.begin(),
1369 capabilities.supported_groups.end(), required_group);
1370 if (it == capabilities.supported_groups.end()) {
1371 return false;
1372 }
1373 }
1374
1375 // Check signature scheme support
1376 for (auto required_signature : criteria.required_signatures) {
1377 auto it = std::find(capabilities.supported_signatures.begin(),
1378 capabilities.supported_signatures.end(), required_signature);
1379 if (it == capabilities.supported_signatures.end()) {
1380 return false;
1381 }
1382 }
1383
1384 return true;

1440Result<ProviderCompatibilityResult> ProviderCapabilityMatcher::check_compatibility(
1441 const std::string& provider_name,
1442 const ProviderSelection& criteria) {
1443
1444 auto& factory = ProviderFactory::instance();
1445 auto capabilities_result = factory.get_capabilities(provider_name);
1446 if (!capabilities_result.is_success()) {
1447 return Result<ProviderCompatibilityResult>(capabilities_result.error());
1448 }
1449
1450 auto capabilities = capabilities_result.value();
1451 ProviderCompatibilityResult result;
1452
1453 result.is_compatible = meets_basic_requirements(capabilities, criteria);
1454 result.compatibility_score = calculate_compatibility_score(capabilities, criteria);
1455
1456 // Check missing features
1457 for (auto suite : criteria.required_cipher_suites) {
1458 auto it = std::find(capabilities.supported_cipher_suites.begin(),
1459 capabilities.supported_cipher_suites.end(), suite);
1460 if (it == capabilities.supported_cipher_suites.end()) {
1461 result.missing_features.push_back("Cipher suite: " + std::to_string(static_cast<int>(suite)));
1462 }
1463 }
1464
1465 for (auto group : criteria.required_groups) {
1466 auto it = std::find(capabilities.supported_groups.begin(),
1467 capabilities.supported_groups.end(), group);
1468 if (it == capabilities.supported_groups.end()) {
1469 result.missing_features.push_back("Named group: " + std::to_string(static_cast<int>(group)));
1470 }
1471 }
1472
1473 for (auto sig : criteria.required_signatures) {
1474 auto it = std::find(capabilities.supported_signatures.begin(),
1475 capabilities.supported_signatures.end(), sig);
1476 if (it == capabilities.supported_signatures.end()) {
1477 result.missing_features.push_back("Signature scheme: " + std::to_string(static_cast<int>(sig)));
1478 }
1479 }
1480
1481 // Add warnings and recommendations based on compatibility
1482 if (result.compatibility_score < 0.7) {
1483 result.warnings.push_back("Low compatibility score");
1484 }
1485
1486 if (!capabilities.hardware_acceleration && criteria.require_hardware_acceleration) {
1487 result.warnings.push_back("Hardware acceleration not available");
1488 }
1489
1490 if (!capabilities.fips_mode && criteria.require_fips_compliance) {
1491 result.warnings.push_back("FIPS compliance not available");
1492 }
1493
1494 if (result.compatibility_score > 0.8) {
1495 result.recommendations.push_back("Excellent compatibility - recommended for use");
1496 } else if (result.compatibility_score > 0.6) {
1497 result.recommendations.push_back("Good compatibility - suitable for most use cases");
1498 } else {
1499 result.recommendations.push_back("Limited compatibility - consider alternative provider");
1500 }
1501
1502 return Result<ProviderCompatibilityResult>(std::move(result));
1503}

1506 const ProviderSelection& criteria) {
1507
1508 auto& factory = ProviderFactory::instance();
1509 auto available_providers = factory.available_providers();
1510
1511 std::vector<std::pair<std::string, double>> ranked_providers;
1512
1513 for (const auto& provider_name : available_providers) {
1514 auto capabilities_result = factory.get_capabilities(provider_name);

1517 }
1518

1521 ranked_providers.emplace_back(provider_name, score);
1522 }
1523

1525 std::sort(ranked_providers.begin(), ranked_providers.end(),
1526 [](const auto& a, const auto& b) {
1527 return a.second > b.second;
1528 });
1529
1530 return ranked_providers;

1549Result<ICryptoOperations*> AgnosticCryptoOperations::get_best_provider_for_operation(
1550 const std::string& operation_category,
1551 const std::function<bool(const ProviderCapabilities&)>& capability_checker) {
1552
1553 std::shared_lock<std::shared_mutex> lock(cache_mutex_);
1554
1555 ProviderSelection criteria = default_criteria_;
1556
1557 // Check for operation-specific criteria
1558 auto it = operation_criteria_.find(operation_category);
1559 if (it != operation_criteria_.end()) {
1560 criteria = it->second;
1561 }
1562
1563 // Find the best provider for this operation
1564 auto current_provider_name_result = ProviderCapabilityMatcher::find_best_provider(criteria);
1565 if (!current_provider_name_result.is_success()) {
1566 return Result<ICryptoOperations*>(current_provider_name_result.error());
1567 }
1568
1569 std::string provider_name = current_provider_name_result.value();
1570
1571 // Check if we already have this provider cached
1572 auto cache_it = provider_cache_.find(provider_name);
1573 if (cache_it != provider_cache_.end()) {
1574 provider_usage_count_[provider_name]++;
1575 return Result<ICryptoOperations*>(cache_it->second.get());
1576 }
1577
1578 // Create new provider (this requires upgrading to exclusive lock)
1579 lock.unlock();
1580 std::unique_lock<std::shared_mutex> exclusive_lock(cache_mutex_);
1581
1582 // Double-check after acquiring exclusive lock
1583 cache_it = provider_cache_.find(provider_name);
1584 if (cache_it != provider_cache_.end()) {
1585 provider_usage_count_[provider_name]++;
1586 return Result<ICryptoOperations*>(cache_it->second.get());
1587 }
1588
1589 // Create new crypto operations for this provider
1590 auto& factory = CryptoOperationsFactory::instance();
1591 auto ops_result = factory.create_operations(provider_name);
1592 if (!ops_result.is_success()) {
1593 return Result<ICryptoOperations*>(ops_result.error());
1594 }
1595
1596 auto ops = std::move(ops_result.value());
1597 ICryptoOperations* ops_ptr = ops.get();
1598 provider_cache_[provider_name] = std::move(ops);
1599 provider_usage_count_[provider_name] = 1;
1600
1601 return Result<ICryptoOperations*>(ops_ptr);
1602}

1626ProviderCapabilities AgnosticCryptoOperations::capabilities() const {
1627 std::shared_lock<std::shared_mutex> lock(cache_mutex_);
1628
1629 // Aggregate capabilities from all cached providers
1630 ProviderCapabilities aggregated;
1631 aggregated.provider_name = "Agnostic";
1632 aggregated.provider_version = "1.0";
1633
1634 std::unordered_set<CipherSuite> all_cipher_suites;
1635 std::unordered_set<NamedGroup> all_groups;

1637 std::unordered_set<HashAlgorithm> all_hashes;
1638

1640 bool any_fips_mode = false;
1641
1642 for (const auto& [provider_name, capabilities] : capability_cache_) {
1643 // Aggregate cipher suites
1644 for (auto suite : capabilities.supported_cipher_suites) {
1645 all_cipher_suites.insert(suite);
1646 }
1647
1648 // Aggregate named groups
1649 for (auto group : capabilities.supported_groups) {
1650 all_groups.insert(group);
1651 }
1652
1653 // Aggregate signature schemes
1654 for (auto signature : capabilities.supported_signatures) {
1655 all_signatures.insert(signature);
1656 }
1657
1658 // Aggregate hash algorithms
1659 for (auto hash : capabilities.supported_hashes) {
1660 all_hashes.insert(hash);
1661 }
1662
1663 // Aggregate boolean capabilities
1664 if (capabilities.hardware_acceleration) {
1665 any_hardware_accel = true;
1666 }
1667 if (capabilities.fips_mode) {
1668 any_fips_mode = true;
1669 }
1670 }
1671
1672 // Convert sets to vectors
1673 aggregated.supported_cipher_suites.assign(all_cipher_suites.begin(), all_cipher_suites.end());
1674 aggregated.supported_groups.assign(all_groups.begin(), all_groups.end());
1675 aggregated.supported_signatures.assign(all_signatures.begin(), all_signatures.end());
1676 aggregated.supported_hashes.assign(all_hashes.begin(), all_hashes.end());
1677
1678 aggregated.hardware_acceleration = any_hardware_accel;
1679 aggregated.fips_mode = any_fips_mode;
1680
1681 return aggregated;
