22void BufferRecyclingManager::track_buffer_usage(size_t buffer_size,
23 std::chrono::steady_clock::time_point access_time) {
24 std::lock_guard<std::mutex> lock(stats_mutex_);
25
26 auto& stats = usage_stats_[buffer_size];
27 if (stats.buffer_size == 0) {
28 stats.buffer_size = buffer_size;
29 stats.last_access = access_time;
30 }
31
32 stats.total_accesses++;
33 stats.recent_accesses++;
34
35 // Calculate access frequency (accesses per minute)
36 auto time_diff = access_time - stats.last_access;
37 auto minutes_diff = std::chrono::duration_cast<std::chrono::minutes>(time_diff).count();
38 if (minutes_diff > 0) {
39 stats.access_frequency = static_cast<double>(stats.recent_accesses) / minutes_diff;
40 stats.is_trending_up = stats.access_frequency > (stats.total_accesses /
41 std::max(1.0, static_cast<double>(std::chrono::duration_cast<std::chrono::minutes>(
42 access_time.time_since_epoch()).count())));
43 }
44
45 stats.last_access = access_time;
46 update_usage_stats(buffer_size);
47}

75

92size_t BufferRecyclingManager::get_optimal_pool_size(size_t buffer_size) const {
93 std::lock_guard<std::mutex> lock(stats_mutex_);
94
95 auto it = usage_stats_.find(buffer_size);
96 if (it == usage_stats_.end()) {
97 return 16; // Default pool size
98 }
99
100 const auto& stats = it->second;
101
102 // Base pool size on recent access frequency and utilization score
103 double base_size = std::max(4.0, stats.access_frequency * 2.0);

137void BufferRecyclingManager::optimize_pools_based_on_usage() {
138 if (!should_optimize_pools()) {
139 return;
140 }
141
142 auto& pool_manager = GlobalPoolManager::instance();
143 size_t pools_optimized = 0;
144
145 std::lock_guard<std::mutex> lock(stats_mutex_);
146
147 for (const auto& [buffer_size, stats] : usage_stats_) {
148 if (should_recycle_buffer(buffer_size)) {
149 size_t optimal_size = get_optimal_pool_size(buffer_size);
150
151 // Check if pool exists and needs adjustment
152 auto pool_result = pool_manager.create_pool(buffer_size, optimal_size);
153 if (pool_result || pool_result.error() == DTLSError::ALREADY_INITIALIZED) {
154 auto& pool = pool_manager.get_pool(buffer_size);
155
156 // Adjust pool size based on utilization
157 if (stats.utilization_score > 0.8 && pool.total_buffers() < optimal_size) {
158 pool.expand_pool(optimal_size - pool.total_buffers());
159 pools_optimized++;
160 } else if (stats.utilization_score < 0.2 && pool.total_buffers() > 4) {
161 pool.shrink_pool(std::max(static_cast<size_t>(4), optimal_size));
162 pools_optimized++;
163 }
164 }
165 }
166 }
167
168 // Update statistics
169 {
170 std::lock_guard<std::mutex> stats_lock(recycling_stats_mutex_);
171 recycling_stats_.pools_optimized += pools_optimized;
172 recycling_stats_.last_optimization = std::chrono::steady_clock::now();
173
174 // Calculate average utilization
175 if (!usage_stats_.empty()) {
176 double total_utilization = 0.0;
177 for (const auto& [size, stats] : usage_stats_) {
178 total_utilization += stats.utilization_score;
179 }
180 recycling_stats_.average_utilization = total_utilization / usage_stats_.size();
181 }
182 }
183
184 // Cleanup old statistics
185 cleanup_old_stats();
186}

239double BufferRecyclingManager::calculate_utilization_score(const BufferUsageStats& stats) const {
240 if (stats.total_accesses == 0) {
241 return 0.0;
242 }
243
244 // Combine multiple factors into utilization score
245 double frequency_score = std::min(1.0, stats.access_frequency / 10.0); // Normalize to [0,1]
246 double recency_score = 1.0;
247
248 auto now = std::chrono::steady_clock::now();
249 auto time_since_access = now - stats.last_access;
250 auto minutes_since = std::chrono::duration_cast<std::chrono::minutes>(time_since_access).count();
251
252 if (minutes_since > 0) {
253 // Exponential decay based on recency
254 recency_score = std::exp(-minutes_since / 60.0); // Decay over 1 hour
255 }
256
257 double lifetime_score = 0.5; // Default
258 if (stats.average_lifetime.count() > 0) {
259 // Longer lived buffers are more valuable for pooling
260 lifetime_score = std::min(1.0, stats.average_lifetime.count() / 60000.0); // Normalize to 1 minute
261 }
262
263 double trending_bonus = stats.is_trending_up ? 0.2 : 0.0;
264
265 // Weighted combination
266 return (0.4 * frequency_score + 0.3 * recency_score + 0.2 * lifetime_score + trending_bonus);
267}

464size_t SmartBufferFactory::round_to_optimal_size(size_t size) const {
465 // Common optimal sizes for DTLS operations
466 static const std::vector<size_t> optimal_sizes = {
467 32, 64, 128, 256, 512, 1024, 2048, 4096, 8192, 16384
468 };
469
470 // Find the smallest optimal size that's >= requested size
471 auto it = std::lower_bound(optimal_sizes.begin(), optimal_sizes.end(), size);
472
473 if (it != optimal_sizes.end()) {
474 return *it;
475 }

477 // For very large sizes, round up to nearest multiple of 4KB
