270struct Alert {
271 std::string id;
272 AlertCondition condition;
273 std::chrono::steady_clock::time_point triggered_time;
274 std::chrono::steady_clock::time_point resolved_time;
275 double actual_value;

277 bool active = true;

280 bool is_active() const { return active; }
