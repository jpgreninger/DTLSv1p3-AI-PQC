293struct CryptoOperationStats {
294 std::string operation_name;
295 size_t call_count{0};
296 std::chrono::nanoseconds total_time{0};
297 std::chrono::nanoseconds min_time{std::chrono::nanoseconds::max()};
298 std::chrono::nanoseconds max_time{0};
299 std::chrono::nanoseconds average_time() const {
300 return call_count > 0 ? std::chrono::duration_cast<std::chrono::nanoseconds>(total_time / call_count) : std::chrono::nanoseconds{0};
301 }
