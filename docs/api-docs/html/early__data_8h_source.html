24// Session ticket for resumption and early data
25struct SessionTicket {
26 std::vector<uint8_t> ticket_data; // Encrypted ticket content
27 std::vector<uint8_t> ticket_nonce; // Unique nonce for this ticket
28 uint32_t ticket_lifetime; // Lifetime in seconds
29 uint32_t ticket_age_add; // Age obfuscation value
30 uint32_t max_early_data_size; // Max early data for this ticket
31 std::chrono::steady_clock::time_point issued_time; // When ticket was issued
32
33 // Cryptographic state
34 std::vector<uint8_t> resumption_master_secret; // For key derivation
35 CipherSuite cipher_suite; // Cipher suite for this session
36
37 SessionTicket() : ticket_lifetime(0), ticket_age_add(0), max_early_data_size(0) {}
38
39 bool is_valid() const {
40 auto now = std::chrono::steady_clock::now();
41 auto elapsed = std::chrono::duration_cast<std::chrono::seconds>(now - issued_time);
42 return elapsed.count() < ticket_lifetime && !ticket_data.empty();
43 }
44
45 uint32_t get_obfuscated_age() const {
46 auto now = std::chrono::steady_clock::now();
47 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(now - issued_time);
48 return static_cast<uint32_t>(elapsed.count()) + ticket_age_add;
49 }
50};
