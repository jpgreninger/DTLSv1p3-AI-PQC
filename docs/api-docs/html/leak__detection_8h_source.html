179// RAII resource tracker for automatic tracking
180template<typename T>
181class DTLS_API ResourceTracker {
182public:
183 ResourceTracker(T* resource, ResourceType type, size_t size,
184 const std::string& allocation_site, const std::string& description = "")
185 : resource_(resource), tracked_(resource != nullptr) {
186 if (tracked_) {
187 LeakDetector::instance().track_resource(resource, type, size, allocation_site, description);
188 }
189 }
190
191 ~ResourceTracker() {
192 if (tracked_ && resource_) {
193 LeakDetector::instance().untrack_resource(resource_);
194 }
195 }
196
197 // Non-copyable, movable
198 ResourceTracker(const ResourceTracker&) = delete;
199 ResourceTracker& operator=(const ResourceTracker&) = delete;
200
201 ResourceTracker(ResourceTracker&& other) noexcept
202 : resource_(other.resource_), tracked_(other.tracked_) {
203 other.resource_ = nullptr;
204 other.tracked_ = false;
205 }
206
207 ResourceTracker& operator=(ResourceTracker&& other) noexcept {
208 if (this != &other) {
209 if (tracked_ && resource_) {
210 LeakDetector::instance().untrack_resource(resource_);
211 }
212 resource_ = other.resource_;
213 tracked_ = other.tracked_;
214 other.resource_ = nullptr;
215 other.tracked_ = false;
216 }
217 return *this;
218 }
219
220 T* get() const { return resource_; }
221 T* operator->() const { return resource_; }
222 T& operator*() const { return *resource_; }
223
224 void update_access() {
225 if (tracked_ && resource_) {
226 LeakDetector::instance().update_resource_access(resource_);
227 }
228 }
229
230 T* release() {
231 if (tracked_ && resource_) {
232 LeakDetector::instance().untrack_resource(resource_);
233 tracked_ = false;
234 }
235 T* result = resource_;
236 resource_ = nullptr;
237 return result;
238 }
239
240private:
241 T* resource_;
242 bool tracked_;
243};
