34enum class RecoveryStrategy : uint8_t {
35 NONE = 0, // No recovery attempted
36 RETRY_IMMEDIATE, // Immediate retry without delay
37 RETRY_WITH_BACKOFF, // Retry with exponential backoff
38 GRACEFUL_DEGRADATION, // Continue with reduced functionality
39 RESET_CONNECTION, // Reset connection state and retry
40 FAILOVER, // Switch to alternative provider/configuration
41 ABORT_CONNECTION // Terminate connection gracefully
42};
