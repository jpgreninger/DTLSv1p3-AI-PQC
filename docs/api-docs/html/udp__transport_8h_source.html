48struct NetworkEndpoint {
49 std::string address;
50 uint16_t port;
51 NetworkAddress::Family family;
52
53 NetworkEndpoint() : port(0), family(NetworkAddress::Family::IPv4) {}
54 NetworkEndpoint(const std::string& addr, uint16_t p, NetworkAddress::Family f = NetworkAddress::Family::IPv4)
55 : address(addr), port(p), family(f) {}
56
57 bool operator==(const NetworkEndpoint& other) const {
58 return address == other.address && port == other.port && family == other.family;
59 }
60
61 bool operator!=(const NetworkEndpoint& other) const {
62 return !(*this == other);
63 }
64
65 std::string to_string() const {
66 return (family == NetworkAddress::Family::IPv6) ?
67 "[" + address + "]:" + std::to_string(port) :
68 address + ":" + std::to_string(port);
69 }
70};
